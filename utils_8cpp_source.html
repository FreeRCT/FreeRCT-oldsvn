88 static int ReadNumber(const char **p)
89 {
90  int num = 0;
91  bool seen_first = false;
92 
93  while (**p >= '0' && **p <= '9') {
94  num = num * 10 + **p - '0';

108 HorVert ParameterizedName::DecodeName(const char *name, const Position &pos)
109 {
110  const char *p = name;
111 
112  delete[] this->name;
113  this->name = nullptr;
114  delete[] this->variant;
115  this->variant = nullptr;
116 
117  this->result = HV_NONE;
118  while (*p) {
119  if (*p != '{') {
120  p++;
121  continue;
122  }
123 
124  /* Check which type of parameter is used in the name. */
125  ParameterizedNameRange *range;
126  const char *range_type;
127  if (strncmp(p, "{hor(", 5) == 0) {
128  this->result = AddHor(this->result);
129  range = &this->hor_range;
130  range->offset = p - name;
131  p += 5;
132  range_type = "hor";
133  } else {
134  assert(strncmp(p, "{vert(", 6) == 0);
135  this->result = AddVert(this->result);
136  range = &this->vert_range;
137  range->offset = p - name;
138  p += 6;
139  range_type = "vert";
140  }
141 
142  /* Process range. */
143  if (range->used) {
144  fprintf(stderr, "Error at line %d: A \"%s\" range is used more than one time in name \"%s\".", pos.line, range_type, name);
145  exit(1);
146  }
147 
148  range->min_value = ReadNumber(&p);
149  assert(p[0] == '.' && p[1] == '.');
150  p += 2;
151  range->max_value = ReadNumber(&p);
152  assert(p[0] == ')' && p[1] == '}');
153  p += 2;
154  if (range->min_value > range->max_value) {
155  fprintf(stderr, "Error at line %d: A \"%s\" range runs from %d to %d in name \"%s\", which is an empty range.", pos.line, range_type, range->min_value, range->max_value, name);
156  exit(1);
157  }
158  range->used = true;
159  range->length = (p - name) - range->offset;
160  continue;
161  }
162  /* Copy name. */
163  this->name = new char[(p - name) + 1];
164  this->variant = new char[(p - name) + 1];
165  memcpy(this->name, name, (p - name) + 1);
166 
167  return this->result;
168 }

176 const char *ParameterizedName::GetParmName(int row, int col)
177 {
178  assert(this->variant != nullptr);
179  char *dest = this->variant;
180  int i = 0;
181  while (this->name[i] != '\0') {
182  if (this->hor_range.used && this->hor_range.offset == i) {
183  dest += sprintf(dest, "%d", this->hor_range.min_value + col);
184  i = this->hor_range.offset + this->hor_range.length;
185  continue;
186  }
187  if (this->vert_range.used && this->vert_range.offset == i) {
188  dest += sprintf(dest, "%d", this->vert_range.min_value + row);
189  i = this->vert_range.offset + this->vert_range.length;
190  continue;
191  }
192  *dest++ = this->name[i++];
193  }
194  *dest = '\0';
195  assert(dest - this->variant <= i); // No buffer overflow.
196  return this->variant;
197 }
