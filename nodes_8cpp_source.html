194 void FilePattern::SetFilename(const std::string &fname)
195 {
196  this->prefix = fname;
197  this->length = 0; // Default, just the filename.
198 
199  const char *txt = fname.c_str();
200  const char *p = txt;
201 
202  p = SkipToCurOpen(p); if (p == nullptr) return;
203 
204  const char *q = p + 1;
205  q = SkipSeq(q); if (q == nullptr) return;
206  q = SkipChar(q, '('); if (q == nullptr) return;
207 
208  int fnum;
209  q = SkipNumber(q, &fnum); if (q == nullptr) return;
210  q = SkipDotDot(q); if (q == nullptr) return;
211  int flast;
212  q = SkipNumber(q, &flast); if (q == nullptr) return;
213 
214  q = SkipChar(q, ','); if (q == nullptr) return;
215  int flen;
216  q = SkipNumber(q, &flen); if (q == nullptr) return;
217  q = SkipChar(q, ')'); if (q == nullptr) return;
218  q = SkipChar(q, '}'); if (q == nullptr) return;
219 
220  if (flast < fnum || flen <= 0 || flen > 4) return;
221 
222  this->prefix = fname.substr(0, p - txt);
223  this->suffix = fname.substr(q - txt);
224  this->first = fnum;
225  this->last = flast;
226  this->length = flen;
227 }

1455 void TrackPieceNode::ComputeTrackLength(const Position &pos)
1456 {
1457  /* Find a spline curve. At least one position should have one. */
1458  std::shared_ptr<CubicSplines> splines = nullptr;
1459  if (splines == nullptr) splines = std::dynamic_pointer_cast<CubicSplines>(this->car_xpos);
1460  if (splines == nullptr) splines = std::dynamic_pointer_cast<CubicSplines>(this->car_ypos);
1461  if (splines == nullptr) splines = std::dynamic_pointer_cast<CubicSplines>(this->car_zpos);
1462  if (splines == nullptr) {
1463  fprintf(stderr, "Error at %s: Track piece must have at least one 'splines' node in its positions.\n", pos.ToString());
1464  ::exit(1);
1465  }
1466 
1467  /* Check splines count. All splines should have the same number of bezier curves. */
1468  this->CheckBezierCount(splines->curve.size(), pos);
1469 
1470  double distance = 0.0; // Cumulative distance of this track piece.
1471  for (int index = 0; index < (int)splines->curve.size(); index++) {
1472  this->SetStartEnd(index, true, distance * 256); // Set start position of all splines at this index.
1473 
1474  int steps = splines->curve[index]->steps;
1475  this->CheckSplineStepCount(index, steps, pos); // Check step count for all splines at this index.
1476 
1477  /* Compute distance of this bezier spline. */
1478  double xp = this->car_xpos->GetValue(index, 0);
1479  double yp = this->car_ypos->GetValue(index, 0);
1480  double zp = this->car_zpos->GetValue(index, 0);
1481 
1482  for (int step = 0; step < steps; step++) {
1483  double xp2 = this->car_xpos->GetValue(index, step);
1484  double yp2 = this->car_ypos->GetValue(index, step);
1485  double zp2 = this->car_zpos->GetValue(index, step);
1486  if (xp2 == xp && yp2 == yp && zp2 == zp) continue;
1487  distance += sqrt((xp2 - xp) * (xp2 - xp) + (yp2 - yp) * (yp2 - yp) + (zp2 - zp) * (zp2 - zp));
1488  xp = xp2;
1489  yp = yp2;
1490  zp = zp2;
1491  }
1492 
1493  this->SetStartEnd(index, false, distance * 256); // Set last position of all splines at this index.
1494  }
1495  if (distance < 127.5) { // A vertical track of one voxel is 128 pixels high, with 0.5 for rounding errors to prevent false positives.
1496  fprintf(stderr, "Error at %s: Track piece is too short (should be at least 128 pixels long).\n", pos.ToString());
1497  ::exit(1);
1498  }
1499 
1500  this->total_distance = distance * 256; // Set total distance of the track piece.
1501 }
