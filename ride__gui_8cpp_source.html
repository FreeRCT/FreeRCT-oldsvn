232 void RideSelectGui::OnClick(WidgetNumber wid_num, const Point16 &pos)
233 {
234  switch (wid_num) {
235  case RSEL_SHOPS:
236  case RSEL_GENTLE:
237  case RSEL_WET:
238  case RSEL_COASTER:
239  if (this->SetNewRideKind(wid_num - RSEL_SHOPS)) this->MarkDirty();
240  break;
241 
242  case RSEL_LIST: {
243  const ScrollbarWidget *sb = this->GetWidget<ScrollbarWidget>(RSEL_SCROLL_LIST);
244  this->SetNewRide(sb->GetClickedRow(pos));
245  this->MarkDirty();
246  break;
247  }
248 
249  case RSEL_SELECT: {
250  if (this->ride_types[this->current_kind] == 0) return;
251 
252  if (this->current_kind == RTK_SHOP) {
253  bool pressed = this->IsWidgetPressed(wid_num);
254  if (pressed) {
255  _shop_placer.SetSelection(-1);
256  pressed = false;
257  } else {
258  pressed = _shop_placer.SetSelection(this->current_ride);
259  }
260  this->SetWidgetPressed(wid_num, pressed);
261  } else {
262  if (this->current_ride != -1) {
263  const RideType *ride_type = _rides_manager.GetRideType(this->current_ride);
264  assert(ride_type != nullptr && ride_type->kind == RTK_COASTER);
265  uint16 instance = _rides_manager.GetFreeInstance(ride_type);
266  if (instance != INVALID_RIDE_INSTANCE) {
267  RideInstance *ri = _rides_manager.CreateInstance(ride_type, instance);
268  _rides_manager.NewInstanceAdded(instance);
269  ShowCoasterManagementGui(ri);
270  }
271  }
272  }
273  break;
274  }
275  case RSEL_ROT_POS:
276  _shop_placer.orientation = (TileEdge)((_shop_placer.orientation + 3) & 3);
277  this->MarkWidgetDirty(RSEL_DISPLAY);
278  _shop_placer.Rotated();
279  break;
280

282  _shop_placer.orientation = (TileEdge)((_shop_placer.orientation + 1) & 3);

284  _shop_placer.Rotated();
285  break;
286  }

319 void RideSelectGui::SetNewRide(int number)
320 {
321  this->current_ride = -1;
322  number = std::min(number, this->ride_types[this->current_kind] - 1);
323  if (this->ride_types[this->current_kind] > 0) {

372 bool ShopPlacementManager::CanPlaceShop(const ShopType *selected_shop, const XYZPoint16 &pos)
373 {
374  /* 1. Can the position itself be used to build a shop? */
375  if (_world.GetTileOwner(pos.x, pos.y) != OWN_PARK) return false;
376  const Voxel *vx = _world.GetVoxel(pos);
377  if (vx != nullptr) {
378  if (!vx->CanPlaceInstance()) return false; // Cannot build on a path or other ride.
379  return vx->GetGroundType() != GTP_INVALID && vx->GetGroundSlope() == SL_FLAT; // Can build at a flat surface.
380  }
381 
382  /* 2. Is the shop just above non-flat ground? */
383  if (pos.z > 0) {
384  vx = _world.GetVoxel(pos + XYZPoint16(0, 0, -1));
385  if (vx != nullptr && vx->GetInstance() == SRI_FREE &&
386  vx->GetGroundType() != GTP_INVALID && vx->GetGroundSlope() != SL_FLAT) return true;
387  }
388 
389  /* 3. Is there a path at the right place? */
390  Viewport *vp = GetViewport();
391  for (TileEdge entrance = EDGE_BEGIN; entrance < EDGE_COUNT; entrance++) { // Loop over the 4 unrotated directions.
392  if ((selected_shop->flags & (1 << entrance)) == 0) continue; // No entrance here.
393  TileEdge entr = (TileEdge)((entrance + vp->orientation + this->orientation) & 3); // Perform rotation specified by the user in the GUI.
394  if (PathExistsAtBottomEdge(pos, entr)) return true;
395  }
396  return false;
397 }

405 RidePlacementResult ShopPlacementManager::ComputeShopVoxel(XYZPoint32 world_pos)
406 {
407  ShopInstance *si = static_cast<ShopInstance *>(_rides_manager.GetRideInstance(this->instance));
408  assert(si != nullptr && si->GetKind() == RTK_SHOP); // It should be possible to set the position of a shop.
409  const ShopType *st = si->GetShopType();
410  assert(st != nullptr);
411 
412  Viewport *vp = GetViewport();
413  if (vp == nullptr) return RPR_FAIL;
414 
415  int dx, dy; // Change of xworld and yworld for every (zworld / 2) change.
416  switch (vp->orientation) {
417  case VOR_NORTH: dx = 1; dy = 1; break;
418  case VOR_WEST: dx = -1; dy = 1; break;
419  case VOR_SOUTH: dx = -1; dy = -1; break;
420  case VOR_EAST: dx = 1; dy = -1; break;
421  default: NOT_REACHED();
422  }
423 
424  XYZPoint16 vox_pos;
425  /* Move to the top voxel of the world. */
426  vox_pos.z = WORLD_Z_SIZE - 1;
427  int dz = vox_pos.z * 256 - world_pos.z;
428  world_pos.x += dx * dz / 2;
429  world_pos.y += dy * dz / 2;
430 
431  while (vox_pos.z >= 0) {
432  vox_pos.x = world_pos.x / 256;
433  vox_pos.y = world_pos.y / 256;
434  if (IsVoxelstackInsideWorld(vox_pos.x, vox_pos.y) && this->CanPlaceShop(st, vox_pos)) {
435  /* Position of the shop the same as previously? */
436  if (si->vox_pos != vox_pos || si->orientation != this->orientation) {
437  si->SetRide((this->orientation + vp->orientation) & 3, vox_pos);
438  return RPR_CHANGED;
439  }
440  return RPR_SAMEPOS;
441  } else {
442  /* Since z gets smaller, we subtract dx and dy, thus the checks reverse. */
443  if (vox_pos.x < 0 && dx > 0) break;
444  if (vox_pos.x >= _world.GetXSize() && dx < 0) break;
445  if (vox_pos.y < 0 && dy > 0) break;
446  if (vox_pos.y >= _world.GetYSize() && dy < 0) break;
447  }
448  world_pos.x -= 128 * dx;
449  world_pos.y -= 128 * dy;
450  vox_pos.z--;
451  }
452  return RPR_FAIL;
453 }

460 void ShopPlacementManager::PlaceShop(const Point16 &pos)
461 {
462  Viewport *vp = GetViewport();
463  Point32 wxy = vp->ComputeHorizontalTranslation(vp->rect.width / 2 - pos.x, vp->rect.height / 2 - pos.y);
464 
465  /* Clean current display if needed. */
466  switch (this->ComputeShopVoxel(XYZPoint32(wxy.x, wxy.y, vp->view_pos.z))) {
467  case RPR_FAIL:
468  if (this->state == SPS_BAD_POS) return; // Nothing to do.
469  _additions.MarkDirty(vp);
470  _additions.Clear();
471  this->state = SPS_BAD_POS;
472  return;
473 
474  case RPR_SAMEPOS:
475  if (this->state == SPS_GOOD_POS) return; // Already placed and shown.
476  /* FALL-THROUGH */
477 
478  case RPR_CHANGED: {
479  _additions.MarkDirty(vp);
480  _additions.Clear();
481 
483  ShopInstance *si = static_cast<ShopInstance *>(_rides_manager.GetRideInstance(this->instance));
484  assert(si != nullptr && si->GetKind() == RTK_SHOP);
485  Voxel *vx = _additions.GetCreateVoxel(si->vox_pos, true);
486  assert(this->instance >= SRI_FULL_RIDES && this->instance <= SRI_LAST);
487  vx->SetInstance((SmallRideInstance)this->instance);
488  uint8 entrances = si->GetEntranceDirections(si->vox_pos);
489  vx->SetInstanceData(entrances);
490  AddRemovePathEdges(si->vox_pos, PATH_EMPTY, entrances, true, PAS_QUEUE_PATH);
491  _additions.MarkDirty(vp);
492  vp->EnsureAdditionsAreVisible();
493  this->state = SPS_GOOD_POS;
494  return;
495  }
496 
497  default: NOT_REACHED();
498  }
