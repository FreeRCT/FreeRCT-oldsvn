148 Point32 ComputeInitialPosition::FindPosition(Window *new_w)
149 {
150  static const Point16 test_positions[] = {
151  {0, 1}, {0, 2}, {3, 1}, {3, 2},
152  {1, 0}, {2, 0}, {1, 3}, {2, 3}
153  };
154 
155  int32 x[4], y[4];
156 
157  this->skip = new_w;
158 
159  Point32 best(this->base_pos, this->base_pos);
160 
161  bool found_empty = false;
162  for (Window *w = _window_manager.top; w != nullptr; w = w->lower) {
163  if (w == new_w) continue;
164  x[0] = w->rect.base.x - new_w->rect.width - GAP;
165  x[1] = w->rect.base.x;
166  x[2] = w->rect.base.x + w->rect.width - new_w->rect.width;
167  x[3] = w->rect.base.x + w->rect.width + GAP;
168 
169  y[0] = w->rect.base.y - new_w->rect.height - GAP;
170  y[1] = w->rect.base.y;
171  y[2] = w->rect.base.y + w->rect.height - new_w->rect.height;
172  y[3] = w->rect.base.y + w->rect.height + GAP;
173 
174  for (uint i = 0; i < lengthof(test_positions); i++) {
175  Point32 pt(x[test_positions[i].x], y[test_positions[i].y]);
176  if (this->IsScreenEmpty(Rectangle32(pt.x, pt.y, new_w->rect.width, new_w->rect.height))) {
177  if (!found_empty || GetDistanceToMouse(best) > GetDistanceToMouse(pt)) {
178  best = pt;
179  found_empty = true;
180  }
181  }
182  }
183  }
184 
185  if (best.x == this->base_pos) {
186  this->base_pos += 10;
187  if (this->base_pos + 100 > _video.GetYSize()) this->base_pos = 10;
188  }
189 
190  return best;
191 }

198 bool ComputeInitialPosition::IsScreenEmpty(const Rectangle32 &rect)
199 {
200  uint new_prio = GetWindowZPriority(this->skip->wtype);
201 
202  if (rect.base.x < 0 || rect.base.y < 0
203  || rect.base.x + rect.width > _video.GetXSize()
204  || rect.base.y + rect.height > _video.GetYSize()) return false;
205 
206  for (Window *w = _window_manager.top; w != nullptr; w = w->lower) {

695 static uint GetWindowZPriority(WindowTypes wt)
696 {
697  switch (wt) {
698  case WC_DROPDOWN: return 12; // Dropdown menus.
699  case WC_ERROR_MESSAGE: return 10; // Error messages.
700  default: return 5; // 'Normal' window.
701  case WC_TOOLBAR: return 1; // Top toolbar.

704  }

712 void WindowManager::AddToStack(Window *w)
713 {
714  assert(w->lower == nullptr && w->higher == nullptr);
715  assert(!this->HasWindow(w));
716 
717  uint w_prio = GetWindowZPriority(w->wtype);
718  if (this->top == nullptr || w_prio >= GetWindowZPriority(this->top->wtype)) {
719  /* Add to the top. */
720  w->lower = this->top;
721  w->higher = nullptr;
722  if (this->top != nullptr) this->top->higher = w;
723  this->top = w;
724  if (this->bottom == nullptr) this->bottom = w;
725  return;
726  }
727  Window *stack = this->top;
728  while (stack->lower != nullptr && w_prio < GetWindowZPriority(stack->lower->wtype)) stack = stack->lower;
729 
730  w->lower = stack->lower;
731  if (stack->lower != nullptr) {
732  stack->lower->higher = w;
733  } else {
734  assert(this->bottom == stack);
735  this->bottom = w;
736  }
737  w->higher = stack;
738  stack->lower = w;
739 }

833 void WindowManager::MouseMoveEvent(const Point16 &pos)
834 {
835  if (pos == this->mouse_pos) return;
836  this->mouse_pos = pos;
837 
838  switch (this->mouse_mode) {
839  case WMMM_PASS_THROUGH: {
840  this->UpdateCurrentWindow();
841  if (this->current_window == nullptr) {
842  return;
843  }
844 
845  /* Compute position relative to window origin. */
846  Point16 pos2;
847  pos2.x = pos.x - this->current_window->rect.base.x;
848  pos2.y = pos.y - this->current_window->rect.base.y;
849  this->current_window->OnMouseMoveEvent(pos2);
850  break;
851  }
852 
853  case WMMM_MOVE_WINDOW: {
854  if ((this->mouse_state & MB_LEFT) != MB_LEFT) {
855  this->mouse_mode = WMMM_PASS_THROUGH;
856  return;
857  }
858  this->current_window->MarkDirty();
859  assert(this->current_window->wtype != WC_MAINDISPLAY); // Cannot move the main display!
860  this->current_window->SetPosition(pos.x - this->move_offset.x, pos.y - this->move_offset.y);
861  this->current_window->MarkDirty();
862  break;
863  }
864 
865  default:
866  NOT_REACHED();
867  }
868 }

912 void WindowManager::MouseButtonEvent(MouseButtons button, bool pressed)
913 {
914  assert(button == MB_LEFT || button == MB_MIDDLE || button == MB_RIGHT);
915  uint8 newstate = this->mouse_state;
916  if (pressed) {
917  newstate |= button;
918  } else {
919  newstate &= ~button;
920  }
921 
922  this->UpdateCurrentWindow();
923  if (this->current_window == nullptr) {
924  this->mouse_mode = WMMM_PASS_THROUGH;
925  this->mouse_state = newstate;
926  return;
927  }
928 
929  /* Close dropdown window if click is not inside it */
930  Window *w = GetWindowByType(WC_DROPDOWN, ALL_WINDOWS_OF_TYPE);
931  if (pressed && w != nullptr && !w->rect.IsPointInside(this->mouse_pos)) {
932  this->DeleteWindow(w);
933  return; // Don't handle click any further.
934  }
935 
936  if (button == MB_LEFT && pressed) this->RaiseWindow(this->current_window);
937 
938  switch (this->mouse_mode) {
939  case WMMM_PASS_THROUGH:
940  if (newstate != this->mouse_state) {
941  WmMouseEvent me = this->current_window->OnMouseButtonEvent((this->mouse_state << 4) | newstate);
942  switch (me) {
943  case WMME_NONE:
944  break;
945 
946  case WMME_MOVE_WINDOW:
947  this->StartWindowMove();
948  break;
949 
950  case WMME_CLOSE_WINDOW:
951  this->DeleteWindow(this->current_window);
952  break;
953 
954  default:
955  NOT_REACHED();
956  }
957  }
958  break;
959 
960  case WMMM_MOVE_WINDOW:
961  this->mouse_mode = WMMM_PASS_THROUGH; // Mouse clicks stop window movement.
962  break;
963 
964  default:
965  NOT_REACHED();
966  }
967  this->mouse_state = newstate;
968 }
