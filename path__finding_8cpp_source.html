100 void PathSearcher::AddOpen(const XYZPoint16 &vox, uint32 traveled, const WalkedPosition *prev_pos)
101 {
102  uint32 estimate = this->GetEstimate(vox);
103  WalkedPosition wp(vox, traveled, estimate, prev_pos);
104 
105  /* Find the position. */
106  PositionSet::iterator pos_iter = this->positions.find(wp);
107  if (pos_iter == positions.end()) { // New position.
108  pos_iter = this->positions.insert(wp).first;
109  this->open_points.emplace(traveled, estimate, &(*pos_iter));
110  return;
111  }
112  /* Existing position, update if needed. */
113  const WalkedPosition *pwp = &(*pos_iter);
114  if (pwp->traveled + pwp->estimate <= traveled + estimate) return;
115 
116  /* New one is better, update. */
117  pwp->traveled = traveled;
118  pwp->estimate = estimate; // The sum is changed, making any old open points invalid.
119  pwp->prev_pos = prev_pos;
120  this->open_points.emplace(traveled, estimate, pwp);
121 }

127 bool PathSearcher::Search()
128 {
129  this->dest_pos = nullptr;
130  while (!open_points.empty()) {
131  WalkedDistance wd = *open_points.begin();
132  open_points.erase(open_points.begin());
133 
134  if (wd.traveled != wd.pos->traveled || wd.estimate != wd.pos->estimate) continue; // Invalid open point.
135 
136  /* Reached the destination? */
137  const WalkedPosition *wp = wd.pos;
138  if (wp->cur_vox == this->dest_vox) {
139  this->dest_pos = wp;
140  return true;
141  }
142 
143  /* Add new open points. */
144  const Voxel *v = _world.GetVoxel(wp->cur_vox);
145  if (v == nullptr) continue; // No voxel at the expected point, don't bother.
146 
147  uint8 exits = GetPathExits(v);
148  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
149  if ((exits & (0x11 << edge)) == 0) continue;
150 
151  /* There is an outgoing connection, is it also on the world? */
152  Point16 dxy = _tile_dxy[edge];
153  if (dxy.x < 0 && wp->cur_vox.x == 0) continue;
154  if (dxy.x > 0 && wp->cur_vox.x + 1 == _world.GetXSize()) continue;
155  if (dxy.y < 0 && wp->cur_vox.y == 0) continue;
156  if (dxy.y > 0 && wp->cur_vox.y + 1 == _world.GetYSize()) continue;
157 
158  int extra_z = ((exits & (0x10 << edge)) != 0);
159  if (wp->cur_vox.z + extra_z < 0 || wp->cur_vox.z + extra_z >= WORLD_Z_SIZE) continue;
160 
161  /* Now check the other side, new_z is the voxel where the path should be at the bottom. */
162  const Voxel *v2 = _world.GetVoxel(wp->cur_vox + XYZPoint16(dxy.x, dxy.y, extra_z));
163  if (v2 == nullptr) continue;
164 
165  uint8 other_exits = GetPathExits(v2);
166  if ((other_exits & (1 << ((edge + 2) % 4))) == 0) { // No path here, try one voxel below
167  extra_z--;
168  if (wp->cur_vox.z + extra_z < 0) continue;
169  v2 = _world.GetVoxel(wp->cur_vox + XYZPoint16(dxy.x, dxy.y, extra_z));
170  if (v2 == nullptr) continue;
171  other_exits = GetPathExits(v2);
172  if ((other_exits & (0x10 << ((edge + 2) % 4))) == 0) continue;
173  }
174  /* Add new open point to the path finder. */
175  this->AddOpen(wp->cur_vox + XYZPoint16(dxy.x, dxy.y, extra_z), wp->traveled + 1, wp);
176  }
177  }
178  return false;
179 }
