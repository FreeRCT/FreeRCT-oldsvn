87 bool PathExistsAtBottomEdge(XYZPoint16 voxel_pos, TileEdge edge)
88 {
89  voxel_pos.x += _tile_dxy[edge].x;
90  voxel_pos.y += _tile_dxy[edge].y;
91  if (!IsVoxelstackInsideWorld(voxel_pos.x, voxel_pos.y)) return false;
92 
93  const Voxel *vx = _world.GetVoxel(voxel_pos);
94  if (vx == nullptr || !HasValidPath(vx)) {
95  /* No path here, check the voxel below. */
96  if (voxel_pos.z == 0) return false;
97  voxel_pos.z--;
98  vx = _world.GetVoxel(voxel_pos);
99  if (vx == nullptr || !HasValidPath(vx)) return false;
100  /* Path must end at the top of the voxel. */
101  return GetImplodedPathSlope(vx) == _path_up_from_edge[edge];
102  }
103  /* Path must end at the bottom of the voxel. */
104  return GetImplodedPathSlope(vx) < PATH_FLAT_COUNT || GetImplodedPathSlope(vx) == _path_down_from_edge[edge];
105 }

115 static bool BuildUpwardPath(const XYZPoint16 &voxel_pos, TileEdge edge, PathType path_type, bool test_only)
116 {
117  /* xy position should be valid, and allow path building. */
118  if (!IsVoxelstackInsideWorld(voxel_pos.x, voxel_pos.y)) return false;
119  if (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(voxel_pos.x, voxel_pos.y) != OWN_PARK) return false;
120 
121  if (voxel_pos.z < 0 || voxel_pos.z > WORLD_Z_SIZE - 3) return false; // Z range should be valid.
122 
123  const VoxelStack *vs = _world.GetStack(voxel_pos.x, voxel_pos.y);
124  const Voxel *v = vs->Get(voxel_pos.z + 1); // Voxel above the path should be empty.
125  if (v != nullptr && !v->IsEmpty()) return false;
126 
127  v = vs->Get(voxel_pos.z + 2); // 2 voxels higher should also be empty.
128  if (v != nullptr && !v->IsEmpty()) return false;
129 
130  v = vs->Get(voxel_pos.z);
131  if (v != nullptr) {
132  if (v->GetInstance() != SRI_FREE) return false; // Voxel should have no other rides.
133  if (v->GetGroundType() != GTP_INVALID) {
134  TileSlope slope = ExpandTileSlope(v->GetGroundSlope());
135  if ((slope & (TSB_STEEP | TSB_TOP)) == TSB_STEEP) return false;
136  if ((slope & _corners_at_edge[edge]) != 0) return false; // A raised corner at 'edge'.
137  } else {
138  /* No surface, but a foundation suggests a nearby hill.
139  * Currently simply deny building here, in the future, consider making a tunnel.
140  */
141  if (v->GetFoundationType() != FDT_INVALID) return false;
142  }
143  }
144 
145  if (!test_only) BuildPathAtTile(voxel_pos, path_type, _path_up_from_edge[edge]);
146  return true;
147 }

266 static uint8 CanBuildPathFromEdge(const XYZPoint16 &voxel_pos, TileEdge edge)
267 {
268  if (!IsVoxelstackInsideWorld(voxel_pos.x, voxel_pos.y)) return 0;
269  if (voxel_pos.z < 0 || voxel_pos.z >= WORLD_Z_SIZE - 1) return 0;
270 
271  /* If other side of the edge is not on-world or not owned, don't compute path options. */
272  Point16 dxy = _tile_dxy[edge];
273  if (!IsVoxelstackInsideWorld(voxel_pos.x + dxy.x, voxel_pos.y + dxy.y) ||
274  (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(voxel_pos.x + dxy.x, voxel_pos.y + dxy.y) != OWN_PARK)) return 0;
275 
276  const Voxel *v = _world.GetVoxel(voxel_pos);
277  if (v != nullptr && HasValidPath(v)) {
278  PathSprites ps = GetImplodedPathSlope(v);
279  if (ps < PATH_FLAT_COUNT) return 1 << TSL_FLAT;
280  if (ps == _path_up_from_edge[edge]) return 1 << TSL_UP;
281  }
282  if (voxel_pos.z > 0) {
283  v = _world.GetVoxel(voxel_pos + XYZPoint16(0, 0, -1));
284  if (v != nullptr && HasValidPath(v) && GetImplodedPathSlope(v) == _path_down_from_edge[edge]) return 1 << TSL_DOWN;
285  }
286 
287  uint8 slopes = 0;
288  // XXX Check for already existing paths.
289  if (BuildDownwardPath(voxel_pos, edge, PAT_INVALID, true)) slopes |= 1 << TSL_DOWN;
290  if (BuildFlatPath(voxel_pos, PAT_INVALID, true)) slopes |= 1 << TSL_FLAT;
291  if (BuildUpwardPath(voxel_pos, edge, PAT_INVALID, true)) slopes |= 1 << TSL_UP;
292  return slopes;
293 }

301 static uint8 GetPathAttachPoints(const XYZPoint16 &voxel_pos)
302 {
303  if (!IsVoxelstackInsideWorld(voxel_pos.x, voxel_pos.y)) return 0;
304  if (voxel_pos.z >= WORLD_Z_SIZE - 1) return 0; // The voxel containing the flat path, and one above it.
305 
306  const Voxel *v = _world.GetVoxel(voxel_pos);
307  if (v == nullptr) return 0;
308 
309  uint8 edges = 0;
310  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
311  uint16 x = voxel_pos.x + _tile_dxy[edge].x;
312  uint16 y = voxel_pos.y + _tile_dxy[edge].y;
313  if (!IsVoxelstackInsideWorld(x, y)) continue;
314 
315  const XYZPoint16 new_pos(x, y, voxel_pos.z);
316  if (HasValidPath(v)) {
317  PathSprites ps = GetImplodedPathSlope(v);
318  if (ps < PATH_FLAT_COUNT) {
319  if (CanBuildPathFromEdge(new_pos, (TileEdge)((edge + 2) % 4)) != 0) edges |= 1 << edge;
320  } else {
321  const XYZPoint16 new_above(x, y, voxel_pos.z + 1);
322  if (_path_up_from_edge[edge] == ps
323  && CanBuildPathFromEdge(new_pos, (TileEdge)((edge + 2) % 4)) != 0) edges |= 1 << edge;
324  if (_path_down_from_edge[edge] == ps
325  && CanBuildPathFromEdge(new_above, (TileEdge)((edge + 2) % 4)) != 0) edges |= (1 << edge) << 4;
326  }
327  continue;
328  }
329  if (v->GetGroundType() != GTP_INVALID) {
330  TileSlope ts = ExpandTileSlope(v->GetGroundSlope());
331  if ((ts & TSB_STEEP) != 0) continue;
332  if ((ts & _corners_at_edge[edge]) == 0) {
333  if (CanBuildPathFromEdge(new_pos, (TileEdge)((edge + 2) % 4)) != 0) edges |= 1 << edge;
334  } else {
335  const XYZPoint16 new_above(x, y, voxel_pos.z + 1);
336  if (CanBuildPathFromEdge(new_above, (TileEdge)((edge + 2) % 4)) != 0) edges |= (1 << edge) << 4;
337  }
338  continue;
339  }
340  /* No path and no ground -> Invalid, do next edge. */
341  }
342  return edges;
343 }

456 void PathBuildManager::TileClicked(const XYZPoint16 &click_pos)
457 {
458  if (this->state == PBS_IDLE || this->state > PBS_WAIT_BUY) return;
459 
460  if (this->state == PBS_SINGLE) {
461  this->pos = click_pos;
462  const Voxel *v = _world.GetCreateVoxel(this->pos, false);
463  TileSlope ts = ExpandTileSlope(v->GetGroundSlope());
464 
465  if (ts == SL_FLAT) {
466  BuildFlatPath(this->pos, this->path_type, false);

473  break;
474  case TSB_NORTHWEST:
475  edge = EDGE_SE;
476  break;
477  case TSB_SOUTHEAST:
478  edge = EDGE_NW;
479  break;
480  case TSB_SOUTHWEST:
481  edge = EDGE_NE;
482  break;

484  return;
485  }
486 
487  BuildUpwardPath(this->pos, edge, this->path_type, false);
488  }
489 
490  _additions.Commit();
491  } else {
492  uint8 dirs = GetPathAttachPoints(click_pos);
493  if (dirs == 0) return;

495  this->pos = click_pos;

531 void PathBuildManager::MoveCursor(TileEdge edge, bool move_up)
532 {
533  if (this->state <= PBS_WAIT_ARROW || this->state > PBS_WAIT_BUY || edge == INVALID_EDGE) return;
534 
535  /* Test whether we can move in the indicated direction. */
536  Point16 dxy = _tile_dxy[edge];
537  if ((dxy.x < 0 && this->pos.x == 0) || (dxy.x > 0 && this->pos.x == _world.GetXSize() - 1)) return;
538  if ((dxy.y < 0 && this->pos.y == 0) || (dxy.y > 0 && this->pos.y == _world.GetYSize() - 1)) return;
539  if (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(this->pos.x + dxy.x, this->pos.y + dxy.y) != OWN_PARK) return;
540 
541  const Voxel *v_top, *v_bot;
542  if (move_up) {
543  /* Exit of current tile is at the top. */
544  v_top = (this->pos.z > WORLD_Z_SIZE - 2) ? nullptr : _world.GetVoxel(this->pos + XYZPoint16(dxy.x, dxy.y, 1));
545  v_bot = _world.GetVoxel(this->pos + XYZPoint16(dxy.x, dxy.y, 0));
546  } else {
547  /* Exit of current tile is at the bottom. */
548  v_top = _world.GetVoxel(this->pos + XYZPoint16(dxy.x, dxy.y, 0));
549  v_bot = (this->pos.z == 0) ? nullptr : _world.GetVoxel(this->pos + XYZPoint16(dxy.x, dxy.y, -1));
550  }
551 
552  /* Try to find a voxel with a path. */
553  if (v_top != nullptr && HasValidPath(v_top)) {
554  this->pos.x += dxy.x;
555  this->pos.y += dxy.y;
556  if (move_up) this->pos.z++;
557  this->SetState(PBS_WAIT_ARROW);
558  return;
559  }
560  if (v_bot != nullptr && HasValidPath(v_bot)) {
561  this->pos.x += dxy.x;
562  this->pos.y += dxy.y;
563  if (!move_up) this->pos.z--;
564  this->SetState(PBS_WAIT_ARROW);
565  return;
566  }
567 
568  /* Try to find a voxel with surface. */
569  if (v_top != nullptr && v_top->GetGroundType() != GTP_INVALID && !IsImplodedSteepSlope(v_top->GetGroundSlope())) {
570  this->pos.x += dxy.x;
571  this->pos.y += dxy.y;
572  if (move_up) this->pos.z++;
573  this->SetState(PBS_WAIT_ARROW);
574  return;
575  }
576  if (v_bot != nullptr && v_bot->GetGroundType() != GTP_INVALID && !IsImplodedSteepSlope(v_bot->GetGroundSlope())) {
577  this->pos.x += dxy.x;
578  this->pos.y += dxy.y;
579  if (!move_up) this->pos.z--;
580  this->SetState(PBS_WAIT_ARROW);
581  return;
582  }
583 }

589 void PathBuildManager::SelectMovement(bool move_forward)
590 {
591  if (this->state <= PBS_WAIT_ARROW || this->state > PBS_WAIT_BUY) return;
592 
593  TileEdge edge = (move_forward) ? this->selected_arrow : (TileEdge)((this->selected_arrow + 2) % 4);
594 
595  bool move_up;
596  const Voxel *v = _world.GetVoxel(this->pos);
597  if (v == nullptr) return;
598  if (HasValidPath(v)) {
599  move_up = (GetImplodedPathSlope(v) == _path_down_from_edge[edge]);
600  } else if (v->GetGroundType() != GTP_INVALID) {
601  TileSlope ts = ExpandTileSlope(v->GetGroundSlope());
602  if ((ts & TSB_STEEP) != 0) return;

616 {

636 {

663 void PathBuildManager::UpdateState()
664 {
665  Viewport *vp = GetViewport();
666 
667  if (this->state == PBS_IDLE || this->state == PBS_SINGLE) {
668  DisableWorldAdditions();
669  this->selected_arrow = INVALID_EDGE;
670  this->selected_slope = this->state == PBS_IDLE ? TSL_INVALID : TSL_FLAT;
671  }
672 
673  /* The tile cursor is controlled by the viewport if waiting for a voxel or earlier. */
674  if (vp != nullptr && this->state > PBS_WAIT_VOXEL && this->state <= PBS_WAIT_BUY) {
675  vp->tile_cursor.SetCursor(this->pos, CUR_TYPE_TILE);
676  }
677 
678  /* See whether the PBS_WAIT_ARROW state can be left automatically. */
679  if (this->state == PBS_WAIT_ARROW) {
680  this->allowed_arrows = GetPathAttachPoints(this->pos);
681 
682  /* If a valid selection has been made, or if only one choice exists, take it. */
683  if (this->selected_arrow != INVALID_EDGE && ((0x11 << this->selected_arrow) & this->allowed_arrows) != 0) {
684  this->state = PBS_WAIT_SLOPE;
685  } else if (this->allowed_arrows == (1 << EDGE_NE) || this->allowed_arrows == (0x10 << EDGE_NE)) {
686  this->selected_arrow = EDGE_NE;
687  this->state = PBS_WAIT_SLOPE;
688  } else if (this->allowed_arrows == (1 << EDGE_NW) || this->allowed_arrows == (0x10 << EDGE_NW)) {
689  this->selected_arrow = EDGE_NW;
690  this->state = PBS_WAIT_SLOPE;
691  } else if (this->allowed_arrows == (1 << EDGE_SE) || this->allowed_arrows == (0x10 << EDGE_SE)) {
692  this->selected_arrow = EDGE_SE;
693  this->state = PBS_WAIT_SLOPE;
694  } else if (this->allowed_arrows == (1 << EDGE_SW) || this->allowed_arrows == (0x10 << EDGE_SW)) {
695  this->selected_arrow = EDGE_SW;
696  this->state = PBS_WAIT_SLOPE;
697  }
698  }
699 
700  /* Set the arrow cursor. Note that display is controlled later. */
701  if (vp != nullptr) {
702  if (this->state > PBS_WAIT_ARROW && this->state <= PBS_WAIT_BUY) {
703  XYZPoint16 arrow_pos = this->ComputeArrowCursorPosition();
704  vp->arrow_cursor.SetCursor(arrow_pos, (CursorType)(CUR_TYPE_ARROW_NE + this->selected_arrow));
705  } else {
706  vp->arrow_cursor.SetInvalid();
707  }
708  }
709 
710  /* See whether the PBS_WAIT_SLOPE state can be left automatically. */
711  if (this->state == PBS_WAIT_SLOPE) {
712  /* Compute allowed slopes. */
713  XYZPoint16 arrow_pos = this->ComputeArrowCursorPosition();
714  this->allowed_slopes = CanBuildPathFromEdge(arrow_pos, (TileEdge)((this->selected_arrow + 2) % 4));
715 
716  /* If a valid selection has been made, or if only one choice exists, take it. */
717  if (this->selected_slope != TSL_INVALID && ((1 << this->selected_slope) & this->allowed_slopes) != 0) {
718  this->state = PBS_WAIT_BUY;
719  } else if (this->allowed_slopes == (1 << TSL_DOWN)) {
720  this->selected_slope = TSL_DOWN;
721  this->state = PBS_WAIT_BUY;
722  } else if (this->allowed_slopes == (1 << TSL_FLAT)) {
723  this->selected_slope = TSL_FLAT;
724  this->state = PBS_WAIT_BUY;
725  } else if (this->allowed_slopes == (1 << TSL_UP)) {
726  this->selected_slope = TSL_UP;
727  this->state = PBS_WAIT_BUY;
728  }
729  }
730 
731  /* Handle _additions display. */
732  if (vp != nullptr) {
733  if (this->state == PBS_SINGLE || this->state == PBS_WAIT_SLOPE) {
734  _additions.Clear();
735  vp->EnableWorldAdditions();
736  } else if (this->state == PBS_WAIT_BUY) {
737  _additions.Clear();
738 
739  this->ComputeWorldAdditions();
740  vp->EnableWorldAdditions();
741  vp->EnsureAdditionsAreVisible();
742  } else {
743  if (this->state != PBS_LONG_BUILD && this->state != PBS_LONG_BUY) vp->DisableWorldAdditions();
744  }
745  }
746 
747  NotifyChange(WC_PATH_BUILDER, ALL_WINDOWS_OF_TYPE, CHG_UPDATE_BUTTONS, 0);
748 }

811 void PathBuildManager::ComputeNewLongPath(const Point32 &mousexy)
812 {
813  static const TrackSlope slope_prios_down[] = {TSL_DOWN, TSL_FLAT, TSL_UP, TSL_INVALID}; // Order of preference when going down.
814  static const TrackSlope slope_prios_flat[] = {TSL_FLAT, TSL_UP, TSL_DOWN, TSL_INVALID}; // Order of preference when at the right height.
815  static const TrackSlope slope_prios_up[] = {TSL_UP, TSL_FLAT, TSL_DOWN, TSL_INVALID}; // Order of preference when going up.
816 
817  Viewport *vp = GetViewport();
818  if (vp == nullptr) return;
819 
820  int c1, c2, c3;
821  switch (vp->orientation) {
822  case VOR_NORTH: c1 = 1; c2 = 2; c3 = 2; break;
823  case VOR_EAST: c1 = -1; c2 = -2; c3 = 2; break;
824  case VOR_SOUTH: c1 = 1; c2 = -2; c3 = -2; break;
825  case VOR_WEST: c1 = -1; c2 = 2; c3 = -2; break;
826  default: NOT_REACHED();
827  }
828 
829  XYZPoint16 path_pos(0, 0, 0);
830  path_pos.y = this->pos.y * 256 + 128;
831  int32 lambda_y = path_pos.y - mousexy.y; // Distance to constant Y plane at current tile cursor.
832  path_pos.x = this->pos.x * 256 + 128;
833  int32 lambda_x = path_pos.x - mousexy.x; // Distance to constant X plane at current tile cursor.
834 
835  if (abs(lambda_x) < abs(lambda_y)) {
836  /* X constant. */
837  path_pos.x /= 256;
838  path_pos.y = Clamp<int32>(mousexy.y + c1 * lambda_x, 0, _world.GetYSize() * 256 - 1) / 256;
839  path_pos.z = Clamp<int32>(vp->view_pos.z + c3 * lambda_x, 0, WORLD_Z_SIZE * 256 - 1) / 256;
840  } else {
841  /* Y constant. */
842  path_pos.x = Clamp<int32>(mousexy.x + c1 * lambda_y, 0, _world.GetXSize() * 256 - 1) / 256;
843  path_pos.y /= 256;
844  path_pos.z = Clamp<int32>(vp->view_pos.z + c2 * lambda_y, 0, WORLD_Z_SIZE * 256 - 1) / 256;
845  }
846 
847  if (this->long_pos != path_pos) {
848  this->long_pos = path_pos;
849 
850  _additions.Clear();
851  path_pos = this->pos;
852  /* Find the right direction from the selected tile to the current cursor location. */
853  TileEdge direction;
854  Point16 dxy;
855  for (direction = EDGE_BEGIN; direction < EDGE_COUNT; direction++) {
856  dxy = _tile_dxy[direction];
857  if (!GoodDirection(dxy.x, path_pos.x, this->long_pos.x) || !GoodDirection(dxy.y, path_pos.y, this->long_pos.y)) continue;
858  break;
859  }
860  if (direction == EDGE_COUNT) return;
861 
862  /* 'Walk' to the cursor as long as possible. */
863  while (path_pos.x != this->long_pos.x || path_pos.y != this->long_pos.y) {
864  uint8 slopes = CanBuildPathFromEdge(path_pos, direction);
865  const TrackSlope *slope_prio;
866  /* Get order of slope preference. */
867  if (path_pos.z > this->long_pos.z) {
868  slope_prio = slope_prios_down;
869  } else if (path_pos.z == this->long_pos.z) {
870  slope_prio = slope_prios_flat;
871  } else {
872  slope_prio = slope_prios_up;
873  }
874  /* Find best slope, and take it. */
875  while (*slope_prio != TSL_INVALID && (slopes & (1 << *slope_prio)) == 0) slope_prio++;
876  if (*slope_prio == TSL_INVALID) break;
877 
878  path_pos.x += dxy.x;
879  path_pos.y += dxy.y;
880  if (*slope_prio == TSL_UP) {
881  if (!BuildUpwardPath(path_pos, static_cast<TileEdge>((direction + 2) & 3), this->path_type, false)) break;
882  path_pos.z++;
883  } else if (*slope_prio == TSL_DOWN) {
884  if (!BuildDownwardPath(path_pos, static_cast<TileEdge>((direction + 2) & 3), this->path_type, false)) break;
885  path_pos.z--;
886  } else {
887  if (!BuildFlatPath(path_pos, this->path_type, false)) break;
888  }
889 
890  }
891  vp->EnableWorldAdditions();
892  vp->EnsureAdditionsAreVisible();
893  }
894 }
