37 uint32 Loader::OpenBlock(const char *name, bool may_fail)
38 {
39  assert(strlen(name) == 4);
40 
41  if (this->fp == nullptr || this->IsFail()) return 0;
42 
43  assert(this->blk_name == nullptr);
44  this->blk_name = name;
45  int i = 0;
46  while (i < 4) {
47  uint8 val = this->GetByte();
48  if (val != this->blk_name[i]) {
49  this->PutByte(val);
50  while (i > 0) {
51  i--;
52  this->PutByte(this->blk_name[i]);
53  }
54  if (may_fail) return UINT32_MAX;
55  this->SetFailMessage("Missing block name");
56  return 0;
57  }
58  i++;
59  }
60 
61  uint32 version = this->GetLong();
62  if (version == 0 || version == UINT32_MAX) {
63  this->SetFailMessage("Incorrect version number");
64  return 0;
65  }
66  return version;
67 }
