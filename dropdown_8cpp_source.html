89 void DropdownMenuWindow::SetDropdownSize(const Point16 &pos, uint16 min_width)
90 {
91  this->size = {pos.x, pos.y, min_width, 0};
92  int max_width = this->size.width;
93  for (auto const &item : items) {
94  int w, unused;
95  _video.GetTextSize(item.str, &w, &unused);
96  max_width = std::max(max_width, w + 2);
97  this->size.height += GetTextHeight();
98  }
99  this->size.width = max_width;
100 }

102 void DropdownMenuWindow::DrawWidget(WidgetNumber wid_num, const BaseWidget *wid) const
103 {
104  if (wid_num != DD_ITEMS) return;
105 
106  int y = this->rect.base.y; // wid->pos is relative to the window
107  int it = 0;
108  for (auto const &item : this->items) {
109  if (it == this->selected_index) {
110  Rectangle32 r = {this->rect.base.x, y, static_cast<uint>(wid->pos.width - 1), static_cast<uint>(GetTextHeight())};
111  _video.FillRectangle(r, _palette[GetColourRangeBase(COL_RANGE_GREY) + 7]);
112  }
113  _video.BlitText(item.str, MakeRGBA(255, 255, 255, OPAQUE), this->rect.base.x, y, wid->pos.width);
114 
115  y += GetTextHeight();
116  it++;
117  }
118 }

128 void DropdownMenuWindow::OnClick(WidgetNumber number, const Point16 &pos)
129 {
130  if (number != DD_ITEMS) return;
131 
132  int index = this->mouse_pos.y / GetTextHeight();
133  if (index >= (int)this->items.size()) return;
134 
135  int send = this->parent_btn << 16 | index;
136  NotifyChange(this->parent_type, this->parent_num, CHG_DROPDOWN_RESULT, send);
137 
138  _window_manager.DeleteWindow(this);
139 }
