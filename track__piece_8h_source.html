105 class CubicBezier {
106 public:
107  CubicBezier(uint32 start, uint32 last, int a, int b, int c, int d);
108 
115  double GetValue(uint32 distance) const
116  {
117  assert(distance >= this->start && distance <= this->last);
118  double t = (double)(distance - this->start) / (this->last - this->start); // XXX Perhaps store the length?
119  double tt = t * t;
120  double t1 = 1 - t;
121  double tt11 = t1 * t1;
122 
123  return (tt11 * t1) * this->a + (3 * tt11 * t) * this->b + (3 * t1 * tt) * this->c + (tt * t) * this->d;
124  }
125 
126  const uint32 start;
127  const uint32 last;
128  const int a;
129  const int b;
130  const int c;
131  const int d;
132 };

135 class BezierTrackCurve : public TrackCurve {
136 public:
137  BezierTrackCurve();
138 
139  double GetValue(uint32 distance) const override
140  {
141  int end = this->curve.size();
142  if (end == 1) return this->curve.front().GetValue(distance); // Handle a common case quickly.
143 
144  /* Bisection to the right Bezier spline. */
145  int first = 0;
146  while (end > first + 1) {
147  int middle = (first + end) / 2;
148  const CubicBezier &bezier = this->curve[middle];
149  if (bezier.start > distance) {
150  end = middle;
151  } else {
152  first = middle;
153  }
154  }
155  return this->curve[first].GetValue(distance);
156  }
157 
158  std::vector<CubicBezier> curve;
159 };

237  inline TrackSlope GetSlope() const
238  {
239  int slope = GB(this->track_flags, 8, 3);
240  if ((slope & 4) != 0) slope |= ~7;
241  switch (slope + 3) {
242  case -3 + 3: return TSL_STRAIGHT_DOWN;
243  case -2 + 3: return TSL_STEEP_DOWN;
244  case -1 + 3: return TSL_DOWN;
245  case 0 + 3: return TSL_FLAT;
246  case 1 + 3: return TSL_UP;
247  case 2 + 3: return TSL_STEEP_UP;
248  case 3 + 3: return TSL_STRAIGHT_UP;
249  default: return TSL_FLAT;
250  }
