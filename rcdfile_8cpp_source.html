108 static std::string GetString(RcdFileReader &rcd_file, uint max_bytes, uint32 *remaining)
109 {
110 
111  /* There needs to be at least one byte available for reading, and there must not be an error. */
112  if (*remaining == 0 || *remaining == UINT32_MAX) {
113  *remaining = UINT32_MAX;
114  return "";
115  }
116 
117  /* Temporary storage. */
118  char buffer[512];
119  if (max_bytes >= lengthof(buffer)) max_bytes = lengthof(buffer);
120 
121  uint8 byte;
122  uint idx = 0;
123  while (idx < max_bytes && *remaining > 0) {
124  byte = rcd_file.GetUInt8();
125  buffer[idx++] = byte;
126  (*remaining)--;
127  if (byte == 0) return buffer; // Seen the nul character.
128  }
129  /* Either the string is too long or there are no bytes left, report an error. */
130  *remaining = UINT32_MAX;
131  return "";
132 }

139 const char *RcdFileCollection::ScanFileForMetaInfo(const char *fname)
140 {
141  RcdFileReader rcd_file(fname);
142  if (!rcd_file.CheckFileHeader("RCDF", 2)) return "Wrong header";
143 
144  /* Load block. */
145  if (!rcd_file.ReadBlockHeader() || (strcmp(rcd_file.name, "INFO") != 0)) {
146  /* End reached or found a non-meta block, end scanning. */
147  return "No INFO block found.";
148  }
149 
150  /* Load INFO block. */
151  if (rcd_file.version != 1) return "INFO block has wrong version";
152  uint32 remaining = rcd_file.size;
153  std::string build = GetString(rcd_file, 16, &remaining);
154  std::string name = GetString(rcd_file, 64, &remaining);
155  std::string uri = GetString(rcd_file, 128, &remaining);
156  std::string website = GetString(rcd_file, 128, &remaining);
157  std::string description = GetString(rcd_file, 512, &remaining);
158  if (remaining != 0) return "Error while reading INFO text.";
159 
160  RcdFileInfo rfi(fname, uri, build);
161  this->AddFile(rfi);
162  return nullptr; // Success.
163 }
