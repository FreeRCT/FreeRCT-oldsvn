43 static bool IsGoodEdgeRoad(int16 x, int16 y)
44 {
45  if (x < 0 || y < 0) return false;
46  int16 z = _world.GetGroundHeight(x, y);
47  const Voxel *vs = _world.GetVoxel(XYZPoint16(x, y, z));
48  return vs && HasValidPath(vs) && GetImplodedPathSlope(vs) < PATH_FLAT_COUNT;
49 }

55 static Point16 FindEdgeRoad()
56 {
57  int16 highest_x = _world.GetXSize() - 1;
58  int16 highest_y = _world.GetYSize() - 1;
59  for (int16 x = 1; x < highest_x; x++) {
60  if (IsGoodEdgeRoad(x, 0)) return {x, 0};
61  if (IsGoodEdgeRoad(x, highest_y)) return {x, highest_y};
62  }
63  for (int16 y = 1; y < highest_y; y++) {
64  if (IsGoodEdgeRoad(0, y)) return {0, y};
65  if (IsGoodEdgeRoad(highest_x, y)) return {highest_x, y};
66  }
67 
68  return {-1, -1};
69 }

146 void Guests::DoTick()
147 {
148  this->daily_frac++;
149  int end_index = std::min(this->daily_frac * GUEST_BLOCK_SIZE / TICK_COUNT_PER_DAY, GUEST_BLOCK_SIZE);
150  while (this->next_daily_index < end_index) {
151  Guest *p = this->block.Get(this->next_daily_index);
152  if (p->IsActive() && !p->DailyUpdate()) {
153  p->DeActivate(OAR_REMOVE);
154  this->AddFree(p);
155  }
156  this->next_daily_index++;
157  }
158  if (this->next_daily_index >= GUEST_BLOCK_SIZE) {
159  this->daily_frac = 0;
160  this->next_daily_index = 0;
161  }
162 }

168 void Guests::OnNewDay()
169 {
170  /* Try adding a new guest to the park. */
171  if (this->CountActiveGuests() >= _scenario.max_guests) return;
172  if (!this->rnd.Success1024(_scenario.GetSpawnProbability(512))) return;
173 
174  if (!IsGoodEdgeRoad(this->start_voxel.x, this->start_voxel.y)) {
175  /* New guest, but no road. */
176  this->start_voxel = FindEdgeRoad();
177  if (!IsGoodEdgeRoad(this->start_voxel.x, this->start_voxel.y)) return;
178  }
179 
180  if (!this->HasFreeGuests()) return; // No more quests available.
181  /* New guest! */
182  Guest *g = this->GetFree();
183  g->Activate(this->start_voxel, PERSON_GUEST);
184 }
