148 PathBuildGui::PathBuildGui() : GuiWindow(WC_PATH_BUILDER, ALL_WINDOWS_OF_TYPE), path_type_button_size()
149 {
150  /* Setup path type bits. */
151  const SpriteStorage *store = _sprite_manager.GetSprites(64); // GUI size.
152  for (int i = 0; i < PAT_COUNT; i++) {
153  switch (store->path_sprites[i].status) {
154  case PAS_UNUSED:
155  this->normal_path_types[i] = false;
156  this->queue_path_types[i] = false;
157  this->path_type_sprites[i] = nullptr;
158  break;
159 
160  case PAS_NORMAL_PATH:
161  this->normal_path_types[i] = true;
162  this->queue_path_types[i] = false;
163  this->path_type_sprites[i] = store->GetPathSprite(i, PATH_NE_NW_SE_SW, VOR_NORTH);
164  this->path_type_button_size.MergeArea(GetSpriteSize(this->path_type_sprites[i]));
165  break;
166 
167  case PAS_QUEUE_PATH:
168  this->normal_path_types[i] = false;
169  this->queue_path_types[i] = true;
170  this->path_type_sprites[i] = store->GetPathSprite(i, PATH_NE_SW, VOR_NORTH);
171  this->path_type_button_size.MergeArea(GetSpriteSize(this->path_type_sprites[i]));
172  break;
173 
174  default:
175  NOT_REACHED();
176  }
177  }
178 
179  this->SetupWidgetTree(_path_build_gui_parts, lengthof(_path_build_gui_parts));
180 
181  /* Select an initial path type if not done already. */
182  if (_path_builder.path_type == PAT_INVALID) {
183  for (int i = 0; i < PAT_COUNT; i++) {
184  if (this->normal_path_types[i]) {
185  _path_builder.path_type = (PathType)i;
186  break;
187  }
188  if (_path_builder.path_type == PAT_INVALID && this->queue_path_types[i]) {
189  _path_builder.path_type = (PathType)i; // Queue path is better than an invalid one, continue searching.
190  }
191  }
192  }
193 
194  _path_builder.SetPathGuiState(true);
195 }

256 void PathBuildGui::OnClick(WidgetNumber number, const Point16 &pos)
257 {
258  switch (number) {
259  case PATH_GUI_SLOPE_DOWN:
260  case PATH_GUI_SLOPE_FLAT:
261  case PATH_GUI_SLOPE_UP:
262  _path_builder.SelectSlope((TrackSlope)(number - PATH_GUI_SLOPE_DOWN));
263  break;
264 
265  case PATH_GUI_NE_DIRECTION:
266  case PATH_GUI_SE_DIRECTION:
267  case PATH_GUI_SW_DIRECTION:
268  case PATH_GUI_NW_DIRECTION: {
269  Viewport *vp = GetViewport();
270  if (vp != nullptr) {
271  TileEdge edge = (TileEdge)AddOrientations((ViewOrientation)(number - PATH_GUI_NE_DIRECTION), vp->orientation);
272  _path_builder.SelectArrow(edge);
273  }
274  break;
275  }
276 
277  case PATH_GUI_FORWARD:
278  case PATH_GUI_BACKWARD:
279  _path_builder.SelectMovement(number == PATH_GUI_FORWARD);
280  break;
281 
282  case PATH_GUI_LONG:
283  _path_builder.SelectLong();
284  break;
285 
286  case PATH_GUI_REMOVE:
287  case PATH_GUI_BUY:
288  _path_builder.SelectBuyRemove(number == PATH_GUI_BUY);
289  break;
290 
291  case PATH_GUI_NORMAL_PATH0:
292  case PATH_GUI_NORMAL_PATH1:
293  case PATH_GUI_NORMAL_PATH2:
294  case PATH_GUI_NORMAL_PATH3:
295  if (this->normal_path_types[number - PATH_GUI_NORMAL_PATH0]) {
296  _path_builder.SelectPathType((PathType)(number - PATH_GUI_NORMAL_PATH0));
297  this->SetButtons();
298  }
299  break;
300 
301  case PATH_GUI_QUEUE_PATH0:
302  case PATH_GUI_QUEUE_PATH1:
303  case PATH_GUI_QUEUE_PATH2:
304  case PATH_GUI_QUEUE_PATH3:
305  if (this->queue_path_types[number - PATH_GUI_QUEUE_PATH0]) {
306  _path_builder.SelectPathType((PathType)(number - PATH_GUI_QUEUE_PATH0));
307  this->SetButtons();
308  }
309  break;
310 
311  case PATH_GUI_SINGLE:
312  _path_builder.SetState(PBS_SINGLE);
313  this->SetButtons();
314  break;
315  case PATH_GUI_DIRECTIONAL:
316  _path_builder.SetState(PBS_WAIT_VOXEL);
317  this->SetButtons();
318  break;
319 
320  default:
321  break;
322  }

326 void PathBuildGui::SetButtons()
327 {
328  Viewport *vp = GetViewport();
329  if (vp == nullptr) return;
330 
331  /* Update arrow buttons. */
332  uint8 directions = _path_builder.GetAllowedArrows();
333  TileEdge sel_dir = _path_builder.GetSelectedArrow();
334  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
335  TileEdge rot_edge = (TileEdge)SubtractOrientations((ViewOrientation)edge, vp->orientation);
336  if (((0x11 << edge) & directions) != 0) {
337  this->SetWidgetShaded(PATH_GUI_NE_DIRECTION + rot_edge, false);
338  this->SetWidgetPressed(PATH_GUI_NE_DIRECTION + rot_edge, edge == sel_dir);
339  } else {
340  this->SetWidgetShaded(PATH_GUI_NE_DIRECTION + rot_edge, true);
341  }
342  }
343  /* Update the slope buttons. */
344  uint8 allowed_slopes = _path_builder.GetAllowedSlopes();
345  TrackSlope sel_slope = _path_builder.GetSelectedSlope();
346  for (TrackSlope ts = TSL_BEGIN; ts < TSL_COUNT_GENTLE; ts++) {
347  bool option_allowed = ((1 << ts) & allowed_slopes) != 0;
348  this->SetWidgetShaded(PATH_GUI_SLOPE_DOWN + ts, !option_allowed);
349  this->SetWidgetPressed(PATH_GUI_SLOPE_DOWN + ts, ts == sel_slope && option_allowed);
350  }
351  this->SetWidgetShaded(PATH_GUI_BUY, !_path_builder.GetBuyIsEnabled());
352  this->SetWidgetShaded(PATH_GUI_REMOVE, !_path_builder.GetRemoveIsEnabled());
353  this->SetWidgetShaded(PATH_GUI_FORWARD, !_path_builder.GetForwardIsEnabled());
354  this->SetWidgetShaded(PATH_GUI_BACKWARD, !_path_builder.GetBackwardIsEnabled());
355  this->SetWidgetShaded(PATH_GUI_LONG, !_path_builder.GetLongButtonIsEnabled());
356  this->SetWidgetPressed(PATH_GUI_LONG, _path_builder.GetLongButtonIsPressed());
357 
358  for (int i = 0; i < PAT_COUNT; i++) {
359  if (this->normal_path_types[i]) {
360  this->SetWidgetShaded(PATH_GUI_NORMAL_PATH0 + i, false);
361  this->SetWidgetShaded(PATH_GUI_QUEUE_PATH0 + i, true);
362  this->SetWidgetPressed(PATH_GUI_NORMAL_PATH0 + i, (i == _path_builder.path_type));
363  } else if (this->queue_path_types[i]) {
364  this->SetWidgetShaded(PATH_GUI_NORMAL_PATH0 + i, true);
365  this->SetWidgetShaded(PATH_GUI_QUEUE_PATH0 + i, false);
366  this->SetWidgetPressed(PATH_GUI_QUEUE_PATH0 + i, (i == _path_builder.path_type));
367  } else {
368  this->SetWidgetShaded(PATH_GUI_NORMAL_PATH0 + i, true);
369  this->SetWidgetShaded(PATH_GUI_QUEUE_PATH0 + i, true);
370  }
371  }
372 
373  this->SetWidgetPressed(PATH_GUI_SINGLE, _path_builder.GetState() == PBS_SINGLE);
374  this->SetWidgetPressed(PATH_GUI_DIRECTIONAL, _path_builder.GetState() != PBS_SINGLE);
375 }
