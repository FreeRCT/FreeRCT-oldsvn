329  case CCW_NOT_POWERED:

333  case CCW_SLOPE_DOWN:

338  case CCW_SLOPE_VERTICAL_DOWN:

356  _additions.Clear();
357  this->ci->RemovePositionedPiece(*this->cur_piece);
358  _additions.Commit();
359  this->cur_piece = pred_index == -1 ? nullptr : &this->ci->pieces[pred_index];
360  break;
361  }
362 
363  case CCW_BEND_WIDE_LEFT:
364  case CCW_BEND_NORMAL_LEFT:
365  case CCW_BEND_TIGHT_LEFT:
366  case CCW_BEND_NONE:
367  case CCW_BEND_TIGHT_RIGHT:
368  case CCW_BEND_NORMAL_RIGHT:
369  case CCW_BEND_WIDE_RIGHT:

434 void CoasterBuildWindow::SetupSelection()
435 {
436  uint directions = 0; // Build directions of initial pieces.
437  uint avail_bank = 0;
438  uint avail_slope = 0;
439  uint avail_bend = 0;
440  uint avail_platform = 0;
441  uint avail_power = 0;
442  this->sel_piece = nullptr;
443 
444  if (this->cur_piece == nullptr || this->cur_sel == nullptr) {
445  /* Only consider track pieces when there is no current positioned track piece. */
446 
447  const CoasterType *ct = this->ci->GetCoasterType();
448 
449  bool selectable[1024]; // Arbitrary limit on the number of non-placed track pieces.
450  uint count = ct->pieces.size();
451  if (count > lengthof(selectable)) count = lengthof(selectable);
452  /* Round 1: Select on connection or initial placement. */
453  for (uint i = 0; i < count; i++) {
454  ConstTrackPiecePtr piece = ct->pieces[i];
455  bool avail = true;
456  if (this->cur_piece != nullptr) {
457  /* Connect after or before 'cur_piece'. */
458  if (this->cur_after) {
459  if (piece->entry_connect != this->cur_piece->piece->exit_connect) avail = false;
460  } else {
461  if (piece->exit_connect != this->cur_piece->piece->entry_connect) avail = false;
462  }
463  } else {
464  /* Initial placement. */
465  if (!piece->IsStartingPiece()) {
466  avail = false;
467  } else {
468  directions |= 1 << piece->GetStartDirection();
469  if (piece->GetStartDirection() != this->build_direction) avail = false;
470  }
471  }
472  selectable[i] = avail;
473  }
474 
475  /* Round 2: Setup banking. */
476  for (uint i = 0; i < count; i++) {
477  if (!selectable[i]) continue;
478  ConstTrackPiecePtr piece = ct->pieces[i];
479  avail_bank |= 1 << piece->GetBanking();
480  }
481  if (this->sel_bank != TPB_INVALID && (avail_bank & (1 << this->sel_bank)) == 0) this->sel_bank = TPB_INVALID;
482 
483  /* Round 3: Setup slopes from pieces with the correct bank. */
484  for (uint i = 0; i < count; i++) {
485  if (!selectable[i]) continue;
486  ConstTrackPiecePtr piece = ct->pieces[i];
487  if (this->sel_bank != TPB_INVALID && piece->GetBanking() != this->sel_bank) {
488  selectable[i] = false;
489  } else {
490  avail_slope |= 1 << piece->GetSlope();
491  }
492  }
493  if (this->sel_slope != TSL_INVALID && (avail_slope & (1 << this->sel_slope)) == 0) this->sel_slope = TSL_INVALID;
494 
495  /* Round 4: Setup bends from pieces with the correct slope. */
496  for (uint i = 0; i < count; i++) {
497  if (!selectable[i]) continue;
498  ConstTrackPiecePtr piece = ct->pieces[i];
499  if (this->sel_slope != TSL_INVALID && piece->GetSlope() != this->sel_slope) {
500  selectable[i] = false;
501  } else {
502  avail_bend |= 1 << piece->GetBend();
503  }
504  }
505  if (this->sel_bend != TBN_INVALID && (avail_bend & (1 << this->sel_bend)) == 0) this->sel_bend = TBN_INVALID;
506 
507  /* Round 5: Setup platform from pieces with the correct bend. */
508  for (uint i = 0; i < count; i++) {
509  if (!selectable[i]) continue;
510  ConstTrackPiecePtr piece = ct->pieces[i];
511  if (this->sel_bend != TBN_INVALID && piece->GetBend() != this->sel_bend) {
512  selectable[i] = false;
513  } else {
514  avail_platform |= 1 << GetPlatform(piece);
515  }
516  }
517  if (this->sel_platform != BSL_NONE && (avail_platform & (1 << this->sel_platform)) == 0) this->sel_platform = BSL_NONE;
518 
519  /* Round 6: Setup power from pieces with the correct platform. */
520  for (uint i = 0; i < count; i++) {
521  if (!selectable[i]) continue;
522  ConstTrackPiecePtr piece = ct->pieces[i];
523  if (this->sel_platform != BSL_NONE && GetPlatform(piece) != this->sel_platform) {
524  selectable[i] = false;
525  } else {
526  avail_power |= 1 << GetPower(piece);
527  }
528  }
529  if (this->sel_power != BSL_NONE && (avail_power & (1 << this->sel_power)) == 0) this->sel_power = BSL_NONE;
530 
531  /* Round 7: Select a piece from the pieces with the correct power. */
532  for (uint i = 0; i < count; i++) {
