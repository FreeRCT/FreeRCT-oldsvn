145 void Recolouring::AssignRandomColours()
146 {
147  Random rnd;
148 
149  for (uint i = 0; i < lengthof(this->entries); i++) {
150  RecolourEntry &re = this->entries[i];
151  if (re.source != COL_RANGE_INVALID && re.dest == COL_RANGE_INVALID) {
152  if (re.dest_set == 0) {
153  re.source = COL_RANGE_INVALID;
154  continue;
155  }
156  int num_bits = CountBits(re.dest_set);
157  num_bits = (num_bits == 1) ? 0 : rnd.Uniform(num_bits - 1);
158  for (int j = 0; j < 32; j++) {
159  if (GB(re.dest_set, j, 1) != 0) {
160  num_bits--;
161  if (num_bits < 0) {
162  re.AssignDest((ColourRange)j);
163  break;
164  }
165  }
166  }
167  }
168  }
169 }

176 const uint8 *Recolouring::GetPalette(GradientShift shift) const
177 {
178  if (this->shift == shift) return this->colour_map;
179 
180  for (int i = 0; i < COL_SERIES_START; i++) this->colour_map[i] = i;
181  for (int i = COL_SERIES_END; i < 256; i++) this->colour_map[i] = i;
182  for (int rng = 0; rng < COL_RANGE_COUNT; rng++) {
183  int base = GetColourRangeBase((ColourRange)rng);
184  int baseval = GetColourRangeBase(this->GetReplacementRange((ColourRange)rng));
185  for (int col = 0; col < COL_SERIES_LENGTH; col++) {
186  this->colour_map[base + col] = baseval + Clamp(col + shift - GS_NORMAL, 0, COL_SERIES_LENGTH - 1);
187  }
188  }
189  this->shift = shift;
190  return this->colour_map;
191 }
