81 const char *ImageFile::LoadFile(const std::string &fname)
82 {
83  FILE *fp = fopen(fname.c_str(), "rb");
84  if (fp == nullptr) return "Input file does not exist";
85 
86  uint8 header[HEADER_SIZE];
87  if (fread(header, 1, HEADER_SIZE, fp) != HEADER_SIZE) {
88  fclose(fp);
89  return "Failed to read the PNG header";
90  }
91  bool is_png = !png_sig_cmp(header, 0, HEADER_SIZE);
92  if (!is_png) {
93  fclose(fp);
94  return "Header indicates it is not a PNG file";
95  }
96 
97  this->png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING, nullptr, nullptr, nullptr);
98  if (!this->png_ptr) {
99  fclose(fp);
100  return "Failed to initialize the png data";
101  }
102 
103  this->info_ptr = png_create_info_struct(this->png_ptr);
104  if (!this->info_ptr) {
105  png_destroy_read_struct(&this->png_ptr, (png_infopp)nullptr, (png_infopp)nullptr);
106  fclose(fp);
107  return "Failed to setup a png info structure";
108  }
109 
110  this->end_info = png_create_info_struct(this->png_ptr);
111  if (!this->end_info) {
112  png_destroy_read_struct(&this->png_ptr, &this->info_ptr, (png_infopp)nullptr);
113  fclose(fp);
114  return "Failed to setup a png end info structure";
115  }
116 
117  /* Setup callback in case of errors. */
118  if (setjmp(png_jmpbuf(this->png_ptr))) {
119  png_destroy_read_struct(&this->png_ptr, &this->info_ptr, &this->end_info);
120  fclose(fp);
121  return "Error detected while reading PNG file";
122  }
123 
124  /* Initialize for file reading. */
125  png_init_io(this->png_ptr, fp);
126  png_set_sig_bytes(this->png_ptr, HEADER_SIZE);
127 
128  png_read_png(this->png_ptr, this->info_ptr, PNG_TRANSFORM_IDENTITY, nullptr);
129  this->row_pointers = png_get_rows(this->png_ptr, this->info_ptr);
130  fclose(fp);
131  this->png_initialized = true; // Clear will now clean up.
132 
133  int bit_depth = png_get_bit_depth(this->png_ptr, this->info_ptr);
134  if (bit_depth != 8) return "Depth of the image channels is not 8 bit";
135 
136  this->width = png_get_image_width(this->png_ptr, this->info_ptr);
137  this->height = png_get_image_height(this->png_ptr, this->info_ptr);
138  this->color_type = png_get_color_type(this->png_ptr, this->info_ptr);
139  if (this->color_type != PNG_COLOR_TYPE_PALETTE && this->color_type != PNG_COLOR_TYPE_RGB_ALPHA) {
140  return "Incorrect type of image (expected either 8bpp paletted image or RGBA)";
141  }
142 
143  this->fname = fname;
144  return nullptr; // Loading was a success.
145 }

262 {

326 uint8 *Image8bpp::Encode(int xpos, int ypos, int width, int height, int *size) const
327 {
328  auto row_sizes = std::vector<int>();
329  row_sizes.reserve(height);
330 
331  /* Examine the sprite, and record length of data for each row. */
332  int data_size = 0;
333  for (int y = 0; y < height; y++) {
334  int length = 0;
335  int last_stored = 0; // Up to this position (exclusive), the row was counted.
336  for (int x = 0; x < width; x++) {
337  if (this->IsTransparent(xpos + x, ypos + y)) continue;
338 
339  int start = x;
340  x++;
341  while (x < width) {
342  if (this->IsTransparent(xpos + x, ypos + y)) break;
343  x++;
344  }
345  /* from 'start' upto and excluding 'x' are pixels to draw. */
346  while (last_stored + 127 < start) {
347  length += 2; // 127 pixels gap, 0 pixels to draw.
348  last_stored += 127;
349  }
350  while (x - start > 255) {
351  length += 2 + 255; // ((start-last_stored) gap, 255 pixels, and 255 colour indices.
352  start += 255;
353  }
354  length += 2 + x - start;
355  last_stored = x;
356  }
357  assert(length <= 0xffff);
358  row_sizes.push_back(length);
359  data_size += length;
360  }
361  if (data_size == 0) { // No pixels -> no need to store any data.
362  *size = 0;
363  return nullptr;
364  }
365 
366  data_size += 4 * height; // Add jump table space.
367  uint8 *data = new uint8[data_size];
368  if (data == nullptr) {
369  *size = 1;
370  return nullptr;
371  }
372  uint8 *ptr = data;
373 
374  /* Construct jump table. */
375  uint32 offset = 4 * height;
376  for (int y = 0; y < height; y++) {
377  uint32 value = (row_sizes[y] == 0) ? 0 : offset;
378  WriteUInt32(value, ptr);
379  ptr += 4;
380  offset += row_sizes[y];
381  }
382 
383  /* Copy sprite pixels. */
384  for (int y = 0; y < height; y++) {
385  if (row_sizes[y] == 0) continue;
386 
387  uint8 *last_header = nullptr;
388  int last_stored = 0; // Up to this position (exclusive), the row was counted.
389  for (int x = 0; x < width; x++) {
390  if (this->IsTransparent(xpos + x, ypos + y)) continue;
391 
392  int start = x;
393  x++;
394  while (x < width) {
395  if (this->IsTransparent(xpos + x, ypos + y)) break;
396  x++;
397  }
398  /* from 'start' up to and excluding 'x' are pixels to draw. */
399  while (last_stored + 127 < start) {
400  *ptr++ = 127; // 127 pixels gap, 0 pixels to draw.
401  *ptr++ = 0;
402  last_stored += 127;
403  }
404  while (x - start > 255) {
405  *ptr++ = start - last_stored;
406  *ptr++ = 255;
407  for (int i = 0; i < 255; i++) {
408  *ptr++ = this->GetPixel(xpos + start, ypos + y);
409  start++;
410  }
411  last_stored = start;
412  }
413  last_header = ptr;
414  *ptr++ = start - last_stored;
415  *ptr++ = x - start;
416  while (x > start) {
417  *ptr++ = this->GetPixel(xpos + start, ypos + y);
418  start++;
419  }
420  last_stored = x;
421  }
422  assert(last_header != nullptr);
423  *last_header |= 128; // This was the last sequence of pixels.
424  }
425  assert(ptr - data == data_size);
426  *size = data_size;
427  return data;
428 }

586 uint8 *Image32bpp::Encode(int xpos, int ypos, int width, int height, int *size) const
587 {
589  //
590  auto row_sizes = std::vector<int>();
591  row_sizes.reserve(height);
592 
593  /* Decide size of each scanline. */
594  for (int y = 0; y < height; y++) {
595  int count = 0;
596  int x = 0;
597  while (x < width) {
598  int recolour_length, opaq_length;
599  uint8 recolour = GetCurrentRecolour(xpos + x, ypos + y, width - x, &recolour_length);
600  uint8 opaq = this->GetSameOpaqueness(xpos + x, ypos + y, recolour_length, &opaq_length);
601 
602  if (recolour != 0) {
603  /* Recoloured pixels. */
604  if (opaq_length > 63) opaq_length = 63;
605  count += 1 + 1 + 1 + opaq_length;
606  x += opaq_length;
607  continue;
608  }
609  if (opaq == FULLY_OPAQUE) {
610  /* Plain fully opaque 24 bit pixels. */
611  if (opaq_length > 63) opaq_length = 63;
612  count += 1 + opaq_length * 3;
613  x += opaq_length;
614  continue;
615  }
616  if (opaq == FULLY_TRANSPARENT) {
617  /* Plain fully transparent pixels. */
618  if (opaq_length + x == width) break; // End of the line reached.
619 
620  if (opaq_length > 63) opaq_length = 63;
621  count += 1;
622  x += opaq_length;
623  continue;
624  }
625  /* Partially transparent pixels. */
626  if (opaq_length > 63) opaq_length = 63;
627  count += 1 + 1 + opaq_length * 3;
628  x += opaq_length;
629  }
630  count++; // Terminating 0 byte
631  row_sizes.push_back(count);
632  }
633 
634  *size = 0;
635  for (int y = 0; y < height; y++) {
636  assert(row_sizes[y] <= 0xFFFF - 2);
637  *size += 2 + row_sizes[y];
638  }
639  uint8 *data = new uint8[*size];
640  uint8 *ptr = data;
641 
642  /* Store pixel data. */
643  for (int y = 0; y < height; y++) {
644  uint size = (y < height - 1) ? row_sizes[y] + 2 : 0;
645  *ptr++ = size & 0xff;
646  *ptr++ = (size >> 8) & 0xff;
647 
648  int x = 0;
649  while (x < width) {
650  int recolour_length, opaq_length;
651  uint8 recolour = GetCurrentRecolour(xpos + x, ypos + y, width - x, &recolour_length);
652  uint8 opaq = this->GetSameOpaqueness(xpos + x, ypos + y, recolour_length, &opaq_length);
653 
654  if (recolour != 0) {
655  /* Recoloured pixels. */
656  if (opaq_length > 63) opaq_length = 63;
657  *ptr++ = 128 + 64 + opaq_length;
658  *ptr++ = recolour;
659  *ptr++ = opaq;
660  for (int dx = 0; dx < opaq_length; dx++) {
661  uint32 pixel = this->GetPixel(xpos + x, ypos + y);
662  *ptr++ = std::max(std::max(GetR(pixel), GetG(pixel)), GetB(pixel));
663  x++;
664  }
665  continue;
666  }
667  if (opaq == FULLY_OPAQUE) {
668  /* Plain fully opaque 24 bit pixels. */
669  if (opaq_length > 63) opaq_length = 63;
670  *ptr++ = 0 + opaq_length;
671  for (int dx = 0; dx < opaq_length; dx++) {
672  uint32 pixel = this->GetPixel(xpos + x, ypos + y);
673  *ptr++ = GetR(pixel);
674  *ptr++ = GetG(pixel);
675  *ptr++ = GetB(pixel);
676  x++;
677  }
678  continue;
679  }
680  if (opaq == FULLY_TRANSPARENT) {
681  /* Plain fully transparent pixels. */
682  if (opaq_length + x == width) break; // End of the line reached.
683 
684  if (opaq_length > 63) opaq_length = 63;
685  *ptr++ = 128 + opaq_length;
686  x += opaq_length;
687  continue;
688  }
689  /* Partially transparent pixels. */
690  if (opaq_length > 63) opaq_length = 63;
691  *ptr++ = 64 + opaq_length;
692  *ptr++ = opaq;
693  for (int dx = 0; dx < opaq_length; dx++) {
694  uint32 pixel = this->GetPixel(xpos + x, ypos + y);
695  *ptr++ = GetR(pixel);
696  *ptr++ = GetG(pixel);
697  *ptr++ = GetB(pixel);
698  x++;
699  }
700  }
701  *ptr++ = 0; // Terminating 0 byte.
702  }
703  assert(ptr - data == *size);
704  return data;
705 }

791  this->width = xsize;
