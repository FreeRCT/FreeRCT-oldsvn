113 std::shared_ptr<const Expression> UnaryOperator::Evaluate(const Symbol *symbols) const
114 {
115  std::shared_ptr<const Expression> result = this->child->Evaluate(symbols);
116  std::shared_ptr<const NumberLiteral> number = std::dynamic_pointer_cast<const NumberLiteral>(result);
117  if (number == nullptr) {
118  fprintf(stderr, "Evaluate error at %s: Cannot apply unary operator '%c' to the child expression", this->pos.ToString(), this->oper);
119  exit(1);
120  }
121  switch (this->oper) {
122  case '-': return std::make_shared<const NumberLiteral>(this->pos, -number->value);
123  case '~': return std::make_shared<const NumberLiteral>(this->pos, ~number->value);
124  default: NOT_REACHED();
125  }

142 std::shared_ptr<const Expression> BinaryOperator::Evaluate(const Symbol *symbols) const
143 {
144  std::shared_ptr<const Expression> result = this->left->Evaluate(symbols);
145  auto left_operand = std::dynamic_pointer_cast<const NumberLiteral>(result);
146  if (left_operand == nullptr) {
147  fprintf(stderr, "Evaluate error at %s: Cannot apply binary operator '%c' to the left operand", this->pos.ToString(), this->oper);
148  exit(1);
149  }
150  result = this->right->Evaluate(symbols);
151  auto right_operand = std::dynamic_pointer_cast<const NumberLiteral>(result);
152  if (right_operand == nullptr) {
153  fprintf(stderr, "Evaluate error at %s: Cannot apply binary operator '%c' to the right operand", this->pos.ToString(), this->oper);
154  exit(1);
155  }
156 
157  switch (this->oper) {
158  case '&': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value & right_operand->value);
159  case '|': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value | right_operand->value);
160  case '^': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value ^ right_operand->value);
161  case '<': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value < right_operand->value);
162  case '>': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value > right_operand->value);
163  case '+': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value + right_operand->value);
164  case '-': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value - right_operand->value);
165  case '*': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value * right_operand->value);
166  case '/': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value / right_operand->value);
167  case '%': return std::make_shared<const NumberLiteral>(this->pos, left_operand->value % right_operand->value);
168  default: NOT_REACHED();
169  }
170 }

379 int NameTable::GetNameCount() const
380 {
381  if (this->HasSingleElement()) {
382  /* Single element table, possibly with parameterized name. */
383  const std::shared_ptr<IdentifierLine> il = this->rows.front()->identifiers.front();
384  assert(il->IsValid());
385  ParameterizedName parms_name;
386  HorVert hv = parms_name.DecodeName(il->name.c_str(), il->pos);
387  switch (hv) {
388  case HV_NONE: return 1;
389  case HV_HOR: return parms_name.hor_range.size();
390  case HV_VERT: return parms_name.vert_range.size();
391  case HV_BOTH: return parms_name.hor_range.size() * parms_name.vert_range.size();
392  default: NOT_REACHED();
393  }
394  }
395  /* General case, assume non-parameterized names, and return their count. */
396  int count = 0;
397  for (const auto &iter : this->rows) {
398  count += iter->GetNameCount();
399  }
400  return count;
401 }

504 void NamedValueList::HandleImports()
505 {
506  bool has_import = false;
507  std::list<std::shared_ptr<BaseNamedValue>> values;
508 
509  for (auto &iter : this->values) {
510  std::shared_ptr<ImportValue> iv = std::dynamic_pointer_cast<ImportValue>(iter);
511  if (iv != nullptr) {
512  has_import = true;
513  std::shared_ptr<NamedValueList> nv = LoadFile(iv->filename.c_str(), iv->pos.line);
514  for (auto &iter2 : nv->values) values.push_back(iter2);
515  } else {
516  iter->HandleImports();
517  values.push_back(iter);
518  }
519  }
520  if (has_import) this->values = values;
521 }

529 std::shared_ptr<NamedValueList> LoadFile(const char *filename, int line)
530 {
531  static int nest_level = 0;
532  static const char *include_cache[10];
533  static int line_number_cache[10];
534 
535  /* Check for too many nested include levels. */
536  if (nest_level > 0) line_number_cache[nest_level - 1] = line;
537 
538  if (nest_level >= (int)lengthof(include_cache)) {
539  fprintf(stderr, "Error: Too many nested file imports\n");
540  fprintf(stderr, " While importing \"%s\"\n", filename);
541  for (int i = nest_level; i >= 0; i--) {
542  fprintf(stderr, " from \"%s\" at line %d\n", include_cache[i], line_number_cache[i]);
543  }
544  exit(1);
545  }
546 
547  include_cache[nest_level] = filename;
548  nest_level++;
549 
550  /* Parse the input. */
551  FILE *infile = nullptr;
552  if (filename != nullptr) {
553  infile = fopen(filename, "rb");
554  if (infile == nullptr) {
555  fprintf(stderr, "Error: Could not open file \"%s\"\n", filename);
556  exit(1);
557  }
558  }
559  _parsed_data = nullptr;
560  SetupScanner(filename, infile);
561  yyparse();
562 
563  if (infile != nullptr) fclose(infile);
564 
565  if (_parsed_data == nullptr) {
566  fprintf(stderr, "Parsing of the input file did not give a result\n");
567  exit(1);
568  }
569 
570  /* Process imports. */
571  std::shared_ptr<NamedValueList> nvs = _parsed_data;
572  nvs->HandleImports(); // Recursively calls this function, so _parsed_data is not safe.
573 
574  /* Restore to pre-call state. */
575  include_cache[nest_level] = nullptr;
576  nest_level--;
577  return nvs;
578 }
