84 void SettingWindow::OnClick(WidgetNumber number, const Point16 &pos)
85 {
86  switch (number) {
87  case SW_LANGUAGE: {
88  DropdownList itemlist;
89  for (int i = 0; i < LANGUAGE_COUNT; i++) {
90  _str_params.SetUint8(1, _language.GetLanguageName(i));
91  itemlist.push_back(DropdownItem(STR_ARG1));
92  }
93  this->ShowDropdownMenu(number, itemlist, _current_language);
94  break;
95  }
96 
97  case SW_RESOLUTION: {
98  /* Get current resolution index. */
99  int index = 0;
100  auto iter = _video.resolutions.find({_video.GetXSize(), _video.GetYSize()});
101  if (iter != _video.resolutions.end()) {
102  index = std::distance(_video.resolutions.begin(), iter);
103  }
104 
105  DropdownList itemlist;
106  for (const auto &res : _video.resolutions) {
107  _str_params.SetNumber(1, res.x);
108  _str_params.SetNumber(2, res.y);
109  itemlist.push_back(DropdownItem(GUI_RESOLUTION));
110  }
111  this->ShowDropdownMenu(number, itemlist, index);
112  }
113  }

116 void SettingWindow::OnChange(ChangeCode code, uint32 parameter)
117 {
118  if (code != CHG_DROPDOWN_RESULT) return;
119 
120  switch ((parameter >> 16) & 0xFF) {
121  case SW_LANGUAGE:
122  _current_language = parameter & 0xFF;
123  _window_manager.ResetAllWindows();
124  break;
125 
126  case SW_RESOLUTION: {
127  int index = 0;
128  for (const auto &it : _video.resolutions) {
129  if (index == (int)(parameter & 0xFF)) {
130  _video.SetResolution(it);
131  break;
132  }
133  index++;
134  }
135  break;
136  }
137 
138  default: NOT_REACHED();
139  }
140 }
