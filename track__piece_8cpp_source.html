121 static bool LoadTrackCurve(RcdFileReader *rcdfile, TrackCurve **curve, uint32 *length)
122 {
123 #define ENSURE_LENGTH(x) do { if (*length < (x)) { *curve = nullptr; return false; } *length -= (x); } while(false)
124 
125  ENSURE_LENGTH(1);
126  uint8 type = rcdfile->GetUInt8();
127  switch (type) {
128  case 0: // No track curve available.
129  *curve = nullptr;
130  return true;
131 
132  case 1: { // Curve consisting of a fixed value.
133  ENSURE_LENGTH(2);
134  int value = rcdfile->GetInt16();
135  *curve = new ConstantTrackCurve(value);
136  return true;
137  }
138 
139  case 2: {
140  ENSURE_LENGTH(1);
141  int count = rcdfile->GetUInt8();
142  ENSURE_LENGTH(count * 16u);
143  BezierTrackCurve *bezier = new BezierTrackCurve();
144  bezier->curve.reserve(count);
145  for (; count > 0; count--) {
146  bezier->curve.push_back(LoadBezier(rcdfile));
147  }
148  bezier->curve.shrink_to_fit();
149  *curve = bezier;
150  return true;
151  }
152 
153  default: // Error.
154  fprintf(stderr, "Unexpected curve type %u.\n", type);
155  *curve = nullptr;
156  return false;
157  }
158 #undef ENSURE_LENGTH
159 }

189 bool TrackPiece::Load(RcdFileReader *rcd_file, const ImageMap &sprites)
190 {
191  uint32 length = rcd_file->size;
192  if (rcd_file->version != 5 || length < 2 + 3 + 1 + 2 + 4 + 2) return false;
193  length -= 2 + 3 + 1 + 2 + 4 + 2;
194 
195  this->entry_connect = rcd_file->GetUInt8();
196  this->exit_connect = rcd_file->GetUInt8();
197  this->exit_dxyz.x = rcd_file->GetInt8();
198  this->exit_dxyz.y = rcd_file->GetInt8();
199  this->exit_dxyz.z = rcd_file->GetInt8();
200  this->speed = rcd_file->GetInt8();
201  this->track_flags = rcd_file->GetUInt16();
202  this->cost = rcd_file->GetUInt32();
203  uint16 voxel_count = rcd_file->GetUInt16();
204 
205  if (length < 36u * voxel_count) return false;
206  length -= 36u * voxel_count;
207 
208  for (uint16 i = 0; i < voxel_count; i++) {
209  this->track_voxels.emplace_back(new TrackVoxel);
210  if (!this->track_voxels.back()->Load(rcd_file, 36, sprites)) return false;
211  }
212  if (length < 4u) return false;
213  length -= 4;
214  this->piece_length = rcd_file->GetUInt32();
215 
216  bool ok = true;
217  ok = ok && LoadTrackCurve(rcd_file, &this->car_xpos, &length);
218  ok = ok && LoadTrackCurve(rcd_file, &this->car_ypos, &length);
219  ok = ok && LoadTrackCurve(rcd_file, &this->car_zpos, &length);
220  ok = ok && LoadTrackCurve(rcd_file, &this->car_pitch, &length);
221  ok = ok && LoadTrackCurve(rcd_file, &this->car_roll, &length);
222  ok = ok && LoadTrackCurve(rcd_file, &this->car_yaw, &length);
223  if (!ok || this->car_xpos == nullptr || this->car_ypos == nullptr || this->car_zpos == nullptr || this->car_roll == nullptr) return false;
224  return length == 0;
225 }

243 bool PositionedTrackPiece::IsOnWorld() const
244 {
245  assert(this->piece != nullptr);
246  if (!IsVoxelInsideWorld(this->base_voxel)) return false;
247  if (!IsVoxelInsideWorld(this->GetEndXYZ())) return false;
248  XYZPoint16 base = this->base_voxel;
249  return std::all_of(this->piece->track_voxels.begin(), this->piece->track_voxels.end(),
250  [base](TrackVoxel * tv){ return IsVoxelInsideWorld(base + tv->dxyz); });
251 }

258 bool PositionedTrackPiece::CanBePlaced() const
259 {
260  if (!this->IsOnWorld()) return false;
261  for (const auto &tvx : this->piece->track_voxels) {
262  /* Is the voxel above ground level? */
263  const XYZPoint16 part_pos = this->base_voxel + tvx->dxyz;
264  if (_world.GetGroundHeight(part_pos.x, part_pos.y) > part_pos.z) return false;
265  const Voxel *vx = _world.GetVoxel(part_pos);
266  if (vx != nullptr && !vx->CanPlaceInstance()) return false;
267  }
268  return true;
269 }
