92 void TerraformGui::DrawWidget(WidgetNumber wid_num, const BaseWidget *wid) const
93 {
94  Point32 base;
95  static const Recolouring recolour; // Not changed.
96 
97  if (wid_num != TERR_DISPLAY) return;
98 
99  /* Draw the tile area, case of a null-area. */
100  if (this->xsize == 0 && this->ysize == 0) {
101  const ImageData *dot = _gui_sprites.dot_sprite;
102  if (dot == nullptr) return;
103 
104  base.x = this->GetWidgetScreenX(wid) + (wid->pos.width - dot->width) / 2;
105  base.y = this->GetWidgetScreenY(wid) + (wid->pos.height - dot->height) / 2;
106  _video.BlitImage(base, dot, recolour, GS_NORMAL);
107  return;
108  }
109 
110  /* Draw area with >= 1 tile.
111  * Here, size == xsize, but also (size / 2) == ysize, since xsize == ysize * 2 in a flat tile world.
112  */
113  int size = this->xsize * TERRAFORM_ELEMENT_SIZE / 2 + this->ysize * TERRAFORM_ELEMENT_SIZE / 2;
114  base.x = this->GetWidgetScreenX(wid) + (wid->pos.width - size) / 2; // Left position of the drawn tiles.
115  base.y = this->GetWidgetScreenY(wid) + (wid->pos.height - size / 2) / 2; // Top position of the drawn tiles.
116  base.y += this->xsize * TERRAFORM_ELEMENT_SIZE / 4; // Lower to (0, 0) of the drawn tiles (left-most position).
117 
118  for (int x = 0; x < this->xsize; x++) {
119  for (int y = 0; y < this->ysize; y++) {
120  Point16 left(base.x + (x + y) * TERRAFORM_ELEMENT_SIZE / 2, base.y + (y - x) * TERRAFORM_ELEMENT_SIZE / 4);
121 
122  Point16 top(left.x + TERRAFORM_ELEMENT_SIZE / 2, left.y - TERRAFORM_ELEMENT_SIZE / 4);
123  _video.DrawLine(left, top, MakeRGBA(255, 255, 255, OPAQUE));
124 
125  Point16 bottom(top.x, left.y + TERRAFORM_ELEMENT_SIZE / 4);
126  _video.DrawLine(left, bottom, MakeRGBA(255, 255, 255, OPAQUE));
127  left.x += TERRAFORM_ELEMENT_SIZE; // Move 'left' to the right.
128  _video.DrawLine(top, left, MakeRGBA(255, 255, 255, OPAQUE));
129  _video.DrawLine(bottom, left, MakeRGBA(255, 255, 255, OPAQUE));
130  }
131  }
132 }

134 void TerraformGui::OnClick(WidgetNumber wid, const Point16 &pos)
135 {
136  switch (wid) {
137  case TERR_ADD:
138  this->IncreaseSize();
139  break;
140 
141  case TERR_SUB:
142  this->DecreaseSize();
143  break;
144 
145  case TERR_LEVEL_TEXT:
146  case TERR_LEVEL:
147  if (!this->level) this->Setlevelling(true);
148  break;
149 
150  case TERR_MOVE_TEXT:
151  case TERR_MOVE:
152  if (this->level) this->Setlevelling(false);
153  break;
154 
155  default:
156  break;
157  }
158 }

194 void TerraformGui::IncreaseSize()
195 {
196  if (this->xsize >= TERRAFORM_MAX_SIZE && this->ysize >= TERRAFORM_MAX_SIZE) return;
197 
198  if (this->xsize == 0 && this->ysize == 0) {
199  this->SetTerraformSize(1, 1);
200  } else if (this->xsize > this->ysize) {
201  this->SetTerraformSize(-1, this->ysize + 1);
202  } else {
203  this->SetTerraformSize(this->xsize + 1, -1);
204  }
205 }
206 
208 void TerraformGui::DecreaseSize()
209 {
210  if (this->xsize < 1 && this->ysize < 1) return;
211 
212  if (this->xsize == 1 && this->ysize == 1) {
213  this->SetTerraformSize(0, 0);
214  } else if (this->xsize > this->ysize) {
215  this->SetTerraformSize(this->xsize - 1, -1);
216  } else {
217  this->SetTerraformSize(-1, this->ysize - 1);
218  }
219 }
