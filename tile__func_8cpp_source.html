53 void ComputeCornerHeight(TileSlope slope, uint8 base_height, uint8 *output)
54 {
55  if ((slope & TSB_STEEP) != 0) {
56  uint i;
57  for (i = TC_NORTH; i < TC_END; i++) {
58  if ((slope & (1 << i)) != 0) break;
59  }
60  assert(i < TC_END);
61 
62  if ((slope & TSB_TOP) != 0) base_height--; // Move to the base height.
63 
64  output[i] = base_height + 2;
65  output[(i + 2) % 4] = base_height;
66  output[(i + 1) % 4] = base_height + 1;
67  output[(i + 3) % 4] = base_height + 1;
68  return;
69  }
70  for (uint i = TC_NORTH; i < TC_END; i++) {
71  output[i] = ((slope & (1 << i)) == 0) ? base_height : base_height + 1;
72  }
73 }

81 void ComputeSlopeAndHeight(uint8 *corners, TileSlope *slope, uint8 *base)
82 {
83  uint8 max_h = 0;
84  uint8 min_h = 255;
85  for (uint i = 0; i < 4; i++) {
86  if (max_h < corners[i]) max_h = corners[i];
87  if (min_h > corners[i]) min_h = corners[i];
88  }
89  *base = min_h;
90 
91  if (max_h - min_h <= 1) {
92  /* Normal slope. */
93  *slope = static_cast<TileSlope>(ImplodeTileSlope(((corners[TC_NORTH] > min_h) ? TSB_NORTH : SL_FLAT)
94  | ((corners[TC_EAST] > min_h) ? TSB_EAST : SL_FLAT)
95  | ((corners[TC_SOUTH] > min_h) ? TSB_SOUTH : SL_FLAT)
96  | ((corners[TC_WEST] > min_h) ? TSB_WEST : SL_FLAT)));
97  } else {
98  assert(max_h - min_h == 2);
99  uint i;
100  for (i = 0; i < 4; i++) {
101  if (corners[i] == max_h) break;
102  }
103  *slope = static_cast<TileSlope>(ImplodeTileSlope(TSB_STEEP | static_cast<TileSlope>(1 << i)));
104  }
105 }

113 bool MayHaveGroundFenceInVoxelAbove(TileSlope slope)
114 {
115  if ((slope & TSB_STEEP) != 0) return true;
116  for (uint8 i = 0; i < 4; i++) {
117  /* Are there two adjacent corners that are raised? */
