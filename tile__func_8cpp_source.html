45 void ComputeCornerHeight(TileSlope slope, uint8 base_height, uint8 *output)
46 {
47  if ((slope & TSB_STEEP) != 0) {
48  uint i;
49  for (i = TC_NORTH; i < TC_END; i++) {
50  if ((slope & (1 << i)) != 0) break;
51  }
52  assert(i < TC_END);
53 
54  if ((slope & TSB_TOP) != 0) base_height--; // Move to the base height.
55 
56  output[i] = base_height + 2;
57  output[(i + 2) % 4] = base_height;
58  output[(i + 1) % 4] = base_height + 1;
59  output[(i + 3) % 4] = base_height + 1;
60  return;
61  }
62  for (uint i = TC_NORTH; i < TC_END; i++) {
63  output[i] = ((slope & (1 << i)) == 0) ? base_height : base_height + 1;
64  }
65 }

73 void ComputeSlopeAndHeight(uint8 *corners, TileSlope *slope, uint8 *base)
74 {
75  uint8 max_h = 0;
76  uint8 min_h = 255;
77  for (uint i = 0; i < 4; i++) {
78  if (max_h < corners[i]) max_h = corners[i];
79  if (min_h > corners[i]) min_h = corners[i];
80  }
81  *base = min_h;
82 
83  if (max_h - min_h <= 1) {
84  /* Normal slope. */
85  *slope = static_cast<TileSlope>(ImplodeTileSlope(((corners[TC_NORTH] > min_h) ? TSB_NORTH : SL_FLAT)
86  | ((corners[TC_EAST] > min_h) ? TSB_EAST : SL_FLAT)
87  | ((corners[TC_SOUTH] > min_h) ? TSB_SOUTH : SL_FLAT)
88  | ((corners[TC_WEST] > min_h) ? TSB_WEST : SL_FLAT)));
89  } else {
90  assert(max_h - min_h == 2);
91  uint i;
92  for (i = 0; i < 4; i++) {
93  if (corners[i] == max_h) break;
94  }
95  *slope = static_cast<TileSlope>(ImplodeTileSlope(TSB_STEEP | static_cast<TileSlope>(1 << i)));
96  }
97 }

105 bool MayHaveGroundFenceInVoxelAbove(TileSlope slope)
