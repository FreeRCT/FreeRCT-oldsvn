132 void DateOnTick()
133 {
134  bool newmonth = false;
135  bool newyear = false;
136 
137  /* New tick. */
138  _date.frac++;
139  if (_date.frac < TICK_COUNT_PER_DAY) return;
140 
141  /* New day. */
142  _date.frac = 0;
143  _date.day++;
144 
145  /* New month. */
146  if (_date.day > _days_per_month[_date.month]) {
147  bool is_last_month = (_date.month == LAST_MONTH);
148  _date.day = 1;
149  _date.month++;
150  newmonth = true;
151 
152  /* New year. */
153  if (is_last_month || _date.month > 12) {
154  _date.month = FIRST_MONTH;
155  _date.year++;
156  newyear = true;
157  }
158  }
159 
160  OnNewDay();
161  if (newmonth) OnNewMonth();
162  if (newyear) OnNewYear();
163 }
