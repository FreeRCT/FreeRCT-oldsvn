122 static char *StripWhitespace(char *first, char *last = nullptr)
123 {
124  if (last == nullptr) {
125  last = first;
126  while (*last != '\0') last++;
127  }
128 
129  assert(*last == '\0');
130 
131  while (first < last && IsSpace(*first)) first++;
132  while (last > first + 1 && IsSpace(last[-1])) last--;
133  *last = '\0';
134  return first;
135 }

162 bool ConfigFile::Load(const char *fname)
163 {
164  ConfigSection *current_sect = nullptr;
165 
166  this->Clear();
167 
168  FILE *fp = fopen(fname, "rb");
169  if (fp == nullptr) return false;
170 
171  for (;;) {
172  char buffer[256];
173  char *line = fgets(buffer, lengthof(buffer), fp);
174  if (line == nullptr) break;
175 
176  while (*line != '\0' && IsSpace(*line)) line++;
177  if (*line == '\0' || *line == ';') continue; // Silently skip empty lines or comment lines.
178 
179  if (*line == '[') {
180  /* New section. */
181  char *line2 = line;
182  while (*line2 != '\0' && *line2 != ']') line2++;
183  if (*line2 == ']') {
184  *line2 = '\0';
185  /* XXX Look for an existing config section? */
186  current_sect = new ConfigSection(StripWhitespace(line + 1, line2));
187  this->sections.push_front(current_sect);
188  }
189  continue;
190  }
191 
192  if (current_sect == nullptr) continue; // No section opened, ignore such lines.
193 
194  /* Key/value line. */
195  char *line2 = line;
196  while (*line2 != '\0' && *line2 != '=') line2++;
197  ConfigItem *item;
198  if (*line2 == '=') {
199  *line2 = '\0';
200  item = new ConfigItem(StripWhitespace(line, line2), StripWhitespace(line2 + 1));
201  } else {
202  /* No value. */
203  item = new ConfigItem(StripWhitespace(line, line2), "");
204  }
205  current_sect->items.push_front(item);
206  }
207  fclose(fp);
208  return true;
209 }
