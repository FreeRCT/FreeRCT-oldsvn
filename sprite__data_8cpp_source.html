45 bool ImageData::Load8bpp(RcdFileReader *rcd_file, size_t length)
46 {
47  if (length < 8) return false; // 2 bytes width, 2 bytes height, 2 bytes x-offset, and 2 bytes y-offset
48  this->width = rcd_file->GetUInt16();
49  this->height = rcd_file->GetUInt16();
50  this->xoffset = rcd_file->GetInt16();
51  this->yoffset = rcd_file->GetInt16();
52 
53  /* Check against some arbitrary limits that look sufficient at this time. */
54  if (this->width == 0 || this->width > 300 || this->height == 0 || this->height > 500) return false;
55 
56  length -= 8;
57  if (length > 100 * 1024) return false; // Another arbitrary limit.
58 
59  size_t jmp_table = 4 * this->height;
60  if (length <= jmp_table) return false; // You need at least place for the jump table.
61  length -= jmp_table;
62 
63  this->table = new uint32[jmp_table / 4];
64  this->data = new uint8[length];
65  if (this->table == nullptr || this->data == nullptr) return false;
66 
67  /* Load jump table, adjusting the entries while loading. */
68  for (uint i = 0; i < this->height; i++) {
69  uint32 dest = rcd_file->GetUInt32();
70  if (dest == 0) {
71  this->table[i] = INVALID_JUMP;
72  continue;
73  }
74  dest -= jmp_table;
75  if (dest >= length) return false;
76  this->table[i] = dest;
77  }
78 
79  rcd_file->GetBlob(this->data, length); // Load the image data.
80 
81  /* Verify the image data. */
82  for (uint i = 0; i < this->height; i++) {
83  uint32 offset = this->table[i];
84  if (offset == INVALID_JUMP) continue;
85 
86  uint32 xpos = 0;
87  for (;;) {
88  if (offset + 2 >= length) return false;
89  uint8 rel_pos = this->data[offset];
90  uint8 count = this->data[offset + 1];
91  xpos += (rel_pos & 127) + count;
92  offset += 2 + count;
93  if ((rel_pos & 128) == 0) {
94  if (xpos >= this->width || offset >= length) return false;
95  } else {
96  if (xpos > this->width || offset > length) return false;
97  break;
98  }
99  }
100  }
101  return true;
102 }

110 bool ImageData::Load32bpp(RcdFileReader *rcd_file, size_t length)
111 {
112  if (length < 8) return false; // 2 bytes width, 2 bytes height, 2 bytes x-offset, and 2 bytes y-offset
113  this->width = rcd_file->GetUInt16();
114  this->height = rcd_file->GetUInt16();
115  this->xoffset = rcd_file->GetInt16();
116  this->yoffset = rcd_file->GetInt16();
117 
118  /* Check against some arbitrary limits that look sufficient at this time. */
119  if (this->width == 0 || this->width > 300 || this->height == 0 || this->height > 500) return false;
120 
121  length -= 8;
122  if (length > 100 * 1024) return false; // Another arbitrary limit.
123 
124  /* Allocate and load the image data. */
125  this->data = new uint8[length];
126  if (this->data == nullptr) return false;
127  rcd_file->GetBlob(this->data, length);
128 
129  /* Verify the data. */
130  uint8 *abs_end = this->data + length;
131  int line_count = 0;
132  const uint8 *ptr = this->data;
133  bool finished = false;
134  while (ptr < abs_end && !finished) {
135  line_count++;
136 
137  /* Find end of this line. */
138  uint16 line_length = ptr[0] | (ptr[1] << 8);
139  const uint8 *end;
140  if (line_length == 0) {
141  finished = true;
142  end = abs_end;
143  } else {
144  end = ptr + line_length;
145  if (end > abs_end) return false;
146  }
147  ptr += 2;
148 
149  /* Read line. */
150  bool finished_line = false;
151  uint xpos = 0;
152  while (ptr < end && !finished_line) {
153  uint8 mode = *ptr++;
154  if (mode == 0) {
155  finished_line = true;
156  break;
157  }
158  xpos += mode & 0x3F;
159  switch (mode >> 6) {
160  case 0: ptr += 3 * (mode & 0x3F); break;
161  case 1: ptr += 1 + 3 * (mode & 0x3F); break;
162  case 2: break;
163  case 3: ptr += 1 + 1 + (mode & 0x3F); break;
164  }
165  }
166  if (xpos > this->width) return false;
167  if (!finished_line) return false;
168  if (ptr != end) return false;
169  }
170  if (line_count != this->height) return false;
171  if (ptr != abs_end) return false;
172  return true;
173 }

183 uint32 ImageData::GetPixel(uint16 xoffset, uint16 yoffset, const Recolouring *recolour, GradientShift shift) const
184 {
185  if (xoffset >= this->width) return _palette[0];
186  if (yoffset >= this->height) return _palette[0];
187 
188  if (GB(this->flags, IFG_IS_8BPP, 1) != 0) {
189  /* 8bpp image. */
190  uint32 offset = this->table[yoffset];
191  if (offset == INVALID_JUMP) return _palette[0];
192 
193  uint16 xpos = 0;
194  while (xpos <= xoffset) {
195  uint8 rel_pos = this->data[offset];
196  uint8 count = this->data[offset + 1];
197  xpos += (rel_pos & 127);
198  if (xpos > xoffset) return _palette[0];
199  if (xoffset - xpos < count) {
200  uint8 pixel = this->data[offset + 2 + xoffset - xpos];
201  if (recolour != nullptr) {
202  const uint8 *recolour_table = recolour->GetPalette(shift);
203  pixel = recolour_table[pixel];
204  }
205  return _palette[pixel];
206  }
207  xpos += count;
208  offset += 2 + count;
209  if ((rel_pos & 128) != 0) break;
210  }
211  return _palette[0];
212  } else {
213  /* 32bpp image. */
214  const uint8 *ptr = this->data;
215  while (yoffset > 0) {
216  uint16 length = ptr[0] | (ptr[1] << 8);
217  ptr += length;
218  yoffset--;
219  }
220  ptr += 2;
221  while (xoffset > 0) {
222  uint8 mode = *ptr++;
223  if (mode == 0) break;
224  if ((mode & 0x3F) < xoffset) {
225  xoffset -= mode & 0x3F;
226  switch (mode >> 6) {
227  case 0: ptr += 3 * (mode & 0x3F); break;
228  case 1: ptr += 1 + 3 * (mode & 0x3F); break;
229  case 2: ptr++; break;
230  case 3: ptr += 1 + 1 + (mode & 0x3F); break;
231  }
232  } else {
233  ShiftFunc sf = GetGradientShiftFunc(shift);
234  switch (mode >> 6) {
235  case 0:
236  ptr += 3 * xoffset;
237  return MakeRGBA(sf(ptr[0]), sf(ptr[1]), sf(ptr[2]), OPAQUE);
238  case 1: {
239  uint8 opacity = *ptr;
240  ptr += 1 + 3 * xoffset;
241  return MakeRGBA(sf(ptr[0]), sf(ptr[1]), sf(ptr[2]), opacity);
242  }
243  case 2:
244  return _palette[0]; // Arbitrary fully transparent.
245  case 3: {
246  uint8 opacity = ptr[1];
247  if (recolour == nullptr) return MakeRGBA(0, 0, 0, opacity); // Arbitrary colour with the correct opacity.
248  const uint32 *table = recolour->GetRecolourTable(ptr[0] - 1);
249  ptr += 2 + xoffset;
250  uint32 recoloured = table[*ptr];
251  return MakeRGBA(sf(GetR(recoloured)), sf(GetG(recoloured)), sf(GetB(recoloured)), opacity);
252  }
253  }
254  }
255  }
256  return _palette[0]; // Arbitrary fully transparent.
257  }
258 }

265 ImageData *LoadImage(RcdFileReader *rcd_file)
266 {
267  bool is_8bpp = strcmp(rcd_file->name, "8PXL") == 0;
268  if (rcd_file->version != (is_8bpp ? 2 : 1)) return nullptr;
269 
270  _sprites.emplace_back();
271  ImageData *imd = &_sprites.back();
272  bool loaded = is_8bpp ? imd->Load8bpp(rcd_file, rcd_file->size) : imd->Load32bpp(rcd_file, rcd_file->size);
273  if (!loaded) {
274  _sprites.pop_back();
275  return nullptr;
276  }
277  imd->flags = is_8bpp ? (1 << IFG_IS_8BPP) : 0;
278  return imd;
279 }
