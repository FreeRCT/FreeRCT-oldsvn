72 uint8 GroundData::GetOrigHeight(TileCorner corner) const
73 {
74  assert(corner >= TC_NORTH && corner < TC_END);
75  if ((this->orig_slope & TSB_STEEP) == 0) { // Normal slope.
76  if ((this->orig_slope & (1 << corner)) == 0) return this->height;
77  return this->height + 1;
78  }
79  // Steep slope (note that the constructor made sure this->height is at the base of the steep slope).
80  if ((this->orig_slope & (1 << corner)) != 0) return this->height + 2;
81  corner = (TileCorner)((corner + 2) % 4);
82  if ((this->orig_slope & (1 << corner)) != 0) return this->height;
83  return this->height + 1;
84 }

155 void TerrainChanges::UpdatelevellingHeight(const Point32 &pos, int direction, uint8 *height)
156 {
157  const GroundData *gd = this->GetGroundData(pos);
158 
159  if (direction > 0) { // Raising terrain, find the lowest corner.
160  *height = std::min(*height, gd->GetOrigHeight(TC_NORTH));
161  *height = std::min(*height, gd->GetOrigHeight(TC_EAST));
162  *height = std::min(*height, gd->GetOrigHeight(TC_SOUTH));
163  *height = std::min(*height, gd->GetOrigHeight(TC_WEST));
164  } else { // Lowering terrain, find the highest corner.
165  *height = std::max(*height, gd->GetOrigHeight(TC_NORTH));
166  *height = std::max(*height, gd->GetOrigHeight(TC_EAST));
167  *height = std::max(*height, gd->GetOrigHeight(TC_SOUTH));
168  *height = std::max(*height, gd->GetOrigHeight(TC_WEST));
169  }

179 bool TerrainChanges::ChangeVoxel(const Point32 &pos, uint8 height, int direction)
180 {
181  GroundData *gd = this->GetGroundData(pos);
182  bool ok = true;
183  if (direction > 0) { // Raising terrain, raise everything at or below 'height'.
184  if (gd->GetOrigHeight(TC_NORTH) <= height) ok &= this->ChangeCorner(pos, TC_NORTH, direction);
185  if (gd->GetOrigHeight(TC_EAST) <= height) ok &= this->ChangeCorner(pos, TC_EAST, direction);
186  if (gd->GetOrigHeight(TC_SOUTH) <= height) ok &= this->ChangeCorner(pos, TC_SOUTH, direction);
187  if (gd->GetOrigHeight(TC_WEST) <= height) ok &= this->ChangeCorner(pos, TC_WEST, direction);
188  } else { // Lowering terrain, lower everything above or at 'height'.
189  if (gd->GetOrigHeight(TC_NORTH) >= height) ok &= this->ChangeCorner(pos, TC_NORTH, direction);
190  if (gd->GetOrigHeight(TC_EAST) >= height) ok &= this->ChangeCorner(pos, TC_EAST, direction);
191  if (gd->GetOrigHeight(TC_SOUTH) >= height) ok &= this->ChangeCorner(pos, TC_SOUTH, direction);
192  if (gd->GetOrigHeight(TC_WEST) >= height) ok &= this->ChangeCorner(pos, TC_WEST, direction);
193  }
194  return ok;
195 }

204 bool TerrainChanges::ChangeCorner(const Point32 &pos, TileCorner corner, int direction)
205 {
206  assert(corner >= TC_NORTH && corner < TC_END);
207  assert(direction == 1 || direction == -1);
208 
209  GroundData *gd = this->GetGroundData(pos);
210  if (gd == nullptr) return true; // Out of the bounds in the world, silently ignore.
211  if (gd->GetCornerModified(corner)) return true; // Corner already changed.
212 
213  if (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(pos.x, pos.y) != OWN_PARK) return false;
214 
215  uint8 old_height = gd->GetOrigHeight(corner);
216  if (direction > 0 && old_height == WORLD_Z_SIZE) return false; // Cannot change above top.
217  if (direction < 0 && old_height == 0) return false; // Cannot change below bottom.
218 
219  gd->SetCornerModified(corner); // Mark corner as modified.
220 
221  /* Change neighbouring corners at the same tile. */
222  if (direction > 0) {
223  uint8 h = gd->GetOrigHeight(neighbours[corner].left_neighbour);
224  if (h < old_height && !this->ChangeCorner(pos, neighbours[corner].left_neighbour, direction)) return false;
225  h = gd->GetOrigHeight(neighbours[corner].right_neighbour);
226  if (h < old_height && !this->ChangeCorner(pos, neighbours[corner].right_neighbour, direction)) return false;
227  } else {
228  uint8 h = gd->GetOrigHeight(neighbours[corner].left_neighbour);
229  if (h > old_height && !this->ChangeCorner(pos, neighbours[corner].left_neighbour, direction)) return false;
230  h = gd->GetOrigHeight(neighbours[corner].right_neighbour);
231  if (h > old_height && !this->ChangeCorner(pos, neighbours[corner].right_neighbour, direction)) return false;
232  }
233 
234  for (uint i = 0; i < 3; i++) {
235  const VoxelCorner &vc = neighbours[corner].neighbour_tiles[i];
236  Point32 pos2(pos.x + vc.rel_xy.x, pos.y + vc.rel_xy.y);
237  gd = this->GetGroundData(pos2);
238  if (gd == nullptr) continue;
239  if (_world.GetTileOwner(pos2.x, pos2.y) != OWN_PARK) continue;
240  uint height = gd->GetOrigHeight(vc.corner);
241  if (old_height == height && !this->ChangeCorner(pos2, vc.corner, direction)) return false;
242  }
243  return true;
244 }

253 static void SetUpperBoundary(const Voxel *v, uint8 height, uint8 *bounds)
254 {
255  if (v == nullptr || v->IsEmpty()) return;
256 
257  assert(v->GetGroundType() == GTP_INVALID);
258  uint16 instance = v->GetInstance();
259  if (instance >= SRI_FULL_RIDES) { // A ride needs the entire voxel.
260  std::fill_n(bounds, 4, height);
261  return;
262  }
263 
264  if (instance < SRI_RIDES_START) return;
265 
266  /* Small rides, that is, a path. */
267  if (!HasValidPath(v)) return;
268  PathSprites ps = GetImplodedPathSlope(v);
269  switch (ps) {
270  case PATH_RAMP_NE:
271  bounds[TC_NORTH] = height; bounds[TC_EAST] = height;
272  bounds[TC_SOUTH] = height + 1; bounds[TC_WEST] = height + 1;
273  break;
274 
275  case PATH_RAMP_NW:
276  bounds[TC_NORTH] = height; bounds[TC_WEST] = height;
277  bounds[TC_SOUTH] = height + 1; bounds[TC_EAST] = height + 1;
278  break;
279 
280  case PATH_RAMP_SE:
281  bounds[TC_SOUTH] = height; bounds[TC_EAST] = height;
282  bounds[TC_NORTH] = height + 1; bounds[TC_WEST] = height + 1;
283  break;
284 
285  case PATH_RAMP_SW:
286  bounds[TC_SOUTH] = height; bounds[TC_WEST] = height;
287  bounds[TC_NORTH] = height + 1; bounds[TC_EAST] = height + 1;
288  break;
289 
290  default:
291  assert(ps < PATH_FLAT_COUNT);
292  std::fill_n(bounds, 4, height);
293  break;
294  }
295 }

307 static void SetFoundations(VoxelStack *stack, uint8 my_first, uint8 my_second, uint8 other_first, uint8 other_second, uint8 first_bit, uint8 second_bit)
308 {
309  if (stack == nullptr) return;
310 
311  uint8 and_bits = ~(first_bit | second_bit);
312  bool is_higher = my_first > other_first || my_second > other_second; // At least one of my corners must be higher to ever add foundations.
313 
314  uint8 h = stack->base;
315  uint8 highest = stack->base + stack->height;
316 
317  if (other_first < h) h = other_first;
318  if (other_second < h) h = other_second;
319 
320  while (h < highest) {
321  uint8 bits = 0;
322  if (is_higher && (h >= other_first || h >= other_second)) {
323  if (h < my_first) bits |= first_bit;
324  if (h < my_second) bits |= second_bit;
325  }
326 
327  Voxel *v = stack->GetCreate(h, true);
328  h++;
329  if (bits == 0) { // Delete foundations.
330  if (v->GetFoundationType() == FDT_INVALID) continue;
331  bits = v->GetFoundationSlope() & and_bits;
332  v->SetFoundationSlope(bits);
333  if (bits == 0) v->SetFoundationType(FDT_INVALID);
334  continue;
335  } else { // Add foundations.
336  if (v->GetFoundationType() == FDT_INVALID) {
337  v->SetFoundationType(FDT_GROUND); // XXX We have nice foundation types, but no way to select them here.
338  } else {
339  bits |= (v->GetFoundationSlope() & and_bits);
340  }
341  v->SetFoundationSlope(bits);
342  continue;
343  }
344  }
345 }

353 static void SetXFoundations(int xpos, int ypos)
354 {
355  VoxelStack *first = (xpos < 0) ? nullptr : _additions.GetModifyStack(xpos, ypos);
356  VoxelStack *second = (xpos + 1 == _world.GetXSize()) ? nullptr : _additions.GetModifyStack(xpos + 1, ypos);
357  assert(first != nullptr || second != nullptr);
358 
359  /* Get ground height at all corners. */
360  uint8 first_south = 0;
361  uint8 first_west = 0;
362  if (first != nullptr) {
363  for (uint i = 0; i < first->height; i++) {
364  const Voxel *v = &first->voxels[i];
365  if (v->GetGroundType() == GTP_INVALID) continue;
366  uint8 heights[4];
367  ComputeCornerHeight(ExpandTileSlope(v->GetGroundSlope()), first->base + i, heights);
368  first_south = heights[TC_SOUTH];
369  first_west = heights[TC_WEST];
370  break;
371  }
372  }
373 
374  uint8 second_north = 0;
375  uint8 second_east = 0;
376  if (second != nullptr) {
377  for (uint i = 0; i < second->height; i++) {
378  const Voxel *v = &second->voxels[i];
379  if (v->GetGroundType() == GTP_INVALID) continue;
380  uint8 heights[4];
381  ComputeCornerHeight(ExpandTileSlope(v->GetGroundSlope()), second->base + i, heights);
382  second_north = heights[TC_NORTH];
383  second_east = heights[TC_EAST];
384  break;
385  }

442 bool TerrainChanges::ModifyWorld(int direction)
443 {
444  assert(_mouse_modes.GetMouseMode() == MM_TILE_TERRAFORM);
445 
446  _additions.Clear();
447 
448  /* First iteration: Change the ground of the tiles, checking
449  * whether the change is actually allowed with the other game elements. */
450  for (auto &iter : this->changes) {
451  const Point32 &pos = iter.first;
452  const GroundData &gd = iter.second;
453  if (gd.modified == 0) continue;
454 
455  uint8 current[4]; // Height of each corner after applying modification.
456  ComputeCornerHeight(static_cast<TileSlope>(gd.orig_slope), gd.height, current);
457 
458  /* Apply modification. */
459  for (uint8 i = TC_NORTH; i < TC_END; i++) {
460  if ((gd.modified & (1 << i)) == 0) continue; // Corner was not changed.
461  current[i] += direction;
462  }
463 
464  /* Clear the current ground from the stack. */
465  VoxelStack *vs = _additions.GetModifyStack(pos.x, pos.y);
466  Voxel *v = vs->GetCreate(gd.height, false); // Should always exist.
467  GroundType gt = v->GetGroundType();
468  assert(gt != GTP_INVALID);
469  FoundationType ft = v->GetFoundationType();
470  FenceType fence[4];
471  for (uint8 i = 0; i < 4; i++) fence[i] = v->GetFenceType((TileEdge)i);
472  TileSlope slope = ExpandTileSlope(v->GetGroundSlope());
473  v->SetGroundType(GTP_INVALID);
474  v->SetFoundationType(FDT_INVALID);
475  v->SetGroundSlope(0);
476  v->SetFoundationSlope(0);
477  for (uint8 i = 0; i < 4; i++) v->SetFenceType((TileEdge)i, FENCE_TYPE_INVALID);
478  if (MayHaveGroundFenceInVoxelAbove(slope)) {
479  Voxel *w = vs->GetCreate(gd.height + 1, false);
480  if ((slope & TSB_STEEP) != 0) {
481  assert(w->GetGroundType() == gt); // Should be the same type of ground as the base voxel.
482  w->SetFoundationType(FDT_INVALID);
483  w->SetGroundType(GTP_INVALID);
484  w->SetGroundSlope(0);
485  w->SetFoundationSlope(0);
486  }
487  if (w != nullptr) { // w will always exist if steep slope, but not always for non-steep.
488  for (uint8 i = 0; i < 4; i++) {
489  FenceType f = w->GetFenceType((TileEdge)i);
490  if (f != FENCE_TYPE_INVALID) fence[i] = f;
491  w->SetFenceType((TileEdge)i, FENCE_TYPE_INVALID);
492  }
493  }
494  }
495 
496  if (direction > 0) {
497  /* Moving upwards, compute upper bound on corner heights. */
498  uint8 max_above[4];
499  std::fill_n(max_above, lengthof(max_above), gd.height + 3);
500 
501  for (int i = 2; i >= 0; i--) {
502  SetUpperBoundary(vs->Get(gd.height + i), gd.height + i, max_above);
503  }
504 
505  /* Check boundaries. */
506  for (uint i = 0; i < 4; i++) {
507  if (current[i] > max_above[i]) return false;
508  }
509  } /* else: Moving downwards always works, since there is nothing underground yet. */
510 
511  /* Add new ground to the stack. */
512  TileSlope new_slope;
513  uint8 height;
514  ComputeSlopeAndHeight(current, &new_slope, &height);
515  TileSlope new_exp_slope = ExpandTileSlope(new_slope);
516  if (height >= WORLD_Z_SIZE) return false;
517 
518  v = vs->GetCreate(height, true);
519  v->SetGroundSlope(new_slope);
520  v->SetGroundType(gt);
521  v->SetFoundationType(ft);
522  v->SetFoundationSlope(0);
523 
524  for (uint8 i = 0; i < 4; i++) {
525  v->SetFenceType((TileEdge)i, StoreFenceInUpperVoxel(new_exp_slope, (TileEdge)i) ? FENCE_TYPE_INVALID : fence[i]);
526  }
527 
528  if (MayHaveGroundFenceInVoxelAbove(new_exp_slope)) {
529  v = vs->GetCreate(height + 1, true);
530  if ((TSB_STEEP & new_exp_slope) != 0) {
531  /* Only for steep slopes, the upper voxel will have actual ground. */
532  v->SetGroundType(gt);
533  v->SetGroundSlope(new_slope + 4); // Set top-part as well for steep slopes.
534  v->SetFoundationType(ft);
535  v->SetFoundationSlope(0);
536  }
537  for (uint8 i = 0; i < 4; i++) {
538  v->SetFenceType((TileEdge)i, StoreFenceInUpperVoxel(new_exp_slope, (TileEdge)i) ? fence[i] : FENCE_TYPE_INVALID);
539  }
540  }
541  }
542 
543  /* Second iteration: Add foundations to every changed tile edge.
544  * The general idea is that each modified voxel handles adding
545  * of foundation to its SE and SW edge. If the NE or NW voxel is not
546  * modified, the voxel will have to perform adding of foundations
547  * there as well. */
548  for (auto &iter : this->changes) {
549  const Point32 &pos = iter.first;
550  const GroundData &gd = iter.second;
551  if (gd.modified == 0) continue;
552 
553  SetXFoundations(pos.x, pos.y);
554  SetYFoundations(pos.x, pos.y);
555 
556  Point32 pt(pos.x - 1, pos.y);
557  auto iter2 = this->changes.find(pt);
558  if (iter2 == this->changes.end()) {
559  SetXFoundations(pt.x, pt.y);
560  } else {
561  const GroundData &gd = iter2->second;
562  if (gd.modified == 0) SetXFoundations(pt.x, pt.y);
563  }
564 
565  pt.x = pos.x;
566  pt.y = pos.y - 1;
567  iter2 = this->changes.find(pt);
568  if (iter2 == this->changes.end()) {
569  SetYFoundations(pt.x, pt.y);
570  } else {
571  const GroundData &gd = (*iter2).second;
572  if (gd.modified == 0) SetYFoundations(pt.x, pt.y);
573  }
574  }
575 
576  _additions.Commit();
577  return true;
578 }

644 void TileTerraformMouseMode::SetCursors()
645 {
646  Viewport *vp = GetViewport();
647  if (vp == nullptr) return;
648 
649  bool single = this->xsize <= 1 && this->ysize <= 1;
650  FinderData fdata(CS_GROUND, single ? FW_CORNER : FW_TILE);
651  if (vp->ComputeCursorPosition(&fdata) != CS_NONE) {
652  if (single) {
653  vp->tile_cursor.SetCursor(fdata.voxel_pos, fdata.cursor);

690 static void ChangeTileCursorMode(Viewport *vp, bool levelling, int direction, bool dot_mode)
691 {
692  Cursor *c = &vp->tile_cursor;
693  if (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(c->cursor_pos.x, c->cursor_pos.y) != OWN_PARK) return;
694 
695  Point32 p;
696  uint16 w, h;
697 
698  if (dot_mode) { // Change entire world.
699  p = {0, 0};
700  w = _world.GetXSize();
701  h = _world.GetYSize();
702  } else { // Single tile mode.
703  p = {c->cursor_pos.x, c->cursor_pos.y};
704  w = 1;
705  h = 1;
706  }
707  TerrainChanges changes(p, w, h);
708 
709  p = {c->cursor_pos.x, c->cursor_pos.y};
710 
711  bool ok;
712  switch (c->type) {
713  case CUR_TYPE_NORTH:
714  case CUR_TYPE_EAST:
715  case CUR_TYPE_SOUTH:
716  case CUR_TYPE_WEST:
717  ok = changes.ChangeCorner(p, (TileCorner)(c->type - CUR_TYPE_NORTH), direction);
718  break;
719 
720  case CUR_TYPE_TILE: {
721  uint8 height = (direction > 0) ? WORLD_Z_SIZE : 0;
722  if (levelling) changes.UpdatelevellingHeight(p, direction, &height);
723  ok = changes.ChangeVoxel(p, height, direction);
724  break;
725  }
726 
727  default:
728  NOT_REACHED();
729  }
730 
731  if (ok) {
732  ok = changes.ModifyWorld(direction);
733  _additions.Clear();
734  if (!ok) return;
735 
736  /* Move voxel selection along with the terrain to allow another mousewheel event at the same place.
737  * Note that the mouse cursor position is not changed at all, it still points at the original position.
738  * The coupling is restored with the next mouse movement.
739  */
740  c->cursor_pos.z = _world.GetGroundHeight(c->cursor_pos.x, c->cursor_pos.y);
741  for (const auto &iter : changes.changes) {
742  const Point32 &pt = iter.first;
743  vp->MarkVoxelDirty(XYZPoint16(pt.x, pt.y, iter.second.height));
744  }
745  }
746 }

754 static void ChangeAreaCursorMode(Viewport *vp, bool levelling, int direction)
755 {
756  Point32 p;
757 
758  MultiCursor *c = &vp->area_cursor;
759  TerrainChanges changes(c->rect.base, c->rect.width, c->rect.height);
760 
761  uint8 height = (direction > 0) ? WORLD_Z_SIZE : 0;
762  if (levelling) {
763  for (uint dx = 0; dx < c->rect.width; dx++) {
764  p.x = c->rect.base.x + dx;
765  for (uint dy = 0; dy < c->rect.height; dy++) {
766  p.y = c->rect.base.y + dy;
767  if (_game_mode_mgr.InEditorMode() || _world.GetTileOwner(p.x, p.y) == OWN_PARK) {
768  changes.UpdatelevellingHeight(p, direction, &height);
769  }
770  }
771  }
772  }
773 
774  /* Make the change in 'changes'. */
775  for (uint dx = 0; dx < c->rect.width; dx++) {
776  p.x = c->rect.base.x + dx;
777  for (uint dy = 0; dy < c->rect.height; dy++) {
778  p.y = c->rect.base.y + dy;
779  if (_game_mode_mgr.InEditorMode() || _world.GetTileOwner(p.x, p.y) == OWN_PARK) {
780  if (!changes.ChangeVoxel(p, height, direction)) return;
781  }
782  }
783  }
784 
785  bool ok = changes.ModifyWorld(direction);
786  _additions.Clear();
787  if (!ok) return;
788 
789  /* Like the dotmode, the cursor position is changed, but the mouse position is not touched to allow more
790  * mousewheel events to happen at the same place. */
791  for (const auto &iter : changes.changes) {
792  const Point32 &pt = iter.first;
793  c->ResetZPosition(pt);
794  vp->MarkVoxelDirty(XYZPoint16(pt.x, pt.y, iter.second.height));
795  }
796 }

807 void TileTerraformMouseMode::OnMouseMoveEvent(Viewport *vp, const Point16 &old_pos, const Point16 &pos)
808 {
809  if ((this->mouse_state & MB_RIGHT) != 0) {
810  /* Drag the window if button is pressed down. */
811  vp->MoveViewport(pos.x - old_pos.x, pos.y - old_pos.y);
812 
813  } else if ((this->mouse_state & MB_LEFT) != 0) {
814  /* Terraforming with holding left mouse button and move mouse up and down. */
815 
816  if (pos.y - old_pos.y > 0) {
817  this->pixel_counter--;
818  } else if (pos.y - old_pos.y < 0) {
819  this->pixel_counter++;
820  } else {
821  return;
822  }
823 
824  if (this->pixel_counter == 10 || this->pixel_counter == -10) {
825  if (vp->tile_cursor.type != CUR_TYPE_INVALID) {
826  bool dot_mode = this->xsize == 0 && this->ysize == 0;
827  ChangeTileCursorMode(vp, this->levelling, pixel_counter / 10, dot_mode);
828  } else {
829  ChangeAreaCursorMode(vp, this->levelling, pixel_counter / 10);
830  }
831  this->pixel_counter = 0;
832  }
833 
834  } else {
835  this->pixel_counter = 0; // Reset counter when LMB is not pressed.
836  this->SetCursors();
837  }
