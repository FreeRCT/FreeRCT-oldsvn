59 int GetLanguageIndex(const char *lang_name)
60 {
61  int start = 0; // Exclusive lower bound.
62  int end = LANGUAGE_COUNT; // Exclusive upper bound.
63 
64  while (start + 1 < end) {
65  int middle = (start + end) / 2;
66  int cmp = strcmp(_lang_names[middle], lang_name);
67  if (cmp == 0) return middle; // Jack pot.
68  if (cmp < 0) {
69  start = middle;
70  } else {
71  end = middle;
72  }
73  }
74  if (!strcmp(_lang_names[start], lang_name)) return start;
75  return -1;
76 }

210 uint16 Language::RegisterStrings(const TextData &td, const char * const names[], uint16 base)
211 {
212  /* Count the number of strings. */
213  uint num_strings = 0;
214  const char * const *str = names;
215  while (*str != nullptr) {
216  num_strings++;
217  str++;
218  }
219 
220  if (base == STR_GENERIC_END) {
221  /* Check for enough free space, and assign a block. */
222  if (this->first_free + num_strings >= STR_END_FREE_SPACE) {
223  fprintf(stderr, "Not enough space to store strings.\n");
224  exit(1);
225  }
226  base = this->first_free;
227  this->first_free += num_strings;
228  } else {
229  /* Pre-defined strings should be completely below the free space. */
230  assert(base + num_strings >= base && base + num_strings <= this->first_free);
231  }
232 
233  /* Copy strings in the expected order. */
234  uint16 number = base;
235  str = names;
236  while (*str != nullptr) {
237  this->registered[number] = nullptr;
238  for (uint i = 0; i < td.string_count; i++) {
239  const TextString *ts = td.strings + i;
240  if (!strcmp(*str, ts->name)) {
241  this->registered[number] = ts;
242  break;
243  }
244  }
245  number++;
246  str++;
247  }
248  return base;
249 }

310 static void MoneyStrFmt(uint8 *dest, size_t size, double amt)
311 {
312  const uint8 *curr_sym = _language.GetText(GUI_MONEY_CURRENCY_SYMBOL);
313  size_t curr_sym_len = StrBytesLength(curr_sym);
314 
315  const uint8 *tho_sep = _language.GetText(GUI_MONEY_THOUSANDS_SEPARATOR);
316  size_t tho_sep_len = StrBytesLength(tho_sep);
317 
318  const uint8 *dec_sep = _language.GetText(GUI_MONEY_DECIMAL_SEPARATOR);
319  size_t dec_sep_len = StrBytesLength(dec_sep);
320 
321  char *buf = new char[size];
322 
323  /* Convert double into a numeric string (with '.' as the decimal separator). */
324  uint len = snprintf(buf, size, "%.2f", amt);
325 
326  /* Compute the offset of where we should begin counting for thousand separators. */
327  uint comma_start = (len - 3) % 3;
328 
329  /* What is the max number of characters we might append in a single loop?
330  * This is used to prevent a potential buffer overflow. */
331  /* max_append_size = 'a "special" symbol plus a single digit [0-9]'. */
332  int max_append_size = std::max(std::max(curr_sym_len, tho_sep_len), dec_sep_len) + 1;
333 
334  uint j = 0;
335  uint curpos = 0;
336 
337  /* Also has the added bonus of 'removing' the
338  * automagically included '-' symbol by snprintf */
339  if (amt < 0) {
340  dest[j++] = '-';
341  curpos++;
342  }
343 
344  /* Copy currency symbol next */
345  SafeStrncpy(dest + j, curr_sym, curr_sym_len + 1);
346  j += curr_sym_len;
347  dest[j++] = buf[curpos++];
348 
349  for (uint i = curpos; i < len && j + max_append_size < size; i++) {
350  if (len - i > 3 && (int)(i - comma_start) % 3 == 0) {
351  SafeStrncpy(dest + j, tho_sep, tho_sep_len + 1);
352  j += tho_sep_len;
353  dest[j++] = buf[i];
354 
355  } else if (buf[i] == '.') { // Decimal separator produced by 'snprintf'.
356  SafeStrncpy(dest + j, dec_sep, dec_sep_len + 1);
357  j += dec_sep_len;
358 
359  } else {
360  dest[j++] = buf[i];
361  }
362  }
363 
364  assert((size_t)j < size);
365  dest[j] = '\0';
366  delete[] buf;
367 }

375 static void TemperatureStrFormat(uint8 *dest, size_t size, int temp)
376 {
377  static const uint8 SUFFIX[] = {' ', 0xE2, 0x84, 0x83, 0}; // " " + degrees Celcius, U+2103
378 
379  temp = ((temp < 0) ? temp - 5 : temp + 5) / 10; // Round to degrees Celcius.
380  int len = snprintf((char *)dest, size, "%d", temp);
381  dest += len;
382  size -= len;
383  if (size >= lengthof(SUFFIX)) StrECpy(dest, dest + size, SUFFIX);
384 }

393 void DrawText(StringID strid, uint8 *buffer, uint length, StringParameters *params)
394 {
395  static uint8 textbuf[64];
396 
397  const uint8 *txt = _language.GetText(strid);
398  const uint8 *last = buffer + ((int)length - 1);
399  const uint8 *ptr = txt;
400  for (;;) {
401  while (*ptr != '\0' && *ptr != '%' && buffer < last) {
402  *buffer = *ptr;
403  buffer++;
404  ptr++;
405  }
406  if (*ptr == '\0' || buffer >= last) break;
407  ptr++;
408  if (*ptr == '%') {
409  *buffer = '%';
410  buffer++;
411  ptr++;
412  continue;
413  }
414  int n = 0;
415  while (*ptr >= '0' && *ptr <= '9') {
416  n = n * 10 + *ptr - '0';
417  ptr++;
418  }
419  if (params != nullptr && n >= 1 && n <= (int)lengthof(params->parms)) {
420  /* Expand parameter 'n-1'. */
421  switch (params->parms[n - 1].parm_type) {
422  case SPT_NONE:
423  buffer = CopyString(buffer, last, (uint8 *)"NONE");
424  break;
425 
426  case SPT_STRID:
427  buffer = CopyString(buffer, last, _language.GetText(params->parms[n - 1].u.str));
428  break;
429 
430  case SPT_UINT8:
431  buffer = CopyString(buffer, last, params->parms[n - 1].u.text);
432  break;
433 
434  case SPT_NUMBER:
435  snprintf((char *)textbuf, lengthof(textbuf), "%lld", params->parms[n - 1].u.number);
436  buffer = CopyString(buffer, last, textbuf);
437  break;
438 
439  case SPT_MONEY:
440  MoneyStrFmt(textbuf, lengthof(textbuf), params->parms[n - 1].u.number / 100.0);
441  buffer = CopyString(buffer, last, textbuf);
442  break;
443 
444  case SPT_TEMPERATURE:
445  TemperatureStrFormat(textbuf, lengthof(textbuf), params->parms[n - 1].u.number);
446  buffer = CopyString(buffer, last, textbuf);
447  break;
448 
449  case SPT_DATE: {
450  Date d(params->parms[n - 1].u.dmy);
451  buffer = CopyString(buffer, last, GetDateString(d));
452  break;
453  }
454 
455  default: NOT_REACHED();
456  }
457  }
458  while (*ptr != '\0' && *ptr != '%') ptr++; // Skip to the next '%'
459  if (*ptr == '\0') break;
460  ptr++;
461  }
462  *buffer = '\0';
463  if (params != nullptr) params->set_mode = false; // Clean parameters on next Set.
464 }
