79 void FenceGui::UpdateWidgetSize(WidgetNumber wid_num, BaseWidget *wid)

81  switch (wid_num) {

83  wid->resize_y = GetTextHeight();

85 
86  for (uint8 i = (uint8)FENCE_TYPE_BUILDABLE_BEGIN; i < (uint8)FENCE_TYPE_COUNT; i++) {

90  int width = fence->sprites[FENCE_NE_FLAT]->width;

92  }
93  break;
94  }

97 void FenceGui::DrawWidget(WidgetNumber wid_num, const BaseWidget *wid) const
98 {
99  switch (wid_num) {
100  case FENCE_GUI_LIST: {
101  Point32 rect;
102  int lines = -1; // computed later when we know sprite size
103  rect.x = this->GetWidgetScreenX(wid);
104  rect.y = this->GetWidgetScreenY(wid);
105  for (uint8 i = FENCE_TYPE_BUILDABLE_BEGIN; i < FENCE_TYPE_COUNT; i++) {
106  const Fence *fence = _sprite_manager.GetFence((FenceType)i);
107  if (fence == nullptr) continue;
108 
109  const ImageData *sprite = fence->sprites[FENCE_NE_FLAT];
110  if (lines == -1) {
111  lines = wid->pos.height / sprite->height;
112  }
113 
114  if (lines <= 0) break;
115  lines--;
116 
117  Recolouring recolouring;
118  _video.BlitImage({rect.x, rect.y - sprite->yoffset}, sprite, recolouring, GS_NORMAL);
119  rect.y += sprite->height;
120  }
121  break;
122  }
123  }
124 }

126 void FenceGui::OnClick(WidgetNumber number, const Point16 &pos)
127 {
128  switch (number) {
129  case FENCE_GUI_LIST: {
130  BaseWidget *wid = GetWidget<BaseWidget>(number);
131  Point16 mouse = _window_manager.GetMousePosition();
132  Point32 rect;
133  int lines = -1; // computed later when we know sprite size
134  rect.x = this->GetWidgetScreenX(wid);
135  rect.y = this->GetWidgetScreenY(wid);
136  for (uint8 i = FENCE_TYPE_BUILDABLE_BEGIN; i < FENCE_TYPE_COUNT; i++) {
137  const Fence *fence = _sprite_manager.GetFence((FenceType)i);
138  if (fence == nullptr) continue;
139 
140  uint16 sprite_height = fence->sprites[FENCE_NE_FLAT]->height;
141  if (lines == -1) {
142  lines = wid->pos.height / sprite_height;
143  }
144 
145  if (lines <= 0) break;
146  lines--;
147 
148  if (mouse.x >= rect.x && mouse.x <= rect.x + wid->pos.width &&
149  mouse.y >= rect.y && mouse.y <= rect.y + sprite_height) {
150  this->OnClickFence(fence);
151  return;
152  }
153 
154  rect.y += sprite_height;
155  }
156  break;
157  }
158 
159  default:
160  break;
161  }
162 }
