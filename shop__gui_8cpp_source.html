120 ShopManagerWindow::ShopManagerWindow(ShopInstance *ri) : GuiWindow(WC_SHOP_MANAGER, ri->GetIndex())
121 {

152 void ShopManagerWindow::SetWidgetStringParameters(WidgetNumber wid_num) const
153 {
154  switch (wid_num) {
155  case SMW_TITLEBAR:
156  _str_params.SetUint8(1, this->shop->name);
157  break;
158 
159  case SMW_ITEM1_COST:
160  case SMW_ITEM2_COST: {
161  const ShopType *st = this->shop->GetShopType();
162  _str_params.SetMoney(1, st->item_cost[wid_num - SMW_ITEM1_COST]);
163  break;
164  }
165 
166  case SMW_ITEM1_SELL:
167  case SMW_ITEM2_SELL:
168  _str_params.SetMoney(1, this->shop->GetSaleItemPrice(wid_num - SMW_ITEM1_SELL));
169  break;
170 
171  case SMW_ITEM1_PROFIT:
172  case SMW_ITEM2_PROFIT: {
173  const ShopType *st = this->shop->GetShopType();
174  const Money &cost = st->item_cost[wid_num - SMW_ITEM1_PROFIT];
175  const Money &sell = this->shop->GetSaleItemPrice(wid_num - SMW_ITEM1_PROFIT);
176  _str_params.SetMoney(1, sell - cost);
177  break;
178  }
179 
180  case SMW_ITEM1_COUNT:
181  case SMW_ITEM2_COUNT:
182  _str_params.SetNumber(1, this->shop->item_count[wid_num - SMW_ITEM1_COUNT]);
183  break;
184 
185  case SMW_SELL_PROFIT:
186  _str_params.SetMoney(1, this->shop->total_sell_profit);
187  break;
188 
189  case SMW_SHOP_COST:
190  _str_params.SetMoney(1, this->shop->total_profit - this->shop->total_sell_profit);
191  break;
192 
193  case SMW_TOTAL_PROFIT:
194  _str_params.SetMoney(1, this->shop->total_profit);
195  break;
196  }
197 }

199 void ShopManagerWindow::OnClick(WidgetNumber wid_num, const Point16 &pos)
200 {
201  switch (wid_num) {
202  case SMW_SHOP_OPENED_TEXT:
203  case SMW_SHOP_OPENED:
204  if (this->shop->state != RIS_OPEN) {
205  this->shop->OpenRide();
206  this->SetShopToggleButtons();
207  }
208  break;
209 
210  case SMW_SHOP_CLOSED_TEXT:
211  case SMW_SHOP_CLOSED:
212  if (this->shop->state != RIS_CLOSED) {
213  this->shop->CloseRide();
214  this->SetShopToggleButtons();
215  }
216  break;
217 
218  case SMW_RECOLOUR1:
219  case SMW_RECOLOUR2:
220  case SMW_RECOLOUR3: {
221  RecolourEntry *re = &this->shop->recolours.entries[wid_num - SMW_RECOLOUR1];
222  if (re->IsValid()) {
223  this->ShowRecolourDropdown(wid_num, re, COL_RANGE_DARK_RED);
224  }
225  break;
226  }
227  }
228 }
