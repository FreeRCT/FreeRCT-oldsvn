308 void RideInstance::HandleBreakdown()
309 {
310  if (this->state != RIS_OPEN) return;
311 
312  switch (this->breakdown_state) {
313  case BDS_UNOPENED:
314  case BDS_BROKEN:
315  break;
316 
317  case BDS_NEEDS_NEW_CTR:
318  this->breakdown_ctr = this->rnd.Exponential(this->reliability);
319  this->breakdown_state = BDS_WILL_BREAK;
320  break;
321 
322  case BDS_WILL_BREAK:
323  this->breakdown_ctr--;
324  break;
325 
326  default:
327  NOT_REACHED();
328  }
329 
330  if (this->breakdown_ctr <= 0) {
331  this->breakdown_state = BDS_BROKEN;
332  /* \todo Call a mechanic. */
333  }
334 }

340 void RideInstance::OpenRide()
341 {
342  assert(this->state == RIS_CLOSED);
343  this->state = RIS_OPEN;
344  if (this->breakdown_state == BDS_UNOPENED) {
345  this->breakdown_ctr = this->rnd.Exponential(this->reliability) + BREAKDOWN_GRACE_PERIOD;
346  this->breakdown_state = BDS_WILL_BREAK;
347  }
348 
349  /* Perform payments if they have not been done this month. */
350  bool money_paid = false;
351  if (GB(this->flags, RIF_MONTHLY_PAID, 1) == 0) {
352  this->total_profit -= this->type->monthly_cost;
353  _finances_manager.PayStaffWages(this->type->monthly_cost);
354  SB(this->flags, RIF_MONTHLY_PAID, 1, 1);
355  money_paid = true;
356  }
357  if (GB(this->flags, RIF_OPENED_PAID, 1) == 0) {
358  this->total_profit -= this->type->monthly_open_cost;
359  _finances_manager.PayStaffWages(this->type->monthly_open_cost);
360  SB(this->flags, RIF_OPENED_PAID, 1, 1);
361  money_paid = true;
362  }
363  if (money_paid) NotifyChange(WC_SHOP_MANAGER, this->GetIndex(), CHG_DISPLAY_OLD, 0);
364 }

531 void RidesManager::NewInstanceAdded(uint16 num)
532 {
533  RideInstance *ri = this->GetRideInstance(num);
534  const RideType *rt = ri->GetRideType();
535  assert(ri->state == RIS_ALLOCATED);
536 
537  /* Find a new name for the instance. */
538  const StringID *names = rt->GetInstanceNames();
539  assert(names != nullptr && names[0] != STR_INVALID); // Empty array of names loops forever below.
540  int idx = 0;
541  int shop_num = 1;
542  for (;;) {
543  if (names[idx] == STR_INVALID) {
544  shop_num++;
545  idx = 0;
546  }
547 
548  /* Construct a new name. */
549  if (shop_num == 1) {
550  DrawText(rt->GetString(names[idx]), ri->name, lengthof(ri->name));
551  } else {
552  _str_params.SetStrID(1, rt->GetString(names[idx]));
553  _str_params.SetNumber(2, shop_num);
554  DrawText(GUI_NUMBERED_INSTANCE_NAME, ri->name, lengthof(ri->name));
555  }
556 
557  if (this->FindRideByName(ri->name) == nullptr) break;
558 
559  idx++;
560  }
561 
562  /* Initialize money and counters. */
563  ri->total_profit = 0;
564  ri->total_sell_profit = 0;
565  for (int i = 0; i < NUMBER_ITEM_TYPES_SOLD; i++) {
566  ri->item_price[i] = rt->item_cost[i] * 12 / 10; // Make 20% profit.
567  ri->item_count[i] = 0;
568  }
569 
570  switch (ri->GetKind()) {
571  case RTK_SHOP:
572  ri->CloseRide();
573  break;
574 
575  case RTK_COASTER:
576  ri->BuildRide();
577  break;
578 
579  default:
580  NOT_REACHED();
581  }
582 }
