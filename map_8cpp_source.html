180 bool VoxelStack::MakeVoxelStack(int16 new_base, uint16 new_height)
181 {
182  /* Make sure the voxels live between 0 and WORLD_Z_SIZE. */
183  if (new_base < 0 || new_base + (int)new_height > WORLD_Z_SIZE) return false;
184 
185  Voxel *new_voxels = MakeNewVoxels(new_height);
186  assert(this->height == 0 || (this->base >= new_base && this->base + this->height <= new_base + new_height));
187  CopyStackData(new_voxels + (this->base - new_base), this->voxels, this->height, true);
188 
189  delete[] this->voxels;
190  this->voxels = new_voxels;
191  this->height = new_height;
192  this->base = new_base;
193  return true;
194 }

328 void VoxelStack::MoveStack(VoxelStack *vs)
329 {
330  /* Clean up the stack a bit before copying it, and get lowest and highest non-empty voxel. */
331  int vs_first = 0;
332  int vs_last = 0;
333  for (int i = 0; i < (int)vs->height; i++) {
334  Voxel *v = &vs->voxels[i];
335  assert(!v->HasVoxelObjects()); // There should be no voxel objects in the stack being moved.
336 
337  if (!v->IsEmpty()) {
338  vs_last = i;
339  } else {
340  if (vs_first == i) vs_first++;
341  }
342  }
343 
344  /* There should be at least one surface voxel. */
345  assert(vs_first <= vs_last);
346 
347  /* Examine current stack with respect to persons. */
348  int old_first = 0;
349  int old_last = 0;
350  for (int i = 0; i < (int)this->height; i++) {
351  const Voxel *v = &this->voxels[i];
352  if (v->HasVoxelObjects()) {
353  old_last = i;
354  } else {
355  if (old_first == i) old_first++;
356  }
357  }
358 
359  int new_base = std::min(vs->base + vs_first, this->base + old_first);
360  int new_height = std::max(vs->base + vs_last, this->base + old_last) - new_base + 1;
361  assert(new_base >= 0);
362 
363  /* Make a new stack. Copy new surface, then copy the persons. */
364  Voxel *new_voxels = MakeNewVoxels(new_height);
365  CopyStackData(new_voxels + (vs->base + vs_first) - new_base, vs->voxels + vs_first, vs_last - vs_first + 1, false);
366  int i = (this->base + old_first) - new_base;
367  while (old_first <= old_last) {
368  CopyVoxelObjectList(&new_voxels[i], &this->voxels[old_first]);
369  i++;
370  old_first++;
371  }
372 
373  this->base = new_base;
374  this->height = new_height;
375  delete[] this->voxels;
376  this->voxels = new_voxels;
377 }

495 void UpdateLandBorderFence(uint16 x, uint16 y, uint16 width, uint16 height)
496 {
497  /*
498  * Iterate over given rectangle plus one tile, unless the map border
499  * is reached.
500  */
501  uint16 x_min = x > 0 ? x-1 : x;
502  uint16 y_min = y > 0 ? y-1 : y;
503  uint16 x_max = std::min(_world.GetXSize()-1, x + width + 1);
504  uint16 y_max = std::min(_world.GetYSize()-1, y + height + 1);
505 
506  for (uint16 ix = x_min; ix < x_max; ix++) {
507  for (uint16 iy = y_min; iy < y_max; iy++) {
508  VoxelStack *vs = _world.GetModifyStack(ix, iy);
509  TileOwner owner = vs->owner;
510 
511  for (int16 i = vs->height - 1; i >= 0; i--) {
512  Voxel &v = vs->voxels[i];
513 
514  /* Only update border fence on the surface. */
515  if (v.GetGroundType() == GTP_INVALID) continue;
516  TileSlope slope = ExpandTileSlope(v.GetGroundSlope());
517  if ((slope & TSB_TOP) != 0) continue; // Handle steep slopes at the bottom voxel
518 
519  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
520  bool fence;
521  if (owner == OWN_PARK) {
522  fence = false;
523  } else {
524  if ((ix == 0 && _tile_dxy[edge].x == -1) || (iy == 0 && _tile_dxy[edge].y == -1)) continue;
525  uint16 neighbour_x = ix + _tile_dxy[edge].x;
526  uint16 neighbour_y = iy + _tile_dxy[edge].y;
527  if (neighbour_x >= _world.GetXSize() || neighbour_y >= _world.GetYSize()) continue;
528  TileOwner neighbour_owner = _world.GetTileOwner(neighbour_x, neighbour_y);
529  fence = neighbour_owner == OWN_PARK;
530  }
531 
532  /* Only set/unset land border fence if the edge doesn't have some other fence type. */
533  Voxel *fence_voxel = StoreFenceInUpperVoxel(slope, edge) ? vs->GetCreate(i + 1, true) : &v;
534  FenceType curr_type = fence_voxel != nullptr ? fence_voxel->GetFenceType(edge) : FENCE_TYPE_INVALID;
535  if (curr_type != (fence ? FENCE_TYPE_LAND_BORDER : FENCE_TYPE_INVALID)) {
536  fence_voxel->SetFenceType(edge, fence ? FENCE_TYPE_LAND_BORDER : FENCE_TYPE_INVALID);
537  }
538  }
539 
540  /* There are no fences below ground. */
541  break;
542  }
543  }
544  }
545 }
