77 int main(int argc, char *argv[])
78 {
79  GetOptData opt_data(argc - 1, argv + 1, _options);
80  const char *header = nullptr;
81  const char *code = nullptr;
82  const char *prefix = nullptr;
83  const char *base = "0";

85  int opt_id;
86  do {
87  opt_id = opt_data.GetOpt();
88  switch (opt_id) {
89  case 'h':
90  PrintUsage();
91  exit(0);
92 
93  case 'd':
94  header = opt_data.opt;
95  break;
96 
97  case 'c':
98  code = opt_data.opt;
99  break;
100 
101  case 'b':
102  base = opt_data.opt;
103  break;
104 
105  case 'p':
106  prefix = opt_data.opt;
107  break;
108 
109  case -1:
110  break;
111 
112  default:
113  /* -2 or some other weird thing happened. */
114  fprintf(stderr, "ERROR while processing the command-line\n");
115  exit(1);
116  }
117  } while (opt_id != -1);
118 
119  if (prefix != nullptr) {
120  /* Prefix specified, generate strings. */
121  if (header != nullptr) GenerateStringsHeaderFile(prefix, base, header);
122  if (code != nullptr) GenerateStringsCodeFile(prefix, code);
123  exit(0);
124  }
125 
126  /* No --prefix, generate an RCD file. */
127  if (header != nullptr) printf("Warning: --header option is not used.\n");
128  if (code != nullptr) printf("Warning: --code option is not used.\n");
129 
130  int num_files = std::max(1, opt_data.numleft);
131  for (int i = 0; i < num_files; i++) {
132  /* Phase 1: Parse the input file. */
133  std::shared_ptr<NamedValueList> nvs = LoadFile((i < opt_data.numleft) ? opt_data.argv[i] : nullptr);
134 
135  /* Phase 2: Check and simplify the loaded input. */
136  FileNodeList *file_nodes = CheckTree(nvs);
137  nvs = nullptr;
138 
139  /* Phase 3: Construct output files. */
140  for (auto iter : file_nodes->files) {
141  FileWriter fw;
142  iter->Write(&fw);
143  fw.WriteFile(iter->file_name);
144  }
145 
146  delete file_nodes;
147  }
148  exit(0);
149 }
