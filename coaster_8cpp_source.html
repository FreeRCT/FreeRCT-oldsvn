51 bool CarType::Load(RcdFileReader *rcdfile, const ImageMap &sprites)
52 {
53  if (rcdfile->version != 2 || rcdfile->size != 2 + 2 + 4 + 4 + 2 + 2 + 16384) return false;
54 
55  this->tile_width = rcdfile->GetUInt16();
56  if (this->tile_width != 64) return false; // Do not allow anything else than 64 pixels tile width.
57 
58  this->z_height = rcdfile->GetUInt16();
59  if (this->z_height != this->tile_width / 4) return false;
60 
61  this->car_length = rcdfile->GetUInt32();
62  if (this->car_length > 65535) return false; // Assumption is that a car fits in a single tile, at least some of the time.
63 
64  this->inter_car_length = rcdfile->GetUInt32();
65  this->num_passengers = rcdfile->GetUInt16();
66  this->num_entrances = rcdfile->GetUInt16();
67  if (this->num_entrances == 0 || this->num_entrances > 4) return false; // Seems like a nice arbitrary upper limit on the number of rows of a car.
68  uint16 pass_per_row = this->num_passengers / this->num_entrances;
69  if (this->num_passengers != pass_per_row * this->num_entrances) return false;
70 
71  for (uint i = 0; i < 4096; i++) {
72  if (!LoadSpriteFromFile(rcdfile, sprites, &this->cars[i])) return false;
73  }
74  return true;
75 }

93 bool CoasterType::Load(RcdFileReader *rcd_file, const TextMap &texts, const TrackPiecesMap &piece_map)
94 {
95  uint32 length = rcd_file->size;
96  if (rcd_file->version != 5 || length < 2 + 1 + 1 + 1 + 4 + 2) return false;
97  length -= 2 + 1 + 1 + 1 + 4 + 2;
98  this->coaster_kind = rcd_file->GetUInt16();
99  this->platform_type = rcd_file->GetUInt8();
100  this->max_number_trains = rcd_file->GetUInt8();
101  this->max_number_cars = rcd_file->GetUInt8();
102  if (this->coaster_kind == 0 || this->coaster_kind >= CST_COUNT) return false;
103  if (this->platform_type == 0 || this->platform_type >= CPT_COUNT) return false;
104 
105  TextData *text_data;
106  if (!LoadTextFromFile(rcd_file, texts, &text_data)) return false;
107  StringID base = _language.RegisterStrings(*text_data, _coasters_strings_table);
108  this->SetupStrings(text_data, base, STR_GENERIC_COASTER_START, COASTERS_STRING_TABLE_END, COASTERS_NAME_TYPE, COASTERS_DESCRIPTION_TYPE);
109 
110  int piece_count = rcd_file->GetUInt16();
111  if (length != 4u * piece_count) return false;
112 
113  this->pieces.resize(piece_count);
114  for (auto &piece : this->pieces) {
115  uint32 val = rcd_file->GetUInt32();
116  if (val == 0) return false; // We don't expect missing track pieces (they should not be included at all).
117  auto iter = piece_map.find(val);
118  if (iter == piece_map.end()) return false;
119  piece = (*iter).second;
120  }
121  /* Setup a track voxel list for fast access in the type. */
122  for (const auto &piece : this->pieces) {
123  this->voxels.insert(this->voxels.end(), piece->track_voxels.begin(), piece->track_voxels.end());
124  }
125  return true;
126 }

235 void DisplayCoasterCar::Set(const XYZPoint16 &vox_pos, const XYZPoint16 &pix_pos, uint8 pitch, uint8 roll, uint8 yaw)
236 {
237  bool change_voxel = this->vox_pos != vox_pos;
238 
239  if (!change_voxel && this->pix_pos == pix_pos && this->pitch == pitch && this->roll == roll && this->yaw == yaw) {
240  return; // Nothing changed.
241  }
242 
243  if (this->yaw != 0xff && change_voxel) {
244  /* Valid data, and changing voxel -> remove self from the old voxel. */
245  this->MarkDirty();
246  Voxel *v = _world.GetCreateVoxel(this->vox_pos, false);
247  this->RemoveSelf(v);
248  }
249 
250  /* Update voxel and orientation. */
251  this->vox_pos = vox_pos;
252  this->pix_pos = pix_pos;
253  this->pitch = pitch;
254  this->roll = roll;
255  this->yaw = yaw;
256 
257  if (this->yaw != 0xff) {
258  this->MarkDirty(); // Voxel or orientation has changed, repaint the possibly new voxel.
259 
260  if (change_voxel) { // With a really new voxel, also add self to the new voxel.
261  Voxel *v = _world.GetCreateVoxel(this->vox_pos, false);
262  this->AddSelf(v);
263  }
264  }
265 }

318 static void inline Unroll(uint roll, int32 *dy, int32 *dz)
319 {
320  /* Rotates around X axis, thus dx does not change. */
321  int32 new_dy = *dy * _sin_cos_table[roll + 4] - *dz * _sin_cos_table[roll];
322  int32 new_dz = *dy * _sin_cos_table[roll] + *dz * _sin_cos_table[roll + 4];
323  *dy = new_dy;
324  *dz = new_dz;
325 }

331 void CoasterTrain::OnAnimate(int delay)
332 {
333  if (this->speed >= 0) {
334  this->back_position += this->speed * delay;
335  if (this->back_position >= this->coaster->coaster_length) {
336  this->back_position -= this->coaster->coaster_length;
337  this->cur_piece = this->coaster->pieces;
338  }
339  while (this->cur_piece->distance_base + this->cur_piece->piece->piece_length < this->back_position) this->cur_piece++;
340  } else {
341  uint32 change = -this->speed * delay;
342  if (change > this->back_position) {
343  this->back_position = this->back_position + this->coaster->coaster_length - change;
344  /* Update the track piece as well. There is no simple way to get
345  * the last piece, so moving from the front will have to do. */
346  this->cur_piece = this->coaster->pieces;
347  while (this->cur_piece->distance_base + this->cur_piece->piece->piece_length < this->back_position) this->cur_piece++;
348  } else {
349  this->back_position -= change;
350  while (this->cur_piece->distance_base > this->back_position) this->cur_piece--;
351  }
352  }
353  uint32 car_length = this->coaster->car_type->car_length;
354  uint32 position = this->back_position; // Back position of the train / last car.
355  const PositionedTrackPiece *ptp = this->cur_piece;
356  for (uint i = 0; i < this->cars.size(); i++) {
357  CoasterCar &car = this->cars[i];
358  if (position >= this->coaster->coaster_length) {
359  position -= this->coaster->coaster_length;
360  ptp = this->coaster->pieces;
361  }
362  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
363 
364  /* Get position of the back of the car. */
365  int32 xpos_back = ptp->piece->car_xpos->GetValue(position - ptp->distance_base) + (ptp->base_voxel.x << 8);
366  int32 ypos_back = ptp->piece->car_ypos->GetValue(position - ptp->distance_base) + (ptp->base_voxel.y << 8);
367  int32 zpos_back = ptp->piece->car_zpos->GetValue(position - ptp->distance_base) * 2 + (ptp->base_voxel.z << 8);
368 
369  /* Get roll from the center of the car. */
370  position += car_length / 2;
371  if (position >= this->coaster->coaster_length) {
372  position -= this->coaster->coaster_length;
373  ptp = this->coaster->pieces;
374  }
375  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
376  uint roll = static_cast<uint>(ptp->piece->car_roll->GetValue(position - ptp->distance_base) + 0.5) & 0xf;
377 
378  /* Get position of the front of the car. */
379  position += car_length / 2;
380  if (position >= this->coaster->coaster_length) {
381  position -= this->coaster->coaster_length;
382  ptp = this->coaster->pieces;
383  }
384  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
385  int32 xpos_front = ptp->piece->car_xpos->GetValue(position - ptp->distance_base) + (ptp->base_voxel.x << 8);
386  int32 ypos_front = ptp->piece->car_ypos->GetValue(position - ptp->distance_base) + (ptp->base_voxel.y << 8);
387  int32 zpos_front = ptp->piece->car_zpos->GetValue(position - ptp->distance_base) * 2 + (ptp->base_voxel.z << 8);
388 
389  int32 xder = xpos_front - xpos_back;
390  int32 yder = ypos_front - ypos_back;
391  int32 zder = (zpos_front - zpos_back) / 2; // Tile height is half the width.
392 
393  int32 xpos_middle = xpos_back + xder / 2; // Compute center point of the car as position to render the car.
394  int32 ypos_middle = ypos_back + yder / 2;
395  int32 zpos_middle = zpos_back + zder;
396 
397  float total_speed = sqrt(xder * xder + yder * yder + zder * zder);
398  /* Gravity */
399  this->speed -= zder / total_speed * 9.8;
400 
403  if (this->speed < 65536 / 1000 && (this->cur_piece->piece->HasPower() || this->cur_piece->piece->HasPlatform())) {
404  this->speed = 65536 / 1000;
405  }
406 
407  /* Unroll the orientation vector. */
408  Unroll(roll, &yder, &zder);
409  float horizontal_speed = std::hypot(xder, yder);
410 
411  static const double TAN11_25 = 0.198912367379658; // tan(11.25 degrees).
412  static const double TAN33_75 = 0.6681786379192989; // tan(3*11.25 degrees).
413 
414  /* Compute pitch. */
415  bool swap_dz = false;
416  if (zder < 0) {
417  swap_dz = true;
418  zder = -zder;
419  }
420 
421  uint pitch;
422  if (horizontal_speed < zder) {
423  if (horizontal_speed < zder * TAN11_25) {
424  pitch = 4;
425  } else if (horizontal_speed < zder * TAN33_75) {
426  pitch = 3;
427  } else {
428  pitch = 2;
429  }
430  } else {
431  if (zder < horizontal_speed * TAN11_25) {
432  pitch = 0;
433  } else if (zder < horizontal_speed * TAN33_75) {
434  pitch = 1;
435  } else {
436  pitch = 2;
437  }
438  }
439  if (swap_dz) pitch = (16 - pitch) & 0xf;
440 
441  /* Compute yaw. */
442  bool swap_dx = false;
443  if (xder > 0) {
444  swap_dx = true;
445  xder = -xder;
446  }
447  bool swap_dy = false;
448  if (yder > 0) {
449  swap_dy = true;
450  yder = -yder;
451  }
452  uint yaw;
453  /* There are 16 yaw directions, the 360 degrees needs to be split in 16 pieces.
454  * However the x and y axes are in the middle of a piece, so 360 degrees is split
455  * in 32 parts, and 2 parts form one piece. */
456  if (xder < yder) {
457  /* In the first 45 degrees. It is split in 4 parts (4*11.25 degrees)
458  * where the 1st part is for direction 0. The 2nd and 3rd part are for direction 1,
459  * and the 4th part is for direction 2. */
460  if (xder * TAN11_25 < yder) {
461  yaw = 0;
462  } else if (xder * TAN33_75 < yder) {
463  yaw = 1;
464  } else {
465  yaw = 2;
466  }
467  } else {
468  /* In the second 45 degrees. It is also split in 4 parts (4*11.25 degrees)
469  * where the 1st part is for direction 2. The 2nd and 3rd part are for direction 3,
470  * and the 4th part is for direction 4.
471  *
472  * Rather than re-inventing a solution, re-use the same checks as
473  * above with swapped xder and yder. */
474  if (yder * TAN11_25 < xder) {
475  yaw = 4;
476  } else if (yder * TAN33_75 < xder) {
477  yaw = 3;
478  } else {
479  yaw = 2;
480  }
481  }
482  if (swap_dx) yaw = 8 - yaw;
483  if (swap_dy) yaw = (16 - yaw) & 0xf;
484 
485  xpos_back &= 0xFFFFFF00; // Back and front positions to the north bottom corner of the voxel.
486  ypos_back &= 0xFFFFFF00;
487  zpos_back &= 0xFFFFFF00;
488  xpos_front &= 0xFFFFFF00;
489  ypos_front &= 0xFFFFFF00;
490  zpos_front &= 0xFFFFFF00;
491  XYZPoint16 back( xpos_back >> 8, ypos_back >> 8, zpos_back >> 8);
492  XYZPoint16 front(xpos_front >> 8, ypos_front >> 8, zpos_front >> 8);
493 
494  XYZPoint16 back_pix( xpos_middle - xpos_back, ypos_middle - ypos_back, zpos_middle - zpos_back);
495  XYZPoint16 front_pix(xpos_middle - xpos_front, ypos_middle - ypos_front, zpos_middle - zpos_front);
496 
497  car.back.Set (back, back_pix, pitch, roll, yaw);
498  car.front.Set(front, front_pix, pitch, roll, yaw);
499  position += this->coaster->car_type->inter_car_length;
500  }
501 }

687 bool CoasterInstance::MakePositionedPiecesLooping(bool *modified)
688 {
689  if (modified != nullptr) *modified = false;
690 
691  /* First step, move all non-null track pieces to the start of the array. */
692  int count = 0;
693  for (int i = 0; i < this->capacity; i++) {
694  PositionedTrackPiece *ptp = this->pieces + i;
695  if (ptp->piece == nullptr) continue;
696  if (i == count) {
697  count++;
698  continue;
699  }
700  std::swap(this->pieces[count], *ptp);
701  if (modified != nullptr) *modified = true;
702  ptp->piece = nullptr;
703  count++;
704  }
705 
706  /* Second step, find a loop from start to end. */
707  if (count < 2) return false; // 0 or 1 positioned pieces won't ever make a loop.
708 
709  PositionedTrackPiece *ptp = this->pieces;
710  uint32 distance = 0;
711  if (ptp->distance_base != distance) {
712  if (modified != nullptr) *modified = true;
713  ptp->distance_base = distance;
714  }
715  distance += ptp->piece->piece_length;
716 
717  for (int i = 1; i < count; i++) {
718  int j = this->FindSuccessorPiece(ptp->GetEndXYZ(), ptp->piece->exit_connect, i, count);
719  if (j < 0) return false;
720  ptp++; // Now points to pieces[i].
721  if (i != j) {
722  std::swap(*ptp, this->pieces[j]); // Make piece 'j' the next positioned piece.
723  if (modified != nullptr) *modified = true;
724  }
725  if (ptp->distance_base != distance) {
726  if (modified != nullptr) *modified = true;
727  ptp->distance_base = distance;
728  }
729  distance += ptp->piece->piece_length;
730  }
731  this->coaster_length = distance;
732  return this->pieces[0].CanBeSuccessor(*ptp);
733 }
