1082  for (uint8 y = 0; y < this->num_rows; y++) {
1083  this->rows[y].InitRowColData();
1084  }
1085  for (uint8 x = 0; x < this->num_cols; x++) {
1086  this->columns[x].InitRowColData();
1087  }
1088 
1089  /* Step 2: Process child widgets. */
1090  for (uint8 y = 0; y < this->num_rows; y++) {
1091  for (uint8 x = 0; x < this->num_cols; x++) {
1092  BaseWidget *bw = this->childs[y * (uint16)this->num_cols + x];
1093  bw->SetupMinimalSize(w, wid_array);
1094  this->rows[y].Merge(bw->min_y, bw->fill_y, bw->resize_y);
1095  this->columns[x].Merge(bw->min_x, bw->fill_x, bw->resize_x);
1096  }
1097  }
1098 
1099  /* Step 3a: Handle vertical equal size. */
1100  uint16 max_minsize = 0;
1101  if ((this->flags & EQS_VERTICAL) != 0) { // Equal sizes vertically requested, do a pre-size computation.
1102  for (uint8 y = 0; y < this->num_rows; y++) {
1103  max_minsize = std::max(max_minsize, this->rows[y].min_size);
1104  }
1105  for (uint8 y = 0; y < this->num_rows; y++) {
1106  if (this->rows[y].fill > 0) {
1107  uint16 diff = max_minsize - this->rows[y].min_size;
1108  this->rows[y].min_size += diff - diff % this->rows[y].fill;
1109  }
1110  }
1111  }
1112 
1113  uint16 child_tmp_minsize[50]; // Temporary array for minimal size of children. Size is arbitrary.
1114  assert(this->num_rows < lengthof(child_tmp_minsize));
1115  assert(this->num_cols < lengthof(child_tmp_minsize));

1118  for (uint8 y = 0; y < this->num_rows; y++) {

1122  child_tmp_minsize[x] = bw->min_y;
1123  }

1125  * bigger than the current minimal vertical size.

1234  for (uint8 y = 0; y < this->num_rows; y++) {

1253  continue;
1254  }

1263  max_step = 0;

1268  count++;
