68 void FenceBuildManager::SetCursors()
69 {
70  Viewport *vp = GetViewport();
71  if (vp == nullptr) return;
72 
73  FinderData fdata(CS_GROUND_EDGE, FW_EDGE);
74  if (vp->ComputeCursorPosition(&fdata) != CS_NONE && fdata.cursor >= CUR_TYPE_EDGE_NE && fdata.cursor <= CUR_TYPE_EDGE_NW) {
75  TileEdge edge = (TileEdge)((uint8)fdata.cursor - (uint8)CUR_TYPE_EDGE_NE);
76  const Voxel *v = _world.GetVoxel(fdata.voxel_pos);
77  assert(v != nullptr);
78  TileSlope slope = ExpandTileSlope(v->GetGroundSlope());
79  /*
80  * Fence for steep slopes and fully raised edges go into the voxel above.
81  * Click on steep slope already have the upper edge z given in fdata, but
82  * for clicking on upper edge, we need to get the voxel above the one
83  * clicked.
84  */
85  int32 extra_z = 0;
86  if ((slope & TSB_TOP) == 0 && IsRaisedEdge(edge, slope)) {
87  extra_z = 1;
88  v = _world.GetCreateVoxel(fdata.voxel_pos + XYZPoint16(0, 0, extra_z), true);
89  assert(v != nullptr);
90  }
91  const SpriteStorage *ss = _sprite_manager.GetSprites(vp->tile_width);
92  assert(ss != nullptr);
93  const ImageData *sprite = ss->GetFenceSprite(this->selected_fence_type, edge, ExpandTileSlope(v->GetGroundSlope()), vp->orientation);
94  /*
95  * The lower edges of steep slope needs an yoffset to be rendered
96  * as if they belonged to the lower voxel.
97  */
98  bool steep_lower_edge = (slope & TSB_STEEP) != 0 &&
99  (slope & (1 << edge)) == 0 &&
100  (slope & (1 << ((edge + 1) % 4))) == 0;
101  fdata.voxel_pos.z += extra_z;
102  vp->edge_cursor.SetCursor(fdata.voxel_pos, fdata.cursor, sprite, steep_lower_edge ? vp->tile_height : 0);
103  }
104 }

131 void FenceBuildManager::OnMouseButtonEvent(Viewport *vp, uint8 state)
132 {
133  this->mouse_state = state & MB_CURRENT;
134 
135  if ((this->mouse_state & MB_LEFT) != 0) { // Left-click -> build fence
136  EdgeCursor *c = &vp->edge_cursor;
137  if (_game_mode_mgr.InPlayMode() && _world.GetTileOwner(c->cursor_pos.x, c->cursor_pos.y) != OWN_PARK) return;
138 
139  TileEdge edge = (TileEdge)((uint8)c->type - (uint8)CUR_TYPE_EDGE_NE);
140  VoxelStack *vs = _world.GetModifyStack(c->cursor_pos.x, c->cursor_pos.y);
141  Voxel *v = vs->GetCreate(c->cursor_pos.z, false);
142  assert(v != nullptr);
143  TileSlope slope = ExpandTileSlope(v->GetGroundSlope());
144  int32 extra_z = 0;
145  if ((slope & TSB_TOP) == 0 && IsRaisedEdge(edge, slope)) {
146  extra_z = 1;
147  v = vs->GetCreate(c->cursor_pos.z + extra_z, true);
148  assert(v != nullptr);
149  }
150  v->SetFenceType(edge, this->selected_fence_type);
151  vp->MarkVoxelDirty(c->cursor_pos + XYZPoint16(0, 0, extra_z));
152  }
153 }
