134 void Weather::OnNewDay()
135 {
136  this->SetTemperature();
137 
138  if (this->change != 0) {
139  this->current += this->change;
140  if ((this->change > 0 && this->next <= this->current) || (this->change < 0 && this->next >= this->current)) {
141  this->current = this->next;
142  this->change = 0;
143  }
144  }
145 
146  if (_date.day != 12 && _date.day != 27) return;
147  int month = (_date.day == 12) ? _date.month : _date.GetNextMonth();
148  this->next = _yearly_weather[month - 1].Draw();
149  if (this->current == this->next) return;
150  this->change = (this->next - this->current) / 5;
151  if (this->change == 0) this->change = (this->next - this->current > 0) ? 1 : -1;
152 }

164 void Weather::SetTemperature()
165 {
166  if (_date.day <= 15) {
167  int prev_month = _date.GetPreviousMonth();
168  int prev_length = _days_per_month[prev_month] - 15; // 16, 17, ...
169  prev_length += _date.day - 1;
170  int this_length = 15 - _date.day;
171  this->temperature = (prev_length * _yearly_weather[_date.month - 1].temp + this_length * _yearly_weather[prev_month - 1].temp) / (prev_length + this_length);
172  } else {
173  int next_month = _date.GetNextMonth();
174  int next_length = 15 + (_days_per_month[_date.month] - _date.day);
175  int this_length = _date.day - 15;
176  this->temperature = (next_length * _yearly_weather[_date.month - 1].temp + this_length * _yearly_weather[next_month - 1].temp) / (next_length + this_length);
177  }
178 }
