116 static std::shared_ptr<FileNode> ConvertFileNode(std::shared_ptr<NodeGroup> ng)
117 {
118  std::shared_ptr<Expression> argument;
119  ExpandExpressions(ng->exprs, &argument, 1, ng->pos, "file");
120 
121  std::string filename = GetString(argument, 0, "file");
122  auto fn = std::make_shared<FileNode>(filename);
123 
124  /* Add the meta blocks to the start of the file, while collecting game blocks for appending them later. */
125  bool seen_info = false; // Did we encounter an info meta block?
126  std::list<std::shared_ptr<GameBlock>> game_blocks;
127  for (auto &iter : ng->values->values) {
128  std::shared_ptr<NamedValue> nv = std::dynamic_pointer_cast<NamedValue>(iter);
129  assert(nv != nullptr); // Should always hold, as ImportValue has been eliminated.
130  if (nv->name != nullptr) fprintf(stderr, "Warning at %s: Unexpected name encountered, ignoring\n", nv->name->GetPosition().ToString());
131  std::shared_ptr<NodeGroup> ng = std::dynamic_pointer_cast<NodeGroup>(nv->group);
132  if (ng == nullptr) {
133  fprintf(stderr, "Error at %s: Only node groups may be added\n", nv->group->GetPosition().ToString());
134  exit(1);
135  }
136  auto bn = ConvertNodeGroup(ng);
137  auto mb = std::dynamic_pointer_cast<MetaBlock>(bn);
138  if (mb != nullptr) {
139  seen_info |= strcmp(mb->blk_name, "INFO") == 0;
140  fn->blocks.push_back(mb);
141  continue;
142  }
143  auto gb = std::dynamic_pointer_cast<GameBlock>(bn);
144  if (gb == nullptr) {
145  fprintf(stderr, "Error at %s: Only game blocks can be added to a \"file\" node\n", nv->group->GetPosition().ToString());
146  exit(1);
147  }
148  game_blocks.push_back(gb);
149  }
150  if (!seen_info) {
151  fprintf(stderr, "Error at %s: Missing an INFO block in the file\n", ng->pos.ToString());
152  exit(1);
153  }
154 
155  for (auto &gb : game_blocks) fn->blocks.push_back(gb); // Append the game blocks.
156  return fn;

331 static void AssignNames(std::shared_ptr<BlockNode> bn, std::shared_ptr<NameTable> nt, std::vector<std::shared_ptr<ValueInformation>> &vis, int *length)
332 {
333  /* Is a table consisting of a single parameterized name? */
334  if (nt->HasSingleElement()) {
335  std::shared_ptr<IdentifierLine> il = nt->rows.front()->identifiers.front();
336  if (!il->IsValid()) return; // Only available name cannot be used.
337 
338  ParameterizedName parms_name;
339  HorVert hv = parms_name.DecodeName(il->name.c_str(), il->pos);
340  if (hv != HV_NONE) {
341  /* Expand the parameterized name. */
342  int row = 0;
343  do {
344  int col = 0;
345  do {
346  const char *nm = parms_name.GetParmName(row, col);
347  std::shared_ptr<ValueInformation> vi = std::make_shared<ValueInformation>();
348  vi->expr_value = nullptr;
349  vi->node_value = bn->GetSubNode(row, col, nm, il->pos);
350  vi->name = std::string(nm);
351  vi->pos = il->pos;
352  vi->used = false;
353  vis[*length] = vi;
354  (*length)++;
355 
356  col++;
357  } while (parms_name.vert_range.used && col < parms_name.vert_range.size());
358  row++;
359  } while (parms_name.hor_range.used && row < parms_name.hor_range.size());
360  return;
361  }
362  }
363 
364  /* General case, a 2D table with non-parameterized names. */
365  int row = 0;
366  for (auto &nr : nt->rows) {
367  int col = 0;
368  for (auto &il : nr->identifiers) {
369  if (il->IsValid()) {
370  CheckIsSingleName(il->name, il->pos);
371  std::shared_ptr<ValueInformation> vi = std::make_shared<ValueInformation>();
372  vi->expr_value = nullptr;
373  vi->node_value = bn->GetSubNode(row, col, il->name.c_str(), il->pos);
374  vi->name = std::string(il->name);
375  vi->pos = il->pos;
376  vi->used = false;
377  vis[*length] = vi;
378  (*length)++;
379  }
380  col++;
381  }
382  row++;
383  }

459 void Values::PrepareNamedValues(std::shared_ptr<NamedValueList> values, bool allow_named, bool allow_unnamed, const Symbol *symbols)
460 {
461  /* Count number of named and unnamed values. */
462  int named_count = 0;
463  int unnamed_count = 0;
464  for (auto &iter : values->values) {
465  std::shared_ptr<NamedValue> nv = std::dynamic_pointer_cast<NamedValue>(iter);
466  assert(nv != nullptr); // Should always hold, as ImportValue has been eliminated.
467  if (nv->name == nullptr) { // Unnamed value.
468  if (!allow_unnamed) {
469  fprintf(stderr, "Error at %s: Value should have a name\n", nv->group->GetPosition().ToString());
470  exit(1);
471  }
472  unnamed_count++;
473  } else {
474  if (!allow_named) {
475  fprintf(stderr, "Error at %s: Value should not have a name\n", nv->group->GetPosition().ToString());
476  exit(1);
477  }
478  int count = nv->name->GetNameCount();
479  named_count += count;
480  }
481  }
482 
483  this->CreateValues(named_count, unnamed_count);
484 
485  named_count = 0;
486  unnamed_count = 0;
487  for (auto &iter : values->values) {
488  std::shared_ptr<NamedValue> nv = std::dynamic_pointer_cast<NamedValue>(iter);
489  assert(nv != nullptr); // Should always hold, as ImportValue has been eliminated.
490  if (nv->name == nullptr) { // Unnamed value.
491  std::shared_ptr<NodeGroup> ng = std::dynamic_pointer_cast<NodeGroup>(nv->group);
492  if (ng != nullptr) {
493  this->unnamed_values[unnamed_count]->expr_value = nullptr;
494  this->unnamed_values[unnamed_count]->node_value = ConvertNodeGroup(ng);
495  this->unnamed_values[unnamed_count]->name = "???";
496  this->unnamed_values[unnamed_count]->pos = ng->GetPosition();
497  this->unnamed_values[unnamed_count]->used = false;
498  unnamed_count++;
499  continue;
500  }
501  std::shared_ptr<ExpressionGroup> eg = std::dynamic_pointer_cast<ExpressionGroup>(nv->group);
502  assert(eg != nullptr);
503  this->unnamed_values[unnamed_count]->expr_value = eg->expr->Evaluate(symbols);
504  this->unnamed_values[unnamed_count]->node_value = nullptr;
505  this->unnamed_values[unnamed_count]->name = "???";
506  this->unnamed_values[unnamed_count]->pos = ng->GetPosition();
507  this->unnamed_values[unnamed_count]->used = false;
508  unnamed_count++;
509  continue;
510  } else { // Named value.
511  std::shared_ptr<NodeGroup> ng = std::dynamic_pointer_cast<NodeGroup>(nv->group);
512  if (ng != nullptr) {
513  auto bn = ConvertNodeGroup(ng);
514  std::shared_ptr<SingleName> sn = std::dynamic_pointer_cast<SingleName>(nv->name);
515  if (sn != nullptr) {
516  this->named_values[named_count]->expr_value = nullptr;

532  assert(eg != nullptr);
