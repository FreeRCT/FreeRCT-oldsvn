48 void DrawBorderSprites(const BorderSpriteData &bsd, bool pressed, const Rectangle32 &rect, ColourRange colour)
49 {
50  const ImageData * const *spr_base = pressed ? bsd.pressed : bsd.normal;
51  static Recolouring rc; // Only COL_RANGE_BROWN is modified each time.
52  rc.Set(0, RecolourEntry(COL_RANGE_BROWN, colour));
53 
54  Point32 pt = rect.base;
55  int xleft = pt.x;
56  int ytop = pt.y;
57  if (spr_base[WBS_TOP_LEFT] != nullptr) {
58  _video.BlitImage(pt, spr_base[WBS_TOP_LEFT], rc, GS_NORMAL);
59  xleft += spr_base[WBS_TOP_LEFT]->xoffset + spr_base[WBS_TOP_LEFT]->width;
60  ytop += spr_base[WBS_TOP_LEFT]->yoffset + spr_base[WBS_TOP_LEFT]->height;
61  }
62 
63  pt.x = rect.base.x + rect.width - 1;
64  int xright = pt.x;
65  if (spr_base[WBS_TOP_RIGHT] != nullptr) {
66  _video.BlitImage(pt, spr_base[WBS_TOP_RIGHT], rc, GS_NORMAL);
67  xright += spr_base[WBS_TOP_RIGHT]->xoffset;
68  }
69 
70  uint16 numx = xright - xleft;
71  if (spr_base[WBS_TOP_MIDDLE] != nullptr) {
72  numx /= spr_base[WBS_TOP_MIDDLE]->width;
73  } else if (spr_base[WBS_BOTTOM_MIDDLE] != nullptr) {
74  numx /= spr_base[WBS_BOTTOM_MIDDLE]->width;
75  } // else assume sprite width is 1.
76  if (spr_base[WBS_TOP_MIDDLE] != nullptr) _video.BlitHorizontal(xleft, numx, pt.y, spr_base[WBS_TOP_MIDDLE], rc);
77 
78  pt.x = rect.base.x;
79  pt.y = rect.base.y + rect.height - 1;
80  int ybot = pt.y;
81  if (spr_base[WBS_BOTTOM_LEFT] != nullptr) {
82  _video.BlitImage(pt, spr_base[WBS_BOTTOM_LEFT], rc, GS_NORMAL);

84  }
85 
86  uint16 numy = ybot - ytop;
87  if (spr_base[WBS_MIDDLE_LEFT] != nullptr) {
88  numy /= spr_base[WBS_MIDDLE_LEFT]->height;
89  } else if (spr_base[WBS_MIDDLE_RIGHT] != nullptr) {
90  numy /= spr_base[WBS_MIDDLE_RIGHT]->height;
91  } // else assume sprite height is 1.
92  if (spr_base[WBS_MIDDLE_LEFT] != nullptr) _video.BlitVertical(ytop, numy, pt.x, spr_base[WBS_MIDDLE_LEFT], rc);
93 
94  pt.x = rect.base.x + rect.width - 1;
95  pt.y = rect.base.y + rect.height - 1;
96  if (spr_base[WBS_BOTTOM_RIGHT] != nullptr) _video.BlitImage(pt, spr_base[WBS_BOTTOM_RIGHT], rc, GS_NORMAL);
97

99 
100  pt.x = rect.base.x + rect.width - 1;

111 void OverlayShaded(const Rectangle32 &rect)

116  Rectangle32 r(rect);
117  r.RestrictTo(0, 0, _video.GetXSize(), _video.GetYSize());

119 
120  /* Set clipped area to the rectangle. */
121  ClippedRectangle cr(_video.GetClippedRectangle());
122  ClippedRectangle new_cr(cr, r.base.x, r.base.y, r.width, r.height);
123  _video.SetClippedRectangle(new_cr);
124 
125  /* Align the disabled sprite so it becomes a continuous pattern. */
126  int32 base_x = -(r.base.x % img->width);
127  int32 base_y = -(r.base.y % img->height);
128  uint16 numx = (r.width + img->width - 1) / img->width;
129  uint16 numy = (r.height + img->height - 1) / img->height;
130 
131  static const Recolouring recolour; // Fixed recolouring mapping.
132  _video.BlitImages({base_x, base_y}, img, numx, numy, recolour);
133 
134  _video.SetClippedRectangle(cr); // Restore clipped area.
135 }

171 static uint8 *GetSingleLine(uint8 *text, int max_width, int *width)
172 {
173  uint8 *start = text;
174  uint8 *best_pos = nullptr;
175  for (;;) {
176  uint8 *current = text;
177  /* Proceed to the first white-space. */
178  while (*current != '\n' && *current != '\0' && *current != ' ') current++;
179  uint8 orig = *current;
180  *current = '\0';
181  int line_width, line_height;
182  _video.GetTextSize(start, &line_width, &line_height);
183 
184  if (line_width < max_width) {
185  *width = line_width;
186  if (orig == '\0') return current;
187  if (orig == '\n') {
188  *current = '\n';
189  return current;
190  }
191 
192  *current = ' '; // orig was ' ' too.
193  best_pos = current;
194  text = current + 1;
195  continue;
196  }
197 
198  /* line_width >= max_width */
199  if (best_pos != nullptr) {
200  /* There is a best position to fall back to. */
201  *current = orig; // We will revisit this position later.
202  *best_pos = '\n'; // Insert line break.
203  return best_pos;
204  }
205 
206  /* line_width >= max_width, no fallback to a previous position. */
207  *width = line_width;
208  if (orig == '\n' || orig == ' ') *current = '\n';
209  return current;
210  }
211 }

253 bool DrawMultilineString(StringID strid, int x, int y, int max_width, int max_height, uint8 colour)
254 {
255  uint8 buffer[1024]; // Arbitrary max size.
256  DrawText(strid, buffer, lengthof(buffer));
257 
258  uint8 *text = buffer;
259  while (*text != '\0') {
260  if (max_height < _video.GetTextHeight()) return false;
261  max_height -= _video.GetTextHeight();
262 
263  int line_width;
264  uint8 *end = GetSingleLine(text, max_width, &line_width);
265  if (*end == '\0') {
266  _video.BlitText(text, _palette[colour], x, y, max_width);
267  break;
268  }
269  *end = '\0';
270  _video.BlitText(text, _palette[colour], x, y, max_width);
271  y += _video.GetTextHeight();
272  text = end + 1;
273  }
274  return true;
275 }
