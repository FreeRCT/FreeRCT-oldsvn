50 void SelectMouseMode::OnMouseButtonEvent(Viewport *vp, uint8 state)
51 {
52  this->mouse_state = state & MB_CURRENT;
53  if (this->mouse_state != 0) {
54  FinderData fdata(CS_RIDE | CS_PERSON, FW_TILE);
55  switch (vp->ComputeCursorPosition(&fdata)) {
56  case CS_RIDE: {
57  RideInstance *ri = _rides_manager.GetRideInstance(fdata.ride);
58  if (ri == nullptr) break;
59  switch (ri->GetKind()) {
60  case RTK_SHOP:
61  ShowShopManagementGui(fdata.ride);
62  break;
63 
64  case RTK_COASTER:
65  ShowCoasterManagementGui(ri);
66  break;
67 
68  default: break; // Other types are not implemented yet.
69  }
70  break;
71  }
72 
73  case CS_PERSON:
74  ShowGuestInfoGui(fdata.person);
75  break;
76 
77  default: break;
78  }
79  }
