49 void StringsStorage::AddStrings(std::shared_ptr<StringsNode> strs, const Position &pos)
50 {
51  /* Split the strings into lists with the same key, and push them into a bundle. */
52  while (strs->strings.size() > 0) {
53  if (!NeedsSplit(strs)) {
54  this->AddToBundle(strs, pos);
55  break;
56  }
57  auto new_strs = std::make_shared<StringsNode>();
58  std::string new_key = "";
59  std::list<StringNode>::iterator iter = strs->strings.begin();
60  while (iter != strs->strings.end()) {
61  if (new_key == "") {
62  if (iter->key != "") {
63  new_key = iter->key;
64  new_strs->Add(*iter, pos);
65  std::list<StringNode>::iterator iter2 = iter; iter++;
66  strs->strings.erase(iter2);
67  continue;
68  }
69  iter++;
70  continue;
71  }

73  new_strs->Add(*iter, pos);

79  }
