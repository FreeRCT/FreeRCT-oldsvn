309 void BottomToolbarWindow::UpdateWidgetSize(WidgetNumber wid_num, BaseWidget *wid)
310 {
311  /* -99,999,999.99 = Maximum amount of money that is worth handling for now. */
312  static const int64 LARGE_MONEY_AMOUNT = -9999999999;
313  static const int LARGE_TEMPERATURE = 999; // Large enough to format all temperatures.
314  Point32 p(0, 0);
315 
316  switch (wid_num) {
317  case BTB_STATUS:
318  p = GetMoneyStringSize(LARGE_MONEY_AMOUNT);
319  break;
320 
321  case BTB_VIEW_DIRECTION: {
322  Rectangle16 rect = _sprite_manager.GetTableSpriteSize(SPR_GUI_COMPASS_START); // It's the same size for all compass sprites.
323  p = {rect.width, rect.height};
324  break;
325  }
326 
327  case BTB_WEATHER: {
328  Rectangle16 rect = _sprite_manager.GetTableSpriteSize(SPR_GUI_WEATHER_START);
329  p = {rect.width, rect.height};
330  break;
331  }
332 
333  case BTB_TEMPERATURE:
334  _str_params.SetTemperature(1, LARGE_TEMPERATURE);
335  GetTextSize(STR_ARG1, &p.x, &p.y);
336  break;
337 
338  case BTB_SPACING: {
339  _str_params.SetNumber(1, LARGE_TEMPERATURE);
340  Point32 temp_size;
341  GetTextSize(STR_ARG1, &temp_size.x, &temp_size.y);
342 
343  int32 remaining = _video.GetXSize() - (2 * BOTTOM_BAR_POSITION_X);
344  remaining -= temp_size.x;
345  remaining -= _sprite_manager.GetTableSpriteSize(SPR_GUI_WEATHER_START).width;
346  remaining -= GetMoneyStringSize(LARGE_MONEY_AMOUNT).x;
347  remaining -= GetMaxDateSize().x;
348  remaining -= _sprite_manager.GetTableSpriteSize(SPR_GUI_COMPASS_START).base.x; // It's the same size for all compass sprites.
349  p = {remaining, (int32)BOTTOM_BAR_HEIGHT};
350  break;
351  }
352 
353  case BTB_DATE:
354  p = GetMaxDateSize();
355  break;
356  }
357 
358  wid->min_x = std::max(wid->min_x, (uint16)p.x);
359  wid->min_y = std::max(wid->min_y, (uint16)p.y);
360 }

362 void BottomToolbarWindow::DrawWidget(WidgetNumber wid_num, const BaseWidget *wid) const
363 {
364  static Recolouring recolour; // Never changed.
365 
366  switch (wid_num) {
367  case BTB_VIEW_DIRECTION: {
368  Viewport *vp = GetViewport();
369  int dir = (vp == nullptr) ? 0 : vp->orientation;
370  const ImageData *img = _sprite_manager.GetTableSprite(SPR_GUI_COMPASS_START + dir);
371  if (img != nullptr) _video.BlitImage({GetWidgetScreenX(wid), GetWidgetScreenY(wid)}, img, recolour, GS_NORMAL);
372  break;
373  }
374 
375  case BTB_WEATHER: {
376  int spr = SPR_GUI_WEATHER_START + _weather.GetWeatherType();
377  const ImageData *img = _sprite_manager.GetTableSprite(spr);
378  if (img != nullptr) _video.BlitImage({GetWidgetScreenX(wid), GetWidgetScreenY(wid)}, img, recolour, GS_NORMAL);
379  break;
380  }
381  }
382 }
