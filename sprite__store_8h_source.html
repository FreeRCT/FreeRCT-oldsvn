500  const ImageData *GetFenceSprite(FenceType type, TileEdge edge, TileSlope slope, ViewOrientation orient) const
501  {
502  if (this->fence[type] == nullptr) return nullptr;
503 
504  uint8 corner_height[4];
505  ComputeCornerHeight(slope, (slope & TSB_TOP) != 0 ? 1 : 0, corner_height);
506 
507  /*
508  * Compute edge case 0..2
509  * 0 = flat, 1 = one corner raised, 2 = other corner raised
510  */
511  uint8 edge_case = (corner_height[edge] > corner_height[(edge+1) % 4] ? 1 : 0) +
512  (corner_height[(edge+1) % 4] > corner_height[edge] ? 2 : 0);
513 
514  uint8 sprite = edge_case + edge * FENCE_EDGE_COUNT;
515  sprite = (sprite + FENCE_COUNT - FENCE_EDGE_COUNT * orient) % FENCE_COUNT;
516  return this->fence[type]->sprites[sprite];
517  }

575  const ImageData *GetAnimationSprite(AnimationType anim_type, uint16 frame_index, PersonType pers_type, ViewOrientation view) const
576  {
577  /* anim_type = 1..4, normalize with -1, subtract orientation,
578  * add 4 + mod 4 to get the normalized animation, +1 to get the real animation. */
579  anim_type = (AnimationType)(1 + (4 + (anim_type - 1) - view) % 4);
580 
581  for (auto iter = this->animations.find(anim_type); iter != this->animations.end(); ++iter) {
582  const AnimationSprites *asp = iter->second;
583 
584  if (asp->anim_type != anim_type) return nullptr;
585  if (asp->person_type == pers_type) return asp->sprites[frame_index];
586  }
587  return nullptr;
588  }
