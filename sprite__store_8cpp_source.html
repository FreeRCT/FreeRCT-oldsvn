95 static int DecodeUtf8Char(uint8 *data, size_t length, uint32 *codepoint)
96 {
97  if (length < 1) return 0;
98  uint32 value = *data;
99  data++;
100  if ((value & 0x80) == 0) {
101  *codepoint = value;
102  return 1;
103  }
104  int size;
105  uint32 min_value;
106  if ((value & 0xE0) == 0xC0) {
107  size = 2;
108  min_value = 0x80;
109  value &= 0x1F;
110  } else if ((value & 0xF0) == 0xE0) {
111  size = 3;
112  min_value = 0x800;
113  value &= 0x0F;
114  } else if ((value & 0xF8) == 0xF0) {
115  size = 4;
116  min_value = 0x10000;
117  value &= 0x07;
118  } else {
119  return 0;
120  }
121 
122  if (length < static_cast<size_t>(size)) return 0;
123  for (int n = 1; n < size; n++) {
124  uint8 val = *data;
125  data++;
126  if ((val & 0xC0) != 0x80) return 0;
127  value = (value << 6) | (val & 0x3F);
128  }
129  if (value < min_value || (value >= 0xD800 && value <= 0xDFFF) || value > 0x10FFFF) return 0;
130  *codepoint = value;
131  return size;
132 }

143 static bool ReadUtf8Text(RcdFileReader *rcd_file, size_t expected_length, uint8 *buffer, size_t buf_length, size_t *used_size)
144 {
145  if (buf_length < *used_size + expected_length) return false;
146  rcd_file->GetBlob(buffer + *used_size, expected_length);
147 
148  int real_size = expected_length;
149  uint8 *start = buffer + *used_size;
150 
151  for (;;) {
152  uint32 code_point;
153  int sz = DecodeUtf8Char(start, real_size, &code_point);
154  if (sz == 0 || sz > real_size) return false;
155  real_size -= sz;
156  start += sz;
157  if (code_point == 0) break;
158  }
159  if (real_size != 0) return false;
160 
161  *used_size += expected_length;
162  return true;
163 }

170 bool TextData::Load(RcdFileReader *rcd_file)
171 {
172  uint8 buffer[64*1024]; // Arbitrary sized block of temporary memory to store the text data.
173  size_t used_size = 0;
174  uint32 length = rcd_file->size;
175  if (rcd_file->version != 2) return false;
176 
177  TextString strings[MAX_NUM_TEXT_STRINGS];
178  uint used_strings = 0;
179  while (length > 0) {
180  if (used_strings >= lengthof(strings)) return false; // Too many text strings.
181 
182  if (length < 3) return false;
183  uint16 str_length = rcd_file->GetUInt16();
184  uint8 ident_length = rcd_file->GetUInt8();
185 
186  if (static_cast<uint32>(str_length) > length) return false; // String does not fit in the block.
187  length -= 3;
188 
189  if (ident_length + 2 + 1 >= str_length) return false; // No space for translations.
190  int trs_length = str_length - (ident_length + 2 + 1);
191 
192  /* Read string name. */
193  strings[used_strings].name = (char *)buffer + used_size;
194  if (!ReadUtf8Text(rcd_file, ident_length, buffer, lengthof(buffer), &used_size)) return false;
195  length -= ident_length;
196 
197  while (trs_length > 0) {
198  if (length < 3) return false;
199  uint16 tr_length = rcd_file->GetUInt16();
200  uint8 lang_length = rcd_file->GetUInt8();
201  length -= 3;
202 
203  if (tr_length > trs_length) return false;
204  if (lang_length + 2 + 1 >= tr_length) return false;
205  int text_length = tr_length - (lang_length + 2 + 1);
206 
207  uint8 lang_buffer[1000]; // Arbitrary sized block to store the language name or a single string.
208  size_t used = 0;
209 
210  /* Read translation language string. */
211  if (!ReadUtf8Text(rcd_file, lang_length, lang_buffer, lengthof(lang_buffer), &used)) return false;
212  length -= lang_length;
213 
214  int lang_idx = GetLanguageIndex((char *)lang_buffer);
215  /* Read translation text. */
216  if (lang_idx >= 0) {
217  strings[used_strings].languages[lang_idx] = buffer + used_size;
218  if (!ReadUtf8Text(rcd_file, text_length, buffer, lengthof(buffer), &used_size)) return false;
219  } else {
220  /* Illegal language, read into a dummy buffer. */
221  used = 0;
222  if (!ReadUtf8Text(rcd_file, text_length, lang_buffer, lengthof(lang_buffer), &used)) return false;
223  }
224  length -= text_length;
225 
226  trs_length -= 3 + lang_length + text_length;
227  }
228  assert(trs_length == 0);
229  used_strings++;
230  }
231  assert (length == 0);
232 
233  this->strings = new TextString[used_strings];
234  this->string_count = used_strings;
235  this->text_data = new uint8[used_size];
236  if (this->strings == nullptr || this->text_data == nullptr) return false;
237 
238  memcpy(this->text_data, buffer, used_size);
239  for (uint i = 0; i < used_strings; i++) {
240  this->strings[i].name = (strings[i].name == nullptr) ? nullptr : (char *)this->text_data + ((uint8 *)(strings[i].name) - buffer);
241  for (uint lng = 0; lng < LANGUAGE_COUNT; lng++) {
242  this->strings[i].languages[lng] = (strings[i].languages[lng] == nullptr)
243  ? nullptr : this->text_data + (strings[i].languages[lng] - buffer);
244  }
245  }
246  return true;
247 }
