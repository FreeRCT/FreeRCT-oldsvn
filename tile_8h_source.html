158 static inline TileSlope ExpandTileSlope(uint8 v)
159 {
160  if (v < 15) return (TileSlope)v;
161  if (v < 19) return TSB_STEEP | (TileSlope)(1 << (v - 15));
162  return TSB_STEEP | TSB_TOP | (TileSlope)(1 << (v - 19));
163 }

171 static inline uint8 ImplodeTileSlope(TileSlope s)
172 {
173  if ((s & TSB_STEEP) == 0) return s;
174 
175  uint8 base = ((s & TSB_TOP) == 0) ? 15 : 19;
176  if ((s & TSB_NORTH) != 0) return base;
177  if ((s & TSB_EAST) != 0) return base + 1;
178  if ((s & TSB_SOUTH) != 0) return base + 2;
179  return base + 3;
180 }
