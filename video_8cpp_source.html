375 bool VideoSystem::HandleEvent()
376 {
377  SDL_Event event;
378  if (SDL_PollEvent(&event) != 1) return true;
379 
380  switch (event.type) {
381  case SDL_TEXTINPUT:
382  return HandleKeyInput(WMKC_SYMBOL, (const uint8 *)event.text.text);
383 
384  case SDL_KEYDOWN:
385  switch (event.key.keysym.sym) {
386  case SDLK_UP: return HandleKeyInput(WMKC_CURSOR_UP, nullptr);
387  case SDLK_LEFT: return HandleKeyInput(WMKC_CURSOR_LEFT, nullptr);
388  case SDLK_RIGHT: return HandleKeyInput(WMKC_CURSOR_RIGHT, nullptr);
389  case SDLK_DOWN: return HandleKeyInput(WMKC_CURSOR_DOWN, nullptr);
390  case SDLK_ESCAPE: return HandleKeyInput(WMKC_CANCEL, nullptr);
391  case SDLK_DELETE: return HandleKeyInput(WMKC_DELETE, nullptr);
392  case SDLK_BACKSPACE: return HandleKeyInput(WMKC_BACKSPACE, nullptr);
393 
394  case SDLK_RETURN:
395  case SDLK_RETURN2:
396  case SDLK_KP_ENTER: return HandleKeyInput(WMKC_CONFIRM, nullptr);
397 
398  default: return false;
399  }
400 
401  case SDL_MOUSEMOTION:
402  UpdateMousePosition(event.button.x, event.button.y);
403  return false;
404 
405  case SDL_MOUSEWHEEL:
406  _window_manager.MouseWheelEvent((event.wheel.y > 0) ? 1 : -1);
407  return false;
408 
409  case SDL_MOUSEBUTTONUP:
410  case SDL_MOUSEBUTTONDOWN:
411  switch (event.button.button) {
412  case SDL_BUTTON_LEFT:
413  UpdateMousePosition(event.button.x, event.button.y);
414  _window_manager.MouseButtonEvent(MB_LEFT, (event.type == SDL_MOUSEBUTTONDOWN));
415  break;
416 
417  case SDL_BUTTON_MIDDLE:
418  UpdateMousePosition(event.button.x, event.button.y);
419  _window_manager.MouseButtonEvent(MB_MIDDLE, (event.type == SDL_MOUSEBUTTONDOWN));
420  break;
421 
422  case SDL_BUTTON_RIGHT:
423  UpdateMousePosition(event.button.x, event.button.y);
424  _window_manager.MouseButtonEvent(MB_RIGHT, (event.type == SDL_MOUSEBUTTONDOWN));
425  break;
426 
427  default:
428  break;
429  }
430  return false;
431 
432  case SDL_USEREVENT:
433  return true;
434 
435  case SDL_WINDOWEVENT:
436  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
437  this->SetResolution({event.window.data1, event.window.data2});
438  }
439  if (event.window.event == SDL_WINDOWEVENT_EXPOSED) {
440  _video.MarkDisplayDirty();
441  UpdateWindows();
442  }
443  return false;
444 
445  case SDL_QUIT:
446  QuitProgram();
447  return true;
448 
449  default:
450  return false; // Ignore other events.
451  }

455 void VideoSystem::MainLoop()
456 {
457  static const uint32 FRAME_DELAY = 30; // Number of milliseconds between two frames.
458  bool missing_sprites_check = false;
459  _finish = false;
460 
461  while (!_finish) {
462  uint32 start = SDL_GetTicks();
463 
464  OnNewFrame(FRAME_DELAY);
465 
466  /* Handle input events until time for the next frame has arrived. */
467  for (;;) {
468  if (HandleEvent()) break;
469  }
470  if (_finish) break;
471 
472  uint32 now = SDL_GetTicks();
473  if (now >= start) { // No wrap around.
474  now -= start;
475  if (now < FRAME_DELAY) SDL_Delay(FRAME_DELAY - now); // Too early, wait until next frame.
476  }
477 
478  if (!missing_sprites_check && this->missing_sprites) {
479  ShowGraphicsErrorMessage();
480  missing_sprites_check = true;
481  }
482  }
483 }

525 static void BlitPixel(const ClippedRectangle &cr, uint32 *scr_base,
526  int32 xmin, int32 ymin, uint16 numx, uint16 numy, uint16 width, uint16 height, uint32 colour)
527 {
528  const int32 xend = xmin + numx * width;
529  const int32 yend = ymin + numy * height;
530  while (ymin < yend) {
531  if (ymin >= cr.height) return;
532 
533  if (ymin >= 0) {
534  uint32 *scr = scr_base;
535  int32 x = xmin;
536  while (x < xend) {
537  if (x >= cr.width) break;
538  if (x >= 0) *scr = colour;
539 
540  x += width;
541  scr += width;
542  }
543  }
544  ymin += height;
545  scr_base += height * cr.pitch;
546  }

559 static void Blit8bppImages(const ClippedRectangle &cr, int32 x_base, int32 y_base, const ImageData *spr, uint16 numx, uint16 numy, const uint8 *recoloured)
560 {
561  uint32 *line_base = cr.address + x_base + cr.pitch * y_base;
562  int32 ypos = y_base;
563  for (int yoff = 0; yoff < spr->height; yoff++) {
564  uint32 offset = spr->table[yoff];
565  if (offset != INVALID_JUMP) {
566  int32 xpos = x_base;
567  uint32 *src_base = line_base;
568  for (;;) {
569  uint8 rel_off = spr->data[offset];
570  uint8 count = spr->data[offset + 1];
571  uint8 *pixels = &spr->data[offset + 2];
572  offset += 2 + count;
573 
574  xpos += rel_off & 127;
575  src_base += rel_off & 127;
576  while (count > 0) {
577  uint32 colour = _palette[recoloured[*pixels]];
578  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
579  pixels++;
580  xpos++;
581  src_base++;
582  count--;
583  }
584  if ((rel_off & 128) != 0) break;
585  }
586  }
587  line_base += cr.pitch;
588  ypos++;
589  }
590 }

603 static void Blit32bppImages(const ClippedRectangle &cr, int32 x_base, int32 y_base, const ImageData *spr, uint16 numx, uint16 numy, const Recolouring &recolour, GradientShift shift)
604 {
605  uint32 *line_base = cr.address + x_base + cr.pitch * y_base;
606  ShiftFunc sf = GetGradientShiftFunc(shift);
607  int32 ypos = y_base;
608  const uint8 *src = spr->data + 2; // Skip the length word.
609  for (int yoff = 0; yoff < spr->height; yoff++) {
610  int32 xpos = x_base;
611  uint32 *src_base = line_base;
612  for (;;) {
613  uint8 mode = *src++;
614  if (mode == 0) break;
615  switch (mode >> 6) {
616  case 0: // Fully opaque pixels.
617  mode &= 0x3F;
618  for (; mode > 0; mode--) {
619  uint32 colour = MakeRGBA(sf(src[0]), sf(src[1]), sf(src[2]), OPAQUE);
620  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
621  xpos++;
622  src_base++;
623  src += 3;
624  }
625  break;
626 
627  case 1: { // Partial opaque pixels.
628  uint8 opacity = *src++;
629  mode &= 0x3F;
630  for (; mode > 0; mode--) {
631  /* Cheat transparency a bit by just recolouring the previously drawn pixel */
632  uint32 old_pixel = *src_base;
633 
634  uint r = sf(src[0]) * opacity + GetR(old_pixel) * (256 - opacity);
635  uint g = sf(src[1]) * opacity + GetG(old_pixel) * (256 - opacity);
636  uint b = sf(src[2]) * opacity + GetB(old_pixel) * (256 - opacity);
637 
638  /* Opaque, but colour adjusted depending on the old pixel. */
639  uint32 ndest = MakeRGBA(r >> 8, g >> 8, b >> 8, OPAQUE);
640  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, ndest);
641  xpos++;
642  src_base++;
643  src += 3;
644  }
645  break;
646  }
647  case 2: // Fully transparent pixels.
648  xpos += mode & 0x3F;
649  src_base += mode & 0x3F;
650  break;
651 
652  case 3: { // Recoloured pixels.
653  uint8 layer = *src++;
654  const uint32 *table = recolour.GetRecolourTable(layer - 1);
655  uint8 opacity = *src++;
656  mode &= 0x3F;
657  for (; mode > 0; mode--) {
658  uint32 old_pixel = *src_base;
659  uint32 recoloured = table[*src++];
660 
661  uint r = sf(GetR(recoloured)) * opacity + GetR(old_pixel) * (256 - opacity);
662  uint g = sf(GetG(recoloured)) * opacity + GetG(old_pixel) * (256 - opacity);
663  uint b = sf(GetB(recoloured)) * opacity + GetB(old_pixel) * (256 - opacity);
664 
665  uint32 colour = MakeRGBA(r >> 8, g >> 8, b >> 8, OPAQUE);
666  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
667  xpos++;
668  src_base++;
669  }
670  break;
671  }
672  }
673  }
674  line_base += cr.pitch;
675  ypos++;
676  src += 2; // Skip the length word.
677  }
678 }

689 void VideoSystem::BlitImages(const Point32 &pt, const ImageData *spr, uint16 numx, uint16 numy, const Recolouring &recolour, GradientShift shift)
690 {
691  this->blit_rect.ValidateAddress();
692 
693  int x_base = pt.x + spr->xoffset;
694  int y_base = pt.y + spr->yoffset;
695 
696  /* Don't draw wildly outside the screen. */
697  while (numx > 0 && x_base + spr->width < 0) {
698  x_base += spr->width; numx--;
699  }
700  while (numx > 0 && x_base + (numx - 1) * spr->width >= this->blit_rect.width) numx--;
701  if (numx == 0) return;
702 
703  while (numy > 0 && y_base + spr->height < 0) {
704  y_base += spr->height; numy--;
705  }
706  while (numy > 0 && y_base + (numy - 1) * spr->height >= this->blit_rect.height) numy--;
707  if (numy == 0) return;
708 
709  if (GB(spr->flags, IFG_IS_8BPP, 1) != 0) {
710  Blit8bppImages(this->blit_rect, x_base, y_base, spr, numx, numy, recolour.GetPalette(shift));
711  } else {
712  Blit32bppImages(this->blit_rect, x_base, y_base, spr, numx, numy, recolour, shift);
713  }
714 }

737 void VideoSystem::GetNumberRangeSize(int64 smallest, int64 biggest, int *width, int *height)
738 {
739  assert(smallest <= biggest);
740  if (this->digit_size.x == 0) { // First call, initialize the variable.
741  this->digit_size.x = 0;
742  this->digit_size.y = 0;
743 
744  uint8 buffer[2];
745  buffer[1] = '\0';
746  for (uint8 i = '0'; i <= '9'; i++) {
747  buffer[0] = i;
748  int w, h;
749  this->GetTextSize(buffer, &w, &h);
750  if (w > this->digit_size.x) this->digit_size.x = w;
751  if (h > this->digit_size.y) this->digit_size.y = h;
752  }
753  }
754

756  if (smallest < 0) digit_count++; // Add a 'digit' for the '-' sign.
757 
758  int64 tmp = std::max(std::abs(smallest), std::abs(biggest));
759  while (tmp > 9) {
760  digit_count++;
761  tmp /= 10;
762  }
763 
764  *width = digit_count * this->digit_size.x;
765  *height = this->digit_size.y;

777 void VideoSystem::BlitText(const uint8 *text, uint32 colour, int xpos, int ypos, int width, Alignment align)
778 {
779  SDL_Color col = {0, 0, 0}; // Font colour does not matter as only the bitmap is used.
780  SDL_Surface *surf = TTF_RenderUTF8_Solid(this->font, (const char *)text, col);
781  if (surf == nullptr) {
782  fprintf(stderr, "Rendering text failed (%s)\n", TTF_GetError());
783  return;
784  }
785 
786  if (surf->format->BitsPerPixel != 8 || surf->format->BytesPerPixel != 1) {
787  fprintf(stderr, "Rendering text failed (Wrong surface format)\n");
788  return;
789  }
790 
791  int real_w = std::min(surf->w, width);
792  switch (align) {
793  case ALG_LEFT:
794  break;
795 
796  case ALG_CENTER:
797  xpos += (width - real_w) / 2;
798  break;
799 
800  case ALG_RIGHT:
801  xpos += width - real_w;
802  break;
803 
804  default: NOT_REACHED();
805  }
806 
807  this->blit_rect.ValidateAddress();
808 
809  uint8 *src = ((uint8 *)surf->pixels);
810  uint32 *dest = this->blit_rect.address + xpos + ypos * this->blit_rect.pitch;

860  } else {

866  inc_y = -1;
867  } else {

876  int16 sum_y = 0;
877

880 
881  for (;;) {
882  /* Blit pixel. */
883  if (pos_x >= 0 && pos_x < this->blit_rect.width && pos_y >= 0 && pos_y < this->blit_rect.height) {
884  *dest = colour;
885  }
886  if (pos_x == end.x && pos_y == end.y) break;
887 
888  sum_x += step;
889  sum_y += step;
890  if (sum_x >= dy) {

893  sum_x -= dy;
894  }
895  if (sum_y >= dx) {
