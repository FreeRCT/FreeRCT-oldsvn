346 bool VideoSystem::HandleEvent()
347 {
348  SDL_Event event;
349  if (SDL_PollEvent(&event) != 1) return true;
350  switch (event.type) {
351  case SDL_KEYDOWN:
352  switch (event.key.keysym.sym) {
353  case SDLK_q:
354  QuitProgram();
355  return true;
356 
357  case SDLK_LEFT:
358  GetViewport()->Rotate(-1);
359  break;
360 
361  case SDLK_RIGHT:
362  GetViewport()->Rotate(1);
363  break;
364 
365  default:
366  break;
367  }
368  return false;
369 
370  case SDL_MOUSEMOTION:
371  UpdateMousePosition(event.button.x, event.button.y);
372  return false;
373 
374  case SDL_MOUSEWHEEL:
375  _manager.MouseWheelEvent((event.wheel.y > 0) ? 1 : -1);
376  return false;
377 
378  case SDL_MOUSEBUTTONUP:
379  case SDL_MOUSEBUTTONDOWN:
380  switch (event.button.button) {
381  case SDL_BUTTON_LEFT:
382  UpdateMousePosition(event.button.x, event.button.y);
383  _manager.MouseButtonEvent(MB_LEFT, (event.type == SDL_MOUSEBUTTONDOWN));
384  break;
385 
386  case SDL_BUTTON_MIDDLE:
387  UpdateMousePosition(event.button.x, event.button.y);
388  _manager.MouseButtonEvent(MB_MIDDLE, (event.type == SDL_MOUSEBUTTONDOWN));
389  break;
390 
391  case SDL_BUTTON_RIGHT:
392  UpdateMousePosition(event.button.x, event.button.y);
393  _manager.MouseButtonEvent(MB_RIGHT, (event.type == SDL_MOUSEBUTTONDOWN));
394  break;
395 
396  default:
397  break;
398  }
399  return false;
400 
401  case SDL_USEREVENT:
402  return true;
403 
404  case SDL_WINDOWEVENT:
405  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
406  this->SetResolution({event.window.data1, event.window.data2});
407  }
408  if (event.window.event == SDL_WINDOWEVENT_EXPOSED) {
409  _video.MarkDisplayDirty();
410  UpdateWindows();
411  }
412  return false;
413 
414  case SDL_QUIT:
415  QuitProgram();
416  return true;
417 
418  default:
419  return false; // Ignore other events.
420  }

424 void VideoSystem::MainLoop()
425 {
426  static const uint32 FRAME_DELAY = 30; // Number of milliseconds between two frames.
427  bool missing_sprites_check = false;
428  _finish = false;
429 
430  while (!_finish) {
431  uint32 start = SDL_GetTicks();
432 
433  OnNewFrame(FRAME_DELAY);
434 
435  /* Handle input events until time for the next frame has arrived. */
436  for (;;) {
437  if (HandleEvent()) break;
438  }
439  if (_finish) break;
440 
441  uint32 now = SDL_GetTicks();
442  if (now >= start) { // No wrap around.
443  now -= start;
444  if (now < FRAME_DELAY) SDL_Delay(FRAME_DELAY - now); // Too early, wait until next frame.
445  }
446 
447  if (!missing_sprites_check && this->missing_sprites) {
448  ShowGraphicsErrorMessage();
449  missing_sprites_check = true;
450  }
451  }
452 }

660 static void BlitPixel(const ClippedRectangle &cr, uint32 *scr_base,
661  int32 xmin, int32 ymin, uint16 numx, uint16 numy, uint16 width, uint16 height, uint32 colour)
662 {
663  const int32 xend = xmin + numx * width;
664  const int32 yend = ymin + numy * height;
665  while (ymin < yend) {
666  if (ymin >= cr.height) return;
667 
668  if (ymin >= 0) {
669  uint32 *scr = scr_base;
670  int32 x = xmin;
671  while (x < xend) {
672  if (x >= cr.width) break;
673  if (x >= 0) *scr = colour;
674 
675  x += width;
676  scr += width;
677  }
678  }
679  ymin += height;
680  scr_base += height * cr.pitch;
681  }

694 static void Blit8bppImages(const ClippedRectangle &cr, int32 x_base, int32 y_base, const ImageData *spr, uint16 numx, uint16 numy, const uint8 *recoloured)
695 {
696  uint32 *line_base = cr.address + x_base + cr.pitch * y_base;
697  int32 ypos = y_base;
698  for (int yoff = 0; yoff < spr->height; yoff++) {
699  uint32 offset = spr->table[yoff];
700  if (offset != INVALID_JUMP) {
701  int32 xpos = x_base;
702  uint32 *src_base = line_base;
703  for (;;) {
704  uint8 rel_off = spr->data[offset];
705  uint8 count = spr->data[offset + 1];
706  uint8 *pixels = &spr->data[offset + 2];
707  offset += 2 + count;
708 
709  xpos += rel_off & 127;
710  src_base += rel_off & 127;
711  while (count > 0) {
712  uint32 colour = _palette[recoloured[*pixels]];
713  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
714  pixels++;
715  xpos++;
716  src_base++;
717  count--;
718  }
719  if ((rel_off & 128) != 0) break;
720  }
721  }
722  line_base += cr.pitch;
723  ypos++;
724  }
725 }

738 static void Blit32bppImages(const ClippedRectangle &cr, int32 x_base, int32 y_base, const ImageData *spr, uint16 numx, uint16 numy, const Recolouring &recolour, GradientShift shift)
739 {
740  uint32 *line_base = cr.address + x_base + cr.pitch * y_base;
741  ShiftFunc sf = GetGradientShiftFunc(shift);
742  int32 ypos = y_base;
743  const uint8 *src = spr->data + 2; // Skip the length word.
744  for (int yoff = 0; yoff < spr->height; yoff++) {
745  int32 xpos = x_base;
746  uint32 *src_base = line_base;
747  for (;;) {
748  uint8 mode = *src++;
749  if (mode == 0) break;
750  switch (mode >> 6) {
751  case 0: // Fully opaque pixels.
752  mode &= 0x3F;
753  for (; mode > 0; mode--) {
754  uint32 colour = MakeRGBA(sf(src[0]), sf(src[1]), sf(src[2]), OPAQUE);
755  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
756  xpos++;
757  src_base++;
758  src += 3;
759  }
760  break;
761 
762  case 1: { // Partial opaque pixels.
763  uint8 opacity = *src++;
764  mode &= 0x3F;
765  for (; mode > 0; mode--) {
766  /* Cheat transparency a bit by just recolouring the previously drawn pixel */
767  uint32 old_pixel = *src_base;
768 
769  uint r = sf(src[0]) * opacity + GetR(old_pixel) * (256 - opacity);
770  uint g = sf(src[1]) * opacity + GetG(old_pixel) * (256 - opacity);
771  uint b = sf(src[2]) * opacity + GetB(old_pixel) * (256 - opacity);
772 
773  /* Opaque, but colour adjusted depending on the old pixel. */
774  uint32 ndest = MakeRGBA(r >> 8, g >> 8, b >> 8, OPAQUE);
775  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, ndest);
776  xpos++;
777  src_base++;
778  src += 3;
779  }
780  break;
781  }
782  case 2: // Fully transparent pixels.
783  xpos += mode & 0x3F;
784  src_base += mode & 0x3F;
785  break;
786 
787  case 3: { // Recoloured pixels.
788  uint8 layer = *src++;
789  const uint32 *table = recolour.GetRecolourTable(layer - 1);
790  uint8 opacity = *src++;
791  mode &= 0x3F;
792  for (; mode > 0; mode--) {
793  uint32 old_pixel = *src_base;
794  uint32 recoloured = table[*src++];
795 
796  uint r = sf(GetR(recoloured)) * opacity + GetR(old_pixel) * (256 - opacity);
797  uint g = sf(GetG(recoloured)) * opacity + GetG(old_pixel) * (256 - opacity);
798  uint b = sf(GetB(recoloured)) * opacity + GetB(old_pixel) * (256 - opacity);
799 
800  uint32 colour = MakeRGBA(r >> 8, g >> 8, b >> 8, OPAQUE);
801  BlitPixel(cr, src_base, xpos, ypos, numx, numy, spr->width, spr->height, colour);
802  xpos++;
803  src_base++;
804  }
805  break;
806  }
807  }
808  }
809  line_base += cr.pitch;
810  ypos++;
811  src += 2; // Skip the length word.
812  }
813 }

825 void VideoSystem::BlitImages(int32 x_base, int32 y_base, const ImageData *spr, uint16 numx, uint16 numy, const Recolouring &recolour, GradientShift shift)
826 {

828 
829  x_base += spr->xoffset;
830  y_base += spr->yoffset;
831 
832  /* Don't draw wildly outside the screen. */

913 void VideoSystem::BlitText(const uint8 *text, uint32 colour, int xpos, int ypos, int width, Alignment align)
914 {

916  SDL_Surface *surf = TTF_RenderUTF8_Solid(this->font, (const char *)text, col);
917  if (surf == nullptr) {
918  fprintf(stderr, "Rendering text failed (%s)\n", TTF_GetError());
919  return;
920  }
921 
922  if (surf->format->BitsPerPixel != 8 || surf->format->BytesPerPixel != 1) {
923  fprintf(stderr, "Rendering text failed (Wrong surface format)\n");
924  return;
925  }
926 
927  int real_w = std::min(surf->w, width);
928  switch (align) {
929  case ALG_LEFT:
930  break;
931 
932  case ALG_CENTER:
933  xpos += (width - real_w) / 2;
934  break;
935 
936  case ALG_RIGHT:
937  xpos += width - real_w;
938  break;
939 
940  default: NOT_REACHED();
941  }
942 
943  this->blit_rect.ValidateAddress();
944 
945  uint8 *src = ((uint8 *)surf->pixels);
946  uint32 *dest = this->blit_rect.address + xpos + ypos * this->blit_rect.pitch;
947  int h = surf->h;
948  if (ypos < 0) {
949  h += ypos;
950  src -= ypos * surf->pitch;
951  dest -= ypos * this->blit_rect.pitch;
952  ypos = 0;
953  }
954  while (h > 0) {
955  if (ypos >= this->blit_rect.height) break;
956  uint8 *src2 = src;
957  uint32 *dest2 = dest;
958  int w = real_w;
959  int x = xpos;
960  if (x < 0) {
961  w += x;
962  if (w <= 0) break;
963  dest2 -= x;
964  src2 -= x;
965  x = 0;
966  }
967  while (w > 0) {
968  if (x >= this->blit_rect.width) break;
969  if (*src2 != 0) *dest2 = colour;
970  src2++;
971  dest2++;
972  x++;
973  w--;
974  }
975  ypos++;
976  src += surf->pitch;
977  dest += this->blit_rect.pitch;
978  h--;
979  }

991 {

994  dx = start.x - end.x;
995  inc_x = -1;
996  } else {

998  inc_x = 1;
999  }
1000  if (start.y > end.y) {

1002  inc_y = -1;
1003  } else {
1004  dy = end.y - start.y;
1005  inc_y = 1;
1006  }
1007 
1008  int16 step = std::min(dx, dy);
1009  int16 pos_x = start.x;
1010  int16 pos_y = start.y;
1011  int16 sum_x = 0;
1012  int16 sum_y = 0;
1013 
1014  this->blit_rect.ValidateAddress();
1015  uint32 *dest = this->blit_rect.address + pos_x + pos_y * this->blit_rect.pitch;
1016 
1017  for (;;) {
1018  /* Blit pixel. */
1019  if (pos_x >= 0 && pos_x < this->blit_rect.width && pos_y >= 0 && pos_y < this->blit_rect.height) {
1020  *dest = colour;
1021  }
1022  if (pos_x == end.x && pos_y == end.y) break;
1023 
1024  sum_x += step;
1025  sum_y += step;
1026  if (sum_x >= dy) {
1027  pos_x += inc_x;
1028  dest += inc_x;
1029  sum_x -= dy;
1030  }
1031  if (sum_y >= dx) {
1032  pos_y += inc_y;
1033  dest += inc_y * this->blit_rect.pitch;
1034  sum_y -= dx;
1035  }
1036  }
