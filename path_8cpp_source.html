221 uint8 GetPathExits(const Voxel *v)
222 {
223  SmallRideInstance instance = v->GetInstance();
224  if (instance != SRI_PATH) return 0;
225  uint16 inst_data = v->GetInstanceData();
226  if (!HasValidPath(inst_data)) return 0;
227 
228  PathSprites slope = GetImplodedPathSlope(inst_data);
229  if (slope < PATH_FLAT_COUNT) { // At a flat path tile.
230  return (_path_expand[slope] >> PATHBIT_NE) & 0xF;
231  }
232 
233  switch (slope) {
234  case PATH_RAMP_NE: return (1 << EDGE_NE) | (0x10 << EDGE_SW);
235  case PATH_RAMP_NW: return (1 << EDGE_NW) | (0x10 << EDGE_SE);
236  case PATH_RAMP_SE: return (1 << EDGE_SE) | (0x10 << EDGE_NW);
237  case PATH_RAMP_SW: return (1 << EDGE_SW) | (0x10 << EDGE_NE);
238  default: NOT_REACHED();
239  }
240 }

250 bool TravelQueuePath(XYZPoint16 *voxel_pos, TileEdge *entry)
251 {
252  XYZPoint16 new_pos = *voxel_pos;
253  TileEdge edge = *entry;
254 
255  /* Check that entry voxel actually exists. */
256  if (!IsVoxelstackInsideWorld(new_pos.x, new_pos.y)) return false;
257 
258  for (;;) {
259  new_pos.x += _tile_dxy[edge].x;
260  new_pos.y += _tile_dxy[edge].y;
261  if (!IsVoxelstackInsideWorld(new_pos.x, new_pos.y)) return false;
262 
263  const Voxel *vx = _world.GetVoxel(new_pos);
264  if (vx == nullptr || !HasValidPath(vx)) {
265  /* No path here, check the voxel below. */
266  if (new_pos.z == 0) return true; // Path ends here.
267  new_pos.z--;
268  vx = _world.GetVoxel(new_pos);
269  if (vx == nullptr || !HasValidPath(vx)) return true; // Path ends here.
270  }
271 
272  if (new_pos == *voxel_pos) return false; // Cycle detected.
273 
274  /* Stop if we found a non-queue path. */
275  if (_sprite_manager.GetPathStatus(GetPathType(vx->GetInstanceData())) != PAS_QUEUE_PATH) return true;
276 
277  /* At this point:
278  * voxel_pos, edge (and *entry) contain the last valid voxel edge.
279  * new_pos, vx is the next queue path tile position.
280  */
281 
282  uint8 exits = GetPathExits(vx);
283 
284  /* Check that the new tile can go back to our last tile. */
285  uint8 rev_edge = (edge + 2) % 4;
286  if (!((exits & (0x01 << rev_edge)) != 0 && new_pos.z == voxel_pos->z) &&
287  !((exits & (0x10 << rev_edge)) != 0 && new_pos.z == voxel_pos->z - 1)) {
288  return false;
289  }
290 
291  /* Find exit to the next path tile. */
292  for (edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
293  if (edge == rev_edge) continue; // Skip the direction we came from.
294  if ((exits & (0x01 << edge)) != 0) break;
295  if ((exits & (0x10 << edge)) != 0) {
296  new_pos.z++;
297  break;
298  }
299  }
300 
301  if (edge == EDGE_COUNT) return false; // Queue path doesn't have a second exit.
302 
303  *voxel_pos = new_pos;
304  *entry = edge;
305  }
306 }

315 uint8 SetPathEdge(uint8 slope, TileEdge edge, bool connect)
316 {
317  const uint8 north_edges = (1 << PATHBIT_NE) | (1 << PATHBIT_NW);
318  const uint8 south_edges = (1 << PATHBIT_SE) | (1 << PATHBIT_SW);
319  const uint8 east_edges = (1 << PATHBIT_NE) | (1 << PATHBIT_SE);
320  const uint8 west_edges = (1 << PATHBIT_NW) | (1 << PATHBIT_SW);
321 
322  if (slope >= PATH_FLAT_COUNT) return slope; // Ramps do not have edge to connect.
323 
324  slope = _path_expand[slope];
325  uint8 bit_value;
326  switch (edge) {
327  case EDGE_NE: bit_value = 1 << PATHBIT_NE; break;
328  case EDGE_SE: bit_value = 1 << PATHBIT_SE; break;
329  case EDGE_SW: bit_value = 1 << PATHBIT_SW; break;
330  case EDGE_NW: bit_value = 1 << PATHBIT_NW; break;
331  default: NOT_REACHED();
332  }
333  if (connect) {
334  slope |= bit_value;
335  } else {
336  slope &= ~bit_value;
337  }
338 
339  slope &= ~((1 << PATHBIT_N) | (1 << PATHBIT_E) | (1 << PATHBIT_S) | (1 << PATHBIT_W));
340  if ((slope & north_edges) == north_edges) slope |= 1 << PATHBIT_N;
341  if ((slope & south_edges) == south_edges) slope |= 1 << PATHBIT_S;
342  if ((slope & east_edges) == east_edges) slope |= 1 << PATHBIT_E;
343  if ((slope & west_edges) == west_edges) slope |= 1 << PATHBIT_W;
344  return _path_implode[slope];
345 }

352 static int GetQuePathEdgeConnectCount(uint8 impl_slope)
353 {
354  uint8 exp_edges = _path_expand[impl_slope] & PATHMASK_EDGES;
355  if (exp_edges == 0) return 0;
356  if (exp_edges == PATHMASK_NE || exp_edges == PATHMASK_SE || exp_edges == PATHMASK_SW || exp_edges == PATHMASK_NW) return 1;
357  return 2;

373 static bool ExamineNeighbourPathEdge(const XYZPoint16 &voxel_pos, bool use_additions, TileEdge edge, bool add_edges, bool at_bottom,
374  Voxel **dest_voxel, uint16 *dest_inst_data, PathStatus *dest_status)
375 {
376  Voxel *v;
377 
378  *dest_voxel = nullptr;
379  *dest_status = PAS_UNUSED;
380  *dest_inst_data = PATH_INVALID;
381 
382  if (use_additions) {
383  v = _additions.GetCreateVoxel(voxel_pos, false);
384  } else {
385  v = _world.GetCreateVoxel(voxel_pos, false);
386  }
387  if (v == nullptr) return false;
388 
389  uint16 number = v->GetInstance();
390  if (number == SRI_PATH) {
391  uint16 instance_data = v->GetInstanceData();
392  if (!HasValidPath(instance_data)) return false;
393 
394  uint8 slope = GetImplodedPathSlope(instance_data);
395  if (at_bottom) {
396  if (slope >= PATH_FLAT_COUNT && slope != _path_up_from_edge[edge]) return false;
397  } else {
398  if (slope != _path_down_from_edge[edge]) return false;
399  }
400 
401  PathStatus status = _sprite_manager.GetPathStatus(GetPathType(instance_data));
402  if (add_edges && status == PAS_QUEUE_PATH) { // Only try to connect to queue paths if they are not yet connected to 2 (or more) neighbours.
403  if (GetQuePathEdgeConnectCount(slope) > 1) return false;
404  }
405 
406  slope = SetPathEdge(slope, edge, add_edges);
407  *dest_status = status;
408  *dest_voxel = v;
409  *dest_inst_data = SetImplodedPathSlope(instance_data, slope);
410  return true;
411 
412  } else if (number >= SRI_FULL_RIDES) { // A ride instance. Does it have an entrance here?
413  if ((v->GetInstanceData() & (1 << edge)) != 0) {
414  *dest_status = PAS_QUEUE_PATH;
415  return true;
416  }
417  }
418  return false;
419 }

431 uint8 AddRemovePathEdges(const XYZPoint16 &voxel_pos, uint8 slope, uint8 dirs, bool use_additions, PathStatus status)
432 {
433  PathStatus ngb_status[4]; // Neighbour path status, #PAS_UNUSED means do not connect.
434  Voxel *ngb_voxel[4]; // Neighbour voxels with path, may be null if it doesn't need changing.
435  uint16 ngb_instance_data[4]; // New instance data, if the voxel exists.
436  XYZPoint16 ngb_pos[4]; // Coordinate of the neighbouring voxel.
437 
438  std::fill_n(ngb_status, lengthof(ngb_status), PAS_UNUSED); // Clear path all statuses to prevent connecting to it if an edge is skipped.
439  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
440  if ((dirs & (1 << edge)) == 0) continue; // Skip directions that should not be updated.
441  int delta_z = 0;
442  if (slope >= PATH_FLAT_COUNT) {
443  if (_path_down_from_edge[edge] == slope) {
444  delta_z = 1;
445  } else if (_path_up_from_edge[edge] != slope) {
446  continue;
447  }
448  }
449  Point16 dxy = _tile_dxy[edge];
450  ngb_pos[edge].x = voxel_pos.x + dxy.x;
451  ngb_pos[edge].y = voxel_pos.y + dxy.y;
452  if (!IsVoxelstackInsideWorld(ngb_pos[edge].x, ngb_pos[edge].y)) continue;
453 
454  TileEdge edge2 = (TileEdge)((edge + 2) % 4);
455  bool modified = false;
456  if (delta_z <= 0 || voxel_pos.z < WORLD_Z_SIZE - 1) {
457  ngb_pos[edge].z = voxel_pos.z + delta_z;
458  modified = ExamineNeighbourPathEdge(ngb_pos[edge], use_additions, edge2, status != PAS_UNUSED, true,
459  &ngb_voxel[edge], &ngb_instance_data[edge], &ngb_status[edge]);
460  }
