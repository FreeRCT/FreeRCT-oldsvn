199 void FinancesManager::Load(Loader &ldr)
200 {
201  this->Reset(); // version == 0 already handled.
202 
203  uint32 version = ldr.OpenBlock("FINA");
204  if (version == 1) {
205  this->num_used = ldr.GetByte();
206  this->current = ldr.GetByte();
207  this->cash = ldr.GetLongLong();
208  for (int i = 0; i < this->num_used; i++) this->finances[i].Load(ldr, version);
209  } else {
210  ldr.SetFailMessage("Unknown block in finances manager.");
211  }
212  ldr.CloseBlock();
213  this->NotifyGui();
214 }
