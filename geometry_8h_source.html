312  void RestrictTo(typename PT::CoordType startx, typename PT::CoordType starty, SZ w, SZ h)
313  {
314  if (this->width == 0 || this->height == 0) return;
315 
316  typename PT::CoordType xend = this->base.x + this->width;
317  if (xend > static_cast<typename PT::CoordType>(startx + w)) xend = startx + w;
318  if (this->base.x < startx) this->base.x = startx;
319  this->width = (this->base.x < xend) ? xend - this->base.x : 0;
320 
321  typename PT::CoordType yend = this->base.y + this->height;
322  if (yend > static_cast<typename PT::CoordType>(starty + h)) yend = starty + h;
323  if (this->base.y < starty) this->base.y = starty;
324  this->height = (this->base.y < yend) ? yend - this->base.y : 0;
