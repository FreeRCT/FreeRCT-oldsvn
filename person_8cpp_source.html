58 bool LoadPRSG(RcdFileReader *rcd_file)
59 {
60  uint32 length = rcd_file->size;
61  if (rcd_file->version < 1 || rcd_file->version > 2 || length < 1) return false;
62  uint8 count = rcd_file->GetUInt8();
63  length--;
64 
65  if (length != 13 * count) return false;
66  while (count > 0) {
67  uint8 ps = rcd_file->GetUInt8();
68  uint32 rc1 = rcd_file->GetUInt32();
69  uint32 rc2 = rcd_file->GetUInt32();
70  uint32 rc3 = rcd_file->GetUInt32();
71 
72  PersonType pt;
73  switch (ps) {
74  case 8:
75  case 16: pt = PERSON_GUEST; break;
76  case 17: pt = PERSON_HANDYMAN; break;
77  case 18: pt = PERSON_MECHANIC; break;
78  case 19: pt = PERSON_GUARD; break;
79  case 20: pt = PERSON_ENTERTAINER; break;
80  default: pt = PERSON_INVALID; break;
81  }
82 
83  if (pt != PERSON_INVALID) {
84  PersonTypeData &ptd = ModifyPersonTypeData(pt);
85  ptd.graphics.recolours.Reset();
86  ptd.graphics.recolours.Set(0, RecolourEntry(rc1));

91  }

389 RideVisitDesire Guest::ComputeExitDesire(TileEdge current_edge, XYZPoint16 cur_pos, TileEdge exit_edge, bool *seen_wanted_ride)
390 {
391  if (current_edge == exit_edge) return RVD_NO_VISIT; // Skip incoming edge (may get added later if no other options exist).
392 
393  bool travel = TravelQueuePath(&cur_pos, &exit_edge);
394  if (!travel) return RVD_NO_VISIT; // Path leads to nowhere.
395 
396  if (PathExistsAtBottomEdge(cur_pos, exit_edge)) return RVD_NO_RIDE; // Found a path.
397 
398  RideInstance *ri = RideExistsAtBottom(cur_pos, exit_edge);
399  if (ri == nullptr || ri->state != RIS_OPEN) return RVD_NO_VISIT; // No ride, or a closed one.
400 
401  if (ri == this->ride) { // Guest decided before that this shop/ride should be visited.
402  *seen_wanted_ride = true;
403  return RVD_MUST_VISIT;
404  }
405 
406  Point16 dxy = _tile_dxy[exit_edge];
407  if (!ri->CanBeVisited(cur_pos + XYZPoint16(dxy.x, dxy.y, 0), exit_edge)) return RVD_NO_VISIT; // Ride cannot be entered here.
408 
409  RideVisitDesire rvd = this->WantToVisit(ri);
410  if ((rvd == RVD_MAY_VISIT || rvd == RVD_MUST_VISIT) && this->ride == nullptr) {
411  /* Decided to want to visit one ride, and no wanted ride yet. */
412  // \todo Add a timeout so a guest gets bored waiting for the ride at some point.
413  this->ride = ri;
414  *seen_wanted_ride = true;
415  return RVD_MUST_VISIT;
416  }
417  return rvd;
418 }

471 uint8 Guest::GetExitDirections(const Voxel *v, TileEdge start_edge, bool *seen_wanted_ride, bool *queue_path)
472 {
473  assert(HasValidPath(v));
474 
475  /* If walking on a queue path, enable queue mode. */
476  // \todo Only walk in queue mode when going to a ride.
477  *queue_path = _sprite_manager.GetPathStatus(GetPathType(v->GetInstanceData())) == PAS_QUEUE_PATH;
478  *seen_wanted_ride = false;
479 
480  uint8 shops = 0; // Number of exits with a shop with normal desire to go there.
481  uint8 must_shops = 0; // Shops with a high desire to visit.
482 
483  uint8 exits = GetPathExits(v);
484  uint8 bot_exits = exits & 0x0F; // Exits at the bottom of the voxel.
485  uint8 top_exits = (exits >> 4) & 0x0F; // Exits at the top of the voxel.
486 
487  /* Being at a path tile, make extra sure we don't leave the path. */
488  for (TileEdge exit_edge = EDGE_BEGIN; exit_edge != EDGE_COUNT; exit_edge++) {
489  int extra_z; // Decide z position of the exit.
490  if (GB(bot_exits, exit_edge, 1) != 0) {
491  extra_z = 0;
492  } else if (GB(top_exits, exit_edge, 1) != 0) {
493  extra_z = 1;
494  } else {
495  continue;
496  }
497 
498  RideVisitDesire rvd = ComputeExitDesire(start_edge, this->vox_pos + XYZPoint16(0, 0, extra_z), exit_edge, seen_wanted_ride);
499  switch (rvd) {
500  case RVD_NO_RIDE:
501  break; // A path is one of the options.
502 
503  case RVD_NO_VISIT: // No desire to visit this exit; clear it.
504  SB(bot_exits, exit_edge, 1, 0);
505  SB(top_exits, exit_edge, 1, 0);
506  break;
507 
508  case RVD_MAY_VISIT: // It's one of the options (since the person is not coming from it).
509  SB(shops, exit_edge, 1, 1);
510  break;
511 
512  case RVD_MUST_VISIT: // It is very desirable to visit this shop (since the person is not coming from it).
513  SB(must_shops, exit_edge, 1, 1);
514  break;
515 
516  default: NOT_REACHED();
517  }
518  }
519  bot_exits |= top_exits; // Difference between bottom-exit and top-exit is not relevant any more.
520  if (must_shops != 0) { // If there are shops that must be visited, ignore everything else.
521  bot_exits &= must_shops;
522  shops = must_shops;
523  }
524  return (shops << 4) | bot_exits;
525 }

631  if (this->activity == GA_ENTER_PARK && vs->owner == OWN_PARK) {
632  // \todo Pay the park fee, go home if insufficient monies.
633  this->activity = GA_WANDER;
634  // Add some happiness?? (Somewhat useless as every guest enters the park. On the other hand, a nice point to configure difficulty level perhaps?)
635  }
636 
637  /* Find feasible exits and shops. */
638  uint8 exits, shops;
639  bool queue_path;
640  if (HasValidPath(v)) {
641  bool seen_wanted_ride;
642  exits = GetExitDirections(v, start_edge, &seen_wanted_ride, &queue_path);
643  shops = exits >> 4;
644  exits &= 0xF;
645 
646  if (!seen_wanted_ride) this->ride = nullptr; // Wanted ride has gone missing, stop looking for it.
647  } else { // Not at a path -> lost.
648  exits = 0xF;
649  shops = 0;
650  queue_path = false;
651  this->ride = nullptr;
652  }
653 
654  /* Switch between wandering and queuing depending on being on a queue path and having a desired ride. */
655  if (this->activity == GA_WANDER) {
656  if (queue_path && this->ride != nullptr) {
657  this->activity = GA_QUEUING;
658  } else {
659  queue_path = false;
660  }
661  } else if (this->activity == GA_QUEUING) {
662  if (this->ride == nullptr) {
663  this->activity = GA_WANDER;
664  queue_path = false;
665  }
666  }
667 
668  if (this->activity == GA_WANDER || this->activity == GA_QUEUING) { // Prevent wandering and queuing guests from walking out the park.
669  uint8 exits_viable = this->GetInparkDirections();
670  exits &= exits_viable;
671  shops &= exits_viable;
672  }
673 
674  /* Decide which direction to go. */
675  SB(exits, start_edge, 1, 0); // Drop 'return' option until we find there are no other directions.
676  uint8 walk_count = 0;
677  uint8 shop_count = 0;
678  for (TileEdge exit_edge = EDGE_BEGIN; exit_edge != EDGE_COUNT; exit_edge++) {
679  if (GB(exits, exit_edge, 1) == 0) continue;
680  walk_count++;
681  if (GB(shops, exit_edge, 1) != 0) shop_count++;
682  }
683  /* No exits, or all normal shops: Add 'return' as option. */
684  if (walk_count == 0 || (walk_count == shop_count && this->ride == nullptr)) SB(exits, start_edge, 1, 1);
685 
686  const WalkInformation *walks[4]; // Walks that can be done at this tile.
687  walk_count = 0;
688  for (TileEdge exit_edge = EDGE_BEGIN; exit_edge != EDGE_COUNT; exit_edge++) {
689  if (GB(exits, exit_edge, 1) != 0) {
690  if (GB(shops, exit_edge, 1) != 0) { // Moving to a shop.
691  walks[walk_count++] = _center_path_tile[start_edge][exit_edge];
692  } else if (queue_path) { // Queue path, walk at the center.
693  walks[walk_count++] = _center_path_tile[start_edge][exit_edge];
694  } else {
695  walks[walk_count++] = _walk_path_tile[start_edge][exit_edge];
696  }
697  }
698  }
699 
700  const WalkInformation *new_walk;
701  if (walk_count == 1) {
702  new_walk = walks[0];
703  } else {
704  new_walk = this->WalkForActivity(walks, walk_count, exits);
705  }
706

809 AnimateResult Person::OnAnimate(int delay)
810 {
811  this->frame_time -= delay;
812  if (this->frame_time > 0) return OAR_OK;
813 
814  this->MarkDirty(); // Marks the entire voxel dirty, which should be big enough even after moving.
815 
816  if (this->frames == nullptr || this->frame_count == 0) return OAR_REMOVE;
817 
818  int16 x_limit = -1;
819  switch (GB(this->walk->limit_type, WLM_X_START, WLM_LIMIT_LENGTH)) {
820  case WLM_MINIMAL: x_limit = 0; break;
821  case WLM_LOW: x_limit = 128 - this->offset; break;
822  case WLM_CENTER: x_limit = 128; break;
823  case WLM_HIGH: x_limit = 128 + this->offset; break;
824  case WLM_MAXIMAL: x_limit = 255; break;
825  }
826 
827  int16 y_limit = -1;
828  switch (GB(this->walk->limit_type, WLM_Y_START, WLM_LIMIT_LENGTH)) {
829  case WLM_MINIMAL: y_limit = 0; break;
830  case WLM_LOW: y_limit = 128 - this->offset; break;
831  case WLM_CENTER: y_limit = 128; break;
832  case WLM_HIGH: y_limit = 128 + this->offset; break;
833  case WLM_MAXIMAL: y_limit = 255; break;
834  }
835 
836  uint16 index = this->frame_index;
837  const AnimationFrame *frame = &this->frames[index];
838  this->pix_pos.x += frame->dx;
839  this->pix_pos.y += frame->dy;
840 
841  bool reached = false; // Set to true when we are beyond the limit!
842  if ((this->walk->limit_type & (1 << WLM_END_LIMIT)) == WLM_X_COND) {
843  if (frame->dx > 0) reached |= this->pix_pos.x > x_limit;
844  if (frame->dx < 0) reached |= this->pix_pos.x < x_limit;
845 
846  if (y_limit >= 0) this->pix_pos.y += sign(y_limit - this->pix_pos.y); // Also slowly move the other axis in the right direction.
847  } else {
848  if (frame->dy > 0) reached |= this->pix_pos.y > y_limit;
849  if (frame->dy < 0) reached |= this->pix_pos.y < y_limit;
850 
851  if (x_limit >= 0) this->pix_pos.x += sign(x_limit - this->pix_pos.x); // Also slowly move the other axis in the right direction.
852  }
853 
854  if (!reached) {
855  /* Not reached the end, do the next frame. */
856  index++;
857  if (this->frame_count <= index) index = 0;
858  this->frame_index = index;
859  this->frame_time = this->frames[index].duration;
860 
861  this->pix_pos.z = GetZHeight(this->vox_pos, this->pix_pos.x, this->pix_pos.y);
862  return OAR_OK;
863  }
864 
865  /* Reached the goal, start the next walk. */
866  if (this->walk[1].anim_type != ANIM_INVALID) {
867  this->StartAnimation(this->walk + 1);
868  return OAR_OK;
869  }
870 
871  /* Not only the end of this walk, but the end of the entire walk at the tile. */
872  int dx = 0;
873  int dy = 0;
874  int dz = 0;
875  TileEdge exit_edge = INVALID_EDGE;
876 
877  this->RemoveSelf(_world.GetCreateVoxel(this->vox_pos, false));
878  if (this->pix_pos.x < 0) {
879  dx--;
880  this->vox_pos.x--;
881  this->pix_pos.x += 256;
882  exit_edge = EDGE_NE;
883  } else if (this->pix_pos.x > 255) {
884  dx++;
885  this->vox_pos.x++;
886  this->pix_pos.x -= 256;
887  exit_edge = EDGE_SW;
888  }
889  if (this->pix_pos.y < 0) {
890  dy--;
891  this->vox_pos.y--;
892  this->pix_pos.y += 256;
893  exit_edge = EDGE_NW;
894  } else if (this->pix_pos.y > 255) {
895  dy++;
896  this->vox_pos.y++;
897  this->pix_pos.y -= 256;
898  exit_edge = EDGE_SE;
899  }
900  assert(this->pix_pos.x >= 0 && this->pix_pos.x < 256);
901  assert(this->pix_pos.y >= 0 && this->pix_pos.y < 256);
902 
903  AnimateResult ar = this->EdgeOfWorldOnAnimate();
904  if (ar != OAR_CONTINUE) return ar;
905 
906  /* Handle raising of z position. */
907  if (this->pix_pos.z > 128) {
908  dz++;
909  this->vox_pos.z++;
910  this->pix_pos.z = 0;
911  }
912  /* At bottom of the voxel. */
913  Voxel *v = _world.GetCreateVoxel(this->vox_pos, false);
914  if (v != nullptr) {
915  SmallRideInstance instance = v->GetInstance();
916  if (instance >= SRI_FULL_RIDES) {
917  assert(exit_edge != INVALID_EDGE);
918  RideInstance *ri = _rides_manager.GetRideInstance(instance);
919  AnimateResult ar = this->VisitRideOnAnimate(ri, exit_edge);
920  if (ar != OAR_CONTINUE) return ar;
921 
922  /* Ride is could not be visited, fall-through to reversing movement. */
923 
924  } else if (HasValidPath(v)) {
925  this->AddSelf(v);
926  this->DecideMoveDirection();
927  return OAR_OK;
928 
929  } else if (this->vox_pos.z > 0) { // Maybe a path below this voxel?
930  dz--;
931  this->vox_pos.z--;
932  this->pix_pos.z = 255;
933  Voxel *w = _world.GetCreateVoxel(this->vox_pos, false);
934  if (w != nullptr && HasValidPath(w)) {
935  this->AddSelf(w);
936  this->DecideMoveDirection();
937  return OAR_OK;
938  }
939  }
940 
941  /* Restore the person at the previous tile (ie reverse movement). */
942  if (dx != 0) { this->vox_pos.x -= dx; this->pix_pos.x = (dx > 0) ? 255 : 0; }
943  if (dy != 0) { this->vox_pos.y -= dy; this->pix_pos.y = (dy > 0) ? 255 : 0; }
944  if (dz != 0) { this->vox_pos.z -= dz; this->pix_pos.z = (dz > 0) ? 255 : 0; }
945 
946  this->AddSelf(_world.GetCreateVoxel(this->vox_pos, false));
947  this->DecideMoveDirection();
948  return OAR_OK;
949  }
950  /* No voxel here, try one level below. */
951  if (this->vox_pos.z > 0) {
952  dz--;
953  this->vox_pos.z--;
954  this->pix_pos.z = 255;
955  v = _world.GetCreateVoxel(this->vox_pos, false);
956  }
957  if (v != nullptr && HasValidPath(v)) {
958  this->AddSelf(v);
959  this->DecideMoveDirection();
960  return OAR_OK;
961  }
962  return OAR_DEACTIVATE; // We are truly lost now.
963 }
